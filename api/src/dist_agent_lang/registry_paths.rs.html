1//! Path and name validation for the `dal-registry` HTTP server (`src/bin/registry_server.rs`).
2//! Kept in the library so unit tests and mutation testing can target this logic without the binary.
3
4use std::path::Path;
5
6/// `true` if `s` is a safe package name segment for storage keys (no traversal, allowed charset).
7pub fn sanitize_name(s: &str) -> bool {
8    !s.is_empty()
9        && !s.contains("..")
10        && s.chars().all(|c| {
11            c.is_alphanumeric() || c == '/' || c == '@' || c == '-' || c == '_' || c == '.'
12        })
13}
14
15/// `true` if `s` is a safe version string for filenames (semver-ish, no traversal).
16pub fn sanitize_version(s: &str) -> bool {
17    !s.is_empty()
18        && !s.contains("..")
19        && s.chars()
20            .all(|c| c.is_alphanumeric() || c == '.' || c == '-' || c == '_')
21}

23/// Verify `path` resolves under `storage` (prevents path traversal). Handles non-existent paths
24/// by walking up to the first canonicalized ancestor.
25pub fn path_under_storage(path: &Path, storage: &Path) -> bool {
26    let storage_canon = match storage.canonicalize() {
27        Ok(p) => p,
28        Err(_) => return false,
29    };
30    if let Ok(path_canon) = path.canonicalize() {
31        return path_canon.starts_with(&storage_canon);
32    }
33    let mut current = path;
34    while let Some(parent) = current.parent() {
35        if let Ok(parent_canon) = parent.canonicalize() {
36            return parent_canon.starts_with(&storage_canon);
37        }
38        current = parent;
39    }
40    false
41}

43/// Map package name to a single filesystem directory segment (e.g. `@dal/testing` → `_at_dal_slash_testing`).
44pub fn name_to_storage_dir(name: &str) -> String {
45    name.replace('@', "_at_").replace('/', "_slash_")
