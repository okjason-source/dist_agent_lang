58impl Drop for IdeSseActiveGuard {
59    fn drop(&mut self) {
60        match self.endpoint {
61            IdeSseEndpoint::Run => {
62                IDE_SSE_RUN_STREAM_ACTIVE.fetch_sub(1, Ordering::Relaxed);
63            }
64            IdeSseEndpoint::Events => {
65                IDE_SSE_EVENTS_STREAM_ACTIVE.fetch_sub(1, Ordering::Relaxed);
66            }
67        }
68    }
69}

79            IDE_SSE_EVENTS_STREAM_CONNECTIONS_TOTAL.fetch_add(1, Ordering::Relaxed);
80        }
81    }
82    IdeSseActiveGuard { endpoint }

146/// Initialize `tracing` subscriber and bridge `log::` macros into tracing (`tracing-log`).
147/// Safe to call multiple times; only the first call has effect.
148pub fn init_tracing() {
149    TRACING_INIT.call_once(|| {
150        if try_init_tracing_inner().is_err() {
151            // Another subscriber may already be installed (e.g. tests) — continue without panic.
152        }
153    });
154}
155
156fn try_init_tracing_inner() -> Result<(), Box<dyn std::error::Error + Send + Sync + 'static>> {
157    let _ = tracing_log::LogTracer::init();
158
159    use tracing_subscriber::{fmt, prelude::*, EnvFilter};
160
161    let filter = EnvFilter::try_from_default_env().unwrap_or_else(|_| {
162        EnvFilter::new("warn,dist_agent_lang=info,tower_http=info,security=warn,dal_http=info")
163    });
164
165    let fmt_layer = if std::env::var("DAL_LOG_FORMAT").unwrap_or_default() == "json" {
166        fmt::layer().json().boxed()
167    } else {
168        fmt::layer().boxed()
169    };
170
171    tracing_subscriber::registry()
172        .with(filter)
173        .with(fmt_layer)
174        .try_init()?;
175    Ok(())
176}

260fn record_http_response(status: u16) {
261    HTTP_REQUESTS_TOTAL.fetch_add(1, Ordering::Relaxed);
262    if (400..500).contains(&status) {
263        HTTP_STATUS_4XX.fetch_add(1, Ordering::Relaxed);
264        if status == 401 || status == 403 {
265            HTTP_AUTH_REJECTIONS_TOTAL.fetch_add(1, Ordering::Relaxed);
266        }
267        if status == 429 {
268            HTTP_RATE_LIMIT_REJECTIONS_TOTAL.fetch_add(1, Ordering::Relaxed);
269        }
270    } else if status >= 500 {
271        HTTP_STATUS_5XX.fetch_add(1, Ordering::Relaxed);
272    }
273}

275/// Axum handler: returns Prometheus text if `DAL_METRICS=1`, else 404.
276pub async fn metrics_http_response() -> impl IntoResponse {
277    if std::env::var("DAL_METRICS").ok().as_deref() != Some("1") {
278        return (StatusCode::NOT_FOUND, Body::empty()).into_response();
279    }
280    (
281        StatusCode::OK,
282        [(
283            header::CONTENT_TYPE,
284            "text/plain; version=0.0.4; charset=utf-8",
285        )],
286        prometheus_metrics_text(),
287    )
288        .into_response()
289}

291/// Logs one structured `tracing` event per response and updates counters (when enabled).
292pub async fn http_observability_middleware(request: Request<Body>, next: Next) -> Response {
293    let start = Instant::now();
294    let method = request.method().as_str().to_string();
295    let path = request.uri().path().to_owned();
296
297    let response = next.run(request).await;
298    let status = response.status().as_u16();
299    let ms = start.elapsed().as_millis() as u64;
300
301    record_http_response(status);
302
303    tracing::info!(
304        target: "dal_http",
305        method = %method,
306        path = %path,
307        status = status,
308        latency_ms = ms,
309        "http_request"
310    );
311
312    response
313}
