1//! DAL stdlib: lexical RAG prompt augmentation (CEO / `dal serve` apps).
2//! See `docs/development/RAG_MVP_SPEC.md` and `crate::rag_retrieval`.
3
4use crate::agent_context_schema::ContextBlock;
5
6/// Text for one `## Context` section, or empty if RAG disabled / no index / no hits.
7pub fn prompt_block(query: &str, include_rag: Option<bool>) -> String {
8    let blocks = crate::rag_retrieval::rag_context_blocks_for_query(query, include_rag);
9    blocks
10        .into_iter()
11        .find(|b| b.source == "rag")
12        .map(|ContextBlock { content, .. }| content)
13        .unwrap_or_default()
14}
15
16/// Full `## Context (retrieved documentation)` block with trailing `---`, or empty when there is nothing to inject.
17/// Used by `workflow::run_steps` and matches CEO `enrich_prompt` / agent RAG wrapping.
18pub fn prompt_section_prefix(query: &str, include_rag: Option<bool>) -> String {
19    let inner = prompt_block(query, include_rag);
20    if inner.is_empty() {
21        return String::new();
22    }
23    format!("## Context (retrieved documentation)\n\n{inner}\n\n---\n\n")
24}
