1//! **`schedule::`** — ergonomic scheduling on top of [`crate::stdlib::time`] and [`crate::stdlib::scatter`].
2//!
3//! Use this module when you want **one namespace** for “when does work fire?” instead of splitting
4//! between `time::` (parse / compare) and `scatter::` (arm timers).
5//!
6//! **Semantics:** identical to Scatter — **process-local** only. Jobs exist while the interpreter
7//! process runs and something drains [`pending`]. For **durable** tasks across restarts, use an
8//! application store (e.g. COO `POST /api/schedule`).
9
10use crate::runtime::values::Value;
11
12const MS_PER_SEC: i64 = 1_000;
13const MS_PER_MIN: i64 = 60_000;
14const MS_PER_HOUR: i64 = 3_600_000;
15
16/// Max one-shots in [`series_interval_unix_ms`] / [`series_interval_from_now_ms`] (safety cap).
17pub const MAX_SERIES_COUNT: i64 = 10_000;
18
19/// One-shot after `delay_ms` from now (delegates to [`scatter::after_ms`]).
20pub fn once_after_ms(id: &str, delay_ms: i64) -> Result<(), String> {
21    crate::stdlib::scatter::after_ms(delay_ms, id)
22}
23
24/// One-shot at wall-clock Unix ms (delegates to [`scatter::after_at_unix_ms`]).
25pub fn once_at_unix_ms(id: &str, unix_ms: i64) -> Result<(), String> {
26    crate::stdlib::scatter::after_at_unix_ms(unix_ms, id)
27}
28
29/// Parse RFC3339 / naive UTC (see [`time::parse_rfc3339_or_naive_utc_unix_ms`]) then schedule once.
30pub fn once_at_rfc3339(id: &str, rfc3339: &str) -> Result<(), String> {
31    let ms = crate::stdlib::time::parse_rfc3339_or_naive_utc_unix_ms(rfc3339)?;
32    once_at_unix_ms(id, ms)
33}

35/// Repeating every `seconds` (> 0).
36pub fn every_seconds(id: &str, seconds: i64) -> Result<(), String> {
37    if seconds <= 0 {
38        return Err("schedule::every_seconds: interval must be > 0".into());
39    }
40    let ms = seconds
41        .checked_mul(MS_PER_SEC)
42        .ok_or_else(|| "schedule::every_seconds: overflow".to_string())?;
43    crate::stdlib::scatter::every_ms(ms, id)
44}

68/// `count` one-shots at `start_unix_ms + i * interval_ms` for `i in 0..count`.
69/// Job ids: `{prefix}_0` … `{prefix}_{count-1}`.
70pub fn series_interval_unix_ms(
71    prefix: &str,
72    start_unix_ms: i64,
73    interval_ms: i64,
74    count: i64,
75) -> Result<Vec<String>, String> {
76    if prefix.is_empty() {
77        return Err("schedule::series_interval_unix_ms: prefix must not be empty".into());
78    }
79    if count <= 0 {
80        return Err("schedule::series_interval_unix_ms: count must be > 0".into());
81    }
82    if count > MAX_SERIES_COUNT {
83        return Err(format!(
84            "schedule::series_interval_unix_ms: count must be <= {}",
85            MAX_SERIES_COUNT
86        ));
87    }
88    if interval_ms < 0 {
89        return Err("schedule::series_interval_unix_ms: interval_ms must be >= 0".into());
90    }
91    let mut ids = Vec::with_capacity(count as usize);
92    for i in 0..count {
93        let id = format!("{}_{}", prefix, i);
94        let step = i
95            .checked_mul(interval_ms)
96            .ok_or_else(|| "schedule::series_interval_unix_ms: overflow".to_string())?;
97        let t = start_unix_ms
98            .checked_add(step)
99            .ok_or_else(|| "schedule::series_interval_unix_ms: overflow".to_string())?;
100        crate::stdlib::scatter::after_at_unix_ms(t, &id)?;
101        ids.push(id);
102    }
103    Ok(ids)
104}

106/// Same as [`series_interval_unix_ms`] with `start_unix_ms = time::unix_ms_now()`.
107pub fn series_interval_from_now_ms(
108    prefix: &str,
109    interval_ms: i64,
110    count: i64,
111) -> Result<Vec<String>, String> {
112    series_interval_unix_ms(
113        prefix,
114        crate::stdlib::time::unix_ms_now(),
115        interval_ms,
116        count,
117    )
118}

148#[cfg(test)]
149mod tests {
150    use super::*;
151    use crate::stdlib::scatter;
152
153    #[test]
154    fn series_ids_distinct() {
155        let _lock = scatter::test_lock();
156        scatter::reset_for_test();
157        let ids = series_interval_unix_ms("p", 1_700_000_000_000, 3_600_000, 3).unwrap();
158        assert_eq!(ids, vec!["p_0", "p_1", "p_2"]);
159        scatter::reset_for_test();
160    }
161
162    #[test]
163    fn series_rejects_bad_count() {
164        let _lock = scatter::test_lock();
165        scatter::reset_for_test();
166        assert!(series_interval_unix_ms("p", 0, 1, 0).is_err());
167        assert!(series_interval_unix_ms("", 0, 1, 1).is_err());
168        scatter::reset_for_test();
169    }
170
171    #[test]
172    fn every_minutes_delegates() {
173        let _lock = scatter::test_lock();
174        scatter::reset_for_test();
175        every_minutes("m", 60).unwrap();
176        assert!(scatter::scheduled_count() >= 1);
177        scatter::reset_for_test();
178    }
