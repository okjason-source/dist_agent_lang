10/// Result of running a fleet (dispatching last_deployed_task to each member).
11#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
12pub struct RunReport {
13    pub fleet_name: String,
14    pub members_dispatched: u32,
15    pub errors: Vec<String>,

18/// A fleet is a named set of agent IDs. Optionally created from a mold (one mold + N instances).
19#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
20pub struct Fleet {
21    pub name: String,
22    /// If created from mold, the mold source (path or ipfs://cid).
23    pub mold_source: Option<String>,
24    /// Agent IDs in this fleet (order preserved).
25    pub member_ids: Vec<String>,
26    /// Last task deployed to this fleet (set by deploy). Runners/automation read this to know what to run.
27    #[serde(default, skip_serializing_if = "Option::is_none")]
28    pub last_deployed_task: Option<String>,
29    /// When last_deployed_task was set (ISO-ish or opaque string).
30    #[serde(default, skip_serializing_if = "Option::is_none")]
31    pub last_deployed_at: Option<String>,
32    /// Params used when creating from mold (and when scaling up). Reused by scale() for new members.
33    #[serde(default, skip_serializing_if = "Option::is_none")]
34    pub last_create_params: Option<HashMap<String, String>>,
35}
36
37static FLEETS: OnceLock<Mutex<HashMap<String, Fleet>>> = OnceLock::new();
38
39fn fleets() -> std::sync::MutexGuard<'static, HashMap<String, Fleet>> {
40    FLEETS
41        .get_or_init(|| Mutex::new(HashMap::new()))
42        .lock()
43        .unwrap()
44}
45
46fn path_for_fleets(base: &Path) -> PathBuf {
47    base.join(".dal").join("fleets.json")
48}
49
50fn load_from_file(base: &Path) -> HashMap<String, Fleet> {
51    let p = path_for_fleets(base);
52    if let Ok(data) = fs::read_to_string(&p) {
53        if let Ok(map) = serde_json::from_str::<HashMap<String, Fleet>>(&data) {
54            return map;
55        }
56    }
57    HashMap::new()
58}
59
60fn save_to_file(base: &Path, map: &HashMap<String, Fleet>) -> Result<(), String> {
61    let p = path_for_fleets(base);
62    if let Some(parent) = p.parent() {
63        fs::create_dir_all(parent).map_err(|e| format!("create .dal: {}", e))?;
64    }
65    let data = serde_json::to_string_pretty(map).map_err(|e| format!("serialize fleets: {}", e))?;
66    fs::write(&p, data).map_err(|e| format!("write {}: {}", p.display(), e))?;
67    Ok(())
68}
69
70fn ensure_loaded(base: Option<&Path>) {
71    if let Some(b) = base {
72        let loaded = load_from_file(b);
73        let mut f = fleets();
74        *f = loaded;
75    }
76}
77
78fn save_if_base(base: Option<&Path>) -> Result<(), String> {
79    if let Some(b) = base {
80        let f = fleets();
81        save_to_file(b, &f)?;
82    }
83    Ok(())
84}

86/// Create an empty fleet with the given name. Fails if name already exists.
87/// If base is Some, fleet data is persisted to base/.dal/fleets.json.
88pub fn create(name: &str, base: Option<&Path>) -> Result<(), String> {
89    ensure_loaded(base);
90    let name = name.to_string();
91    if name.is_empty() {
92        return Err("fleet name cannot be empty".to_string());
93    }
94    let mut f = fleets();
95    if f.contains_key(&name) {
96        return Err(format!("fleet '{}' already exists", name));
97    }
98    f.insert(
99        name.clone(),
100        Fleet {
101            name,
102            mold_source: None,
103            member_ids: Vec::new(),
104            last_deployed_task: None,
105            last_deployed_at: None,
106            last_create_params: None,
107        },
108    );
109    drop(f);
110    save_if_base(base)
111}

113/// Create a fleet by spawning N agents from a mold. Same principal/mold/fallback rules as single create.
114/// Each agent is named `<fleet_name>_0`, `<fleet_name>_1`, ...; params apply to each.
115/// Fleet is persisted to base/.dal/fleets.json.
116pub fn create_from_mold(
117    name: &str,
118    mold_source: &str,
119    count: u32,
120    base: &Path,
121    params: Option<&HashMap<String, String>>,
122) -> Result<(), String> {
123    if count == 0 {
124        return Err("fleet from mold requires count >= 1".to_string());
125    }
126    if count > 1000 {
127        return Err("fleet from mold count capped at 1000".to_string());
128    }
129    ensure_loaded(Some(base));
130    let name = name.to_string();
131    let f = fleets();
132    if f.contains_key(&name) {
133        return Err(format!("fleet '{}' already exists", name));
134    }
135    drop(f);
136
137    let mut member_ids = Vec::with_capacity(count as usize);
138    for i in 0..count {
139        let agent_name = format!("{}_{}", name, i);
140        let ctx = crate::mold::create_from_mold_source(
141            mold_source,
142            base,
143            Some(agent_name.as_str()),
144            params,
145        )
146        .map_err(|e| format!("spawn agent {} from mold: {}", i, e))?;
147        member_ids.push(ctx.agent_id);
148    }
149
150    let last_create_params =
151        params.map(|p| p.iter().map(|(k, v)| (k.clone(), v.clone())).collect());
152    let mut f = fleets();
153    f.insert(
154        name.clone(),
155        Fleet {
156            name,
157            mold_source: Some(mold_source.to_string()),
158            member_ids,
159            last_deployed_task: None,
160            last_deployed_at: None,
161            last_create_params,
162        },
163    );
164    drop(f);
165    save_if_base(Some(base))
166}

168/// Add N members to an existing fleet from a mold. If the fleet is empty, sets mold_source.
169/// Use this to populate an empty fleet or add more members (same as scale-up logic but explicit).
170pub fn add_from_mold(
171    name: &str,
172    mold_source: &str,
173    count: u32,
174    base: &Path,
175    params: Option<&HashMap<String, String>>,
176) -> Result<(), String> {
177    if count == 0 {
178        return Err("add_from_mold requires count >= 1".to_string());
179    }
180    if count > 1000 {
181        return Err("add_from_mold count capped at 1000".to_string());
182    }
183    ensure_loaded(Some(base));
184    let f = fleets();
185    let mut fleet = f
186        .get(name)
187        .cloned()
188        .ok_or_else(|| format!("fleet '{}' not found", name))?;
189    let start = fleet.member_ids.len();
190    let last_create_params =
191        params.map(|p| p.iter().map(|(k, v)| (k.clone(), v.clone())).collect());
192    if fleet.mold_source.is_none() {
193        fleet.mold_source = Some(mold_source.to_string());
194    }
195    if fleet.last_create_params.is_none() {
196        fleet.last_create_params = last_create_params.clone();
197    }
198    drop(f);
199
200    let mut member_ids = fleet.member_ids;
201    for i in 0..count {
202        let agent_name = format!("{}_{}", name, start + i as usize);
203        let ctx = crate::mold::create_from_mold_source(
204            mold_source,
205            base,
206            Some(agent_name.as_str()),
207            params,
208        )
209        .map_err(|e| format!("add member {} from mold: {}", i, e))?;
210        member_ids.push(ctx.agent_id);
211    }
212
213    let mut f = fleets();
214    if let Some(fleet) = f.get_mut(name) {
215        fleet.member_ids = member_ids;
216        fleet
217            .mold_source
218            .get_or_insert_with(|| mold_source.to_string());
219        if fleet.last_create_params.is_none() {
220            fleet.last_create_params = last_create_params;
221        }
222    }
223    drop(f);
224    save_if_base(Some(base))
225}

227/// Add a single agent (by ID) to an existing fleet. Agent must already exist elsewhere; this only registers the ID.
228pub fn add_member(name: &str, agent_id: &str, base: Option<&Path>) -> Result<(), String> {
229    if agent_id.trim().is_empty() {
230        return Err("agent_id cannot be empty".to_string());
231    }
232    ensure_loaded(base);
233    let base_path = base.ok_or("add_member requires a base path for persistence")?;
234    let mut f = fleets();
235    let fleet = f
236        .get_mut(name)
237        .ok_or_else(|| format!("fleet '{}' not found", name))?;
238    if fleet.member_ids.contains(&agent_id.to_string()) {
239        drop(f);
240        return Ok(());
241    }
242    fleet.member_ids.push(agent_id.to_string());
243    drop(f);
244    save_if_base(Some(base_path))
245}

247/// Record a deployment (task) for a fleet. Updates last_deployed_task and last_deployed_at; persists if base is set.
248/// Runners/automation can read the fleet file to see what to execute for each fleet.
249pub fn deploy(name: &str, task: &str, base: Option<&Path>) -> Result<(), String> {
250    if task.trim().is_empty() {
251        return Err("deploy task cannot be empty".to_string());
252    }
253    ensure_loaded(base);
254    let base_path = base.ok_or("deploy requires a base path for persistence")?;
255    let mut f = fleets();
256    let mut fleet = f
257        .get(name)
258        .cloned()
259        .ok_or_else(|| format!("fleet '{}' not found", name))?;
260    fleet.last_deployed_task = Some(task.trim().to_string());
261    fleet.last_deployed_at = Some(
262        std::time::SystemTime::now()
263            .duration_since(std::time::UNIX_EPOCH)
264            .map(|d| d.as_secs().to_string())
265            .unwrap_or_else(|_| "0".to_string()),
266    );
267    f.insert(name.to_string(), fleet);
268    drop(f);
269    save_if_base(Some(base_path))
270}

288/// Remove a fleet (agents remain in runtime; only the grouping is removed). Returns true if removed.
289/// If base is Some, persists after remove.
290pub fn delete(name: &str, base: Option<&Path>) -> Result<bool, String> {
291    ensure_loaded(base);
292    let mut f = fleets();

301/// Resize fleet to N members. If base is Some, persists after change.
302/// - Scale down (N < current): truncates member list to N (agent contexts are not removed).
303/// - Scale up (N > current): only allowed when fleet has mold_source; spawns (N - current) new agents from the same mold and appends to members.
304/// - Empty fleets (no mold_source) cannot be scaled up.
305pub fn scale(name: &str, n: u32, base: Option<&Path>) -> Result<(), String> {
306    if n > 1000 {
307        return Err("fleet scale count capped at 1000".to_string());
308    }
309    ensure_loaded(base);
310    let base_path = base.ok_or("scale requires a base path for persistence")?;
311
312    let mut f = fleets();
313    let fleet = f
314        .get(name)
315        .cloned()
316        .ok_or_else(|| format!("fleet '{}' not found", name))?;
317    let current = fleet.member_ids.len() as u32;
318
319    if n == current {
320        drop(f);
321        return Ok(());
322    }
323
324    if n < current {
325        let mut fleet = fleet;
326        fleet.member_ids.truncate(n as usize);
327        f.insert(name.to_string(), fleet);
328        drop(f);
329        return save_if_base(Some(base_path));
330    }
331
332    // n > current: scale up
333    let mold_source = fleet.mold_source.as_deref().ok_or_else(|| {
334        format!(
335            "fleet '{}' has no mold; cannot scale up (create from mold first)",
336            name
337        )
338    })?;
339
340    let mut member_ids = fleet.member_ids;
341    drop(f);
342
343    let params_ref = fleet.last_create_params.as_ref();
344    for i in current..n {
345        let agent_name = format!("{}_{}", name, i);
346        let ctx = crate::mold::create_from_mold_source(
347            mold_source,
348            base_path,
349            Some(agent_name.as_str()),
350            params_ref,
351        )
352        .map_err(|e| format!("spawn agent {} from mold: {}", i, e))?;
353        member_ids.push(ctx.agent_id);
354    }
355
356    let mut f = fleets();
357    if let Some(fleet) = f.get_mut(name) {
358        fleet.member_ids = member_ids;
359    }
360    drop(f);
361    save_if_base(Some(base_path))
362}

364/// Run fleets: for each fleet with last_deployed_task (and name matching filter if given), ensure it has members
365/// (if empty and has mold_source, add one from mold), then dispatch last_deployed_task to each member via
366/// agent::coordinate(agent_id, task, "task_distribution"). Returns one RunReport per fleet processed.
367pub fn run(name_filter: Option<&str>, base: &Path) -> Result<Vec<RunReport>, String> {
368    ensure_loaded(Some(base));
369    let f = fleets();
370    let names: Vec<String> = f
371        .keys()
372        .filter(|n| name_filter.map(|f| *n == f).unwrap_or(true))
373        .cloned()
374        .collect();
375    drop(f);
376
377    let mut reports = Vec::new();
378    for name in names {
379        let fleet = show(&name, Some(base)).ok_or_else(|| format!("fleet '{}' not found", name))?;
380        let Some(ref task_desc) = fleet.last_deployed_task else {
381            continue;
382        };
383        let mut member_ids = fleet.member_ids.clone();
384        if member_ids.is_empty() {
385            let mold = fleet.mold_source.as_deref().ok_or_else(|| {
386                format!(
387                    "fleet '{}' has no members and no mold_source; add members or deploy from mold first",
388                    name
389                )
390            })?;
391            add_from_mold(&name, mold, 1, base, fleet.last_create_params.as_ref())?;
392            if let Some(updated) = show(&name, Some(base)) {
393                member_ids = updated.member_ids;
394            }
395        }
396        let mut errors = Vec::new();
397        let mut dispatched = 0u32;
398        for agent_id in &member_ids {
399            let task_id = uuid::Uuid::new_v4().to_string();
400            let task = match crate::stdlib::agent::create_agent_task(
401                task_id.clone(),
402                task_desc.clone(),
403                "medium",
404            ) {
405                Some(t) => t,
406                None => {
407                    errors.push(format!(
408                        "{}: create_agent_task failed (invalid priority)",
409                        agent_id
410                    ));
411                    continue;
412                }
413            };
414            match crate::stdlib::agent::coordinate(agent_id, task, "task_distribution") {
415                Ok(true) => dispatched += 1,
416                Ok(false) => errors.push(format!("{}: coordinate returned false", agent_id)),
417                Err(e) => errors.push(format!("{}: {}", agent_id, e)),
418            }
419        }
420        reports.push(RunReport {
421            fleet_name: name,
422            members_dispatched: dispatched,
423            errors,
424        });
425    }
426    Ok(reports)
427}

429/// Health summary for a fleet (member count, mold presence, last deployment).
430#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
431pub struct FleetHealth {
432    pub name: String,
433    pub member_count: usize,
434    pub has_mold: bool,
435    #[serde(skip_serializing_if = "Option::is_none")]
436    pub last_deployed_task: Option<String>,
437    #[serde(skip_serializing_if = "Option::is_none")]
438    pub last_deployed_at: Option<String>,
439    pub status: String,
440}
441
442/// Report health for a fleet. Status is "ok" if fleet exists and has at least one member or a mold; "empty" if no members and no mold.
443pub fn health(name: &str, base: Option<&Path>) -> Result<FleetHealth, String> {
444    ensure_loaded(base);
445    let f = fleets();
446    let fleet = f
447        .get(name)
448        .cloned()
449        .ok_or_else(|| format!("fleet '{}' not found", name))?;
450    drop(f);
451    let status = if !fleet.member_ids.is_empty() || fleet.mold_source.is_some() {
452        "ok"
453    } else {
454        "empty"
455    };
456    Ok(FleetHealth {
457        name: fleet.name,
458        member_count: fleet.member_ids.len(),
459        has_mold: fleet.mold_source.is_some(),
460        last_deployed_task: fleet.last_deployed_task,
461        last_deployed_at: fleet.last_deployed_at,
462        status: status.to_string(),
463    })
464}

466/// Export format for fleet(s) to infrastructure-as-code.
467#[derive(Debug, Clone, Copy, PartialEq, Eq)]
468pub enum ExportFormat {
469    K8s,
470    DockerCompose,
471}
472
473/// Export fleet(s) to YAML (k8s Job list or docker-compose). If name is None, exports all fleets.
474pub fn export(
475    name: Option<&str>,
476    base: Option<&Path>,
477    format: ExportFormat,
478) -> Result<String, String> {
479    ensure_loaded(base);
480    let fleets_to_export: Vec<Fleet> = {
481        let f = fleets();
482        let v: Vec<Fleet> = match name {
483            Some(n) => f.get(n).cloned().into_iter().collect(),
484            None => f.values().cloned().collect(),
485        };
486        v
487    };
488    if fleets_to_export.is_empty() {
489        return Err(name
490            .map(|n| format!("fleet '{}' not found", n))
491            .unwrap_or_else(|| "no fleets to export".to_string()));
492    }
493    let refs: Vec<&Fleet> = fleets_to_export.iter().collect();
494    match format {
495        ExportFormat::K8s => export_k8s(&refs),
496        ExportFormat::DockerCompose => export_docker_compose(&refs),
497    }
498}
499
500fn export_k8s(fleets: &[&Fleet]) -> Result<String, String> {
501    let mut out = String::from("apiVersion: batch/v1\nkind: JobList\nitems:\n");
502    for fleet in fleets {
503        for (i, agent_id) in fleet.member_ids.iter().enumerate() {
504            let job_name = format!("{}-{}", fleet.name, i)
505                .chars()
506                .filter(|c| c.is_ascii_alphanumeric() || *c == '-')
507                .collect::<String>();
508            out.push_str(&format!(
509                "- apiVersion: batch/v1\n  kind: Job\n  metadata:\n    name: {}\n  spec:\n    template:\n      spec:\n        containers:\n        - name: agent\n          image: dal-agent:latest\n          env:\n          - name: DAL_AGENT_ID\n            value: \"{}\"\n        restartPolicy: Never\n",
510                job_name, agent_id
511            ));
512        }
513    }
514    Ok(out)
515}
516
517fn export_docker_compose(fleets: &[&Fleet]) -> Result<String, String> {
518    let mut out = String::from("version: \"3\"\nservices:\n");
519    for fleet in fleets {
520        for (i, agent_id) in fleet.member_ids.iter().enumerate() {
521            let svc = format!("{}-{}", fleet.name, i)
522                .chars()
523                .filter(|c| c.is_ascii_alphanumeric() || *c == '-')
524                .collect::<String>();
525            out.push_str(&format!(
526                "  {}:\n    image: dal-agent:latest\n    environment:\n      DAL_AGENT_ID: \"{}\"\n",
527                svc, agent_id
528            ));
529        }
530    }
531    Ok(out)
532}

720                .duration_since(std::time::UNIX_EPOCH)
721                .unwrap()
722                .as_millis()
723        );

735            reports[0].errors.is_empty(),

737            reports[0].errors
738        );
