28#[derive(Debug, Clone, PartialEq, Eq)]
29pub enum ServeSecurityPreset {
30    Legacy,
31    Balanced,
32    Strict,
33}
34
35impl ServeSecurityPreset {
36    pub fn from_env() -> Self {
37        let raw = std::env::var("DAL_SERVE_SECURITY_PRESET")
38            .unwrap_or_else(|_| "legacy".to_string())
39            .to_lowercase();
40        match raw.as_str() {
41            "balanced" => Self::Balanced,
42            "strict" => Self::Strict,
43            _ => Self::Legacy,
44        }
45    }
46
47    pub fn as_str(&self) -> &'static str {
48        match self {
49            Self::Legacy => "legacy",
50            Self::Balanced => "balanced",
51            Self::Strict => "strict",
52        }
53    }
54}

94impl ServeSecurityOptions {
95    pub fn from_env() -> Self {
96        let preset = ServeSecurityPreset::from_env();
97        let mut opts = match preset {
98            ServeSecurityPreset::Legacy => Self {
99                preset,
100                ..Self::default()
101            },
102            ServeSecurityPreset::Balanced => Self {
103                preset,
104                enable_security_headers: true,
105                enable_rate_limit: true,
106                enable_request_size_limit: true,
107                enable_input_validation: false,
108                enable_auth_middleware: false,
109                rate_limit_requests_per_minute: 120,
110                rate_limit_window_seconds: 60,
111                max_body_bytes: 1_000_000,
112                max_header_bytes: 8_192,
113                max_url_length: 2_048,
114                ..Self::default()
115            },
116            ServeSecurityPreset::Strict => Self {
117                preset,
118                enable_security_headers: true,
119                enable_rate_limit: true,
120                enable_request_size_limit: true,
121                enable_input_validation: true,
122                enable_auth_middleware: true,
123                rate_limit_requests_per_minute: 60,
124                rate_limit_window_seconds: 60,
125                max_body_bytes: 512_000,
126                max_header_bytes: 8_192,
127                max_url_length: 2_048,
128                ..Self::default()
129            },
130        };
131
132        if let Ok(v) = std::env::var("DAL_SERVE_ENABLE_AUTH") {
133            opts.enable_auth_middleware = parse_bool_flag(&v);
134        }
135        if let Ok(v) = std::env::var("DAL_SERVE_ENABLE_INPUT_VALIDATION") {
136            opts.enable_input_validation = parse_bool_flag(&v);
137        }
138        if let Ok(v) = std::env::var("DAL_SERVE_RATE_LIMIT_RPM") {
139            if let Ok(n) = v.parse::<usize>() {
140                if n > 0 {
141                    opts.rate_limit_requests_per_minute = n;
142                }
143            }
144        }
145        if let Ok(v) = std::env::var("DAL_SERVE_MAX_BODY_BYTES") {
146            if let Ok(n) = v.parse::<usize>() {
147                if n > 0 {
148                    opts.max_body_bytes = n;
149                }
150            }
151        }
152        opts
153    }
154}

156fn parse_bool_flag(value: &str) -> bool {
157    matches!(
158        value.trim().to_ascii_lowercase().as_str(),
159        "1" | "true" | "yes" | "on"
160    )
161}

163/// True when optional HTTP Basic Auth should be active (`dal serve` layer).
164pub fn dal_serve_basic_auth_configured() -> bool {
165    let user_ok = std::env::var("DAL_HTTP_USER")
166        .ok()
167        .filter(|s| !s.is_empty())
168        .is_some();
169    if !user_ok {
170        return false;
171    }
172    let hash_ok = std::env::var("DAL_HTTP_PASSWORD_HASH")
173        .ok()
174        .map(|s| !s.trim().is_empty())
175        .unwrap_or(false);
176    let pass_ok = std::env::var("DAL_HTTP_PASSWORD")
177        .ok()
178        .filter(|s| !s.is_empty())
179        .is_some();
180    hash_ok || pass_ok
181}

183pub fn apply_standard_http_layers(
184    router: Router,
185    cors_origin: &str,
186    options: &ServeSecurityOptions,
187) -> Router {
188    let cors = if cors_origin == "*" {
189        CorsLayer::new()
190            .allow_methods([
191                Method::GET,
192                Method::POST,
193                Method::PUT,
194                Method::DELETE,
195                Method::PATCH,
196                Method::OPTIONS,
197            ])
198            .allow_origin(Any)
199            .allow_headers(Any)
200    } else {
201        let origin = cors_origin
202            .parse::<axum::http::HeaderValue>()
203            .unwrap_or_else(|_| axum::http::HeaderValue::from_static("*"));
204        CorsLayer::new()
205            .allow_methods([
206                Method::GET,
207                Method::POST,
208                Method::PUT,
209                Method::DELETE,
210                Method::PATCH,
211                Method::OPTIONS,
212            ])
213            .allow_origin(origin)
214            .allow_headers(Any)
215    };
216
217    let mut app = router
218        .route("/metrics", get(crate::observability::metrics_http_response))
219        .layer(middleware::from_fn(
220            crate::observability::http_observability_middleware,
221        ))
222        .layer(cors);
223
224    if options.enable_security_headers {
225        app = app.layer(middleware::from_fn(security_headers_middleware));
226    }
227
228    if options.enable_rate_limit {
229        let limiter = Arc::new(RateLimiter::new(
230            options.rate_limit_requests_per_minute,
231            options.rate_limit_window_seconds,
232        ));
233        app = app
234            .layer(middleware::from_fn(rate_limit_middleware))
235            // Extension must be added last so middleware sees it first.
236            .layer(Extension(limiter));
237    }
238
239    if options.enable_request_size_limit {
240        let size = Arc::new(RequestSizeLimiter {
241            max_body_size: options.max_body_bytes,
242            max_header_size: options.max_header_bytes,
243            max_url_length: options.max_url_length,
244        });
245        app = app
246            .layer(middleware::from_fn(request_size_middleware))
247            .layer(Extension(size));
248    }
249
250    if options.enable_input_validation {
251        let exempt_paths: HashSet<String> = options
252            .public_paths_without_input_validation
253            .iter()
254            .cloned()
255            .collect();
256        app = app
257            .layer(middleware::from_fn(
258                input_validation_middleware_with_exemptions,
259            ))
260            .layer(Extension(InputValidationExemptPaths(Arc::new(
261                exempt_paths,
262            ))));
263    }
264
265    if options.enable_auth_middleware {
266        let exempt_paths: HashSet<String> =
267            options.public_paths_without_auth.iter().cloned().collect();
268        app = app
269            .layer(middleware::from_fn(auth_middleware_with_exemptions))
270            .layer(Extension(AuthExemptPaths(Arc::new(exempt_paths))));
271    }
272
273    // Optional HTTP Basic Auth when DAL_HTTP_USER + (DAL_HTTP_PASSWORD_HASH or DAL_HTTP_PASSWORD).
274    // Independent of DAL_SERVE_ENABLE_AUTH / JWT middleware.
275    if dal_serve_basic_auth_configured() {
276        app = app.layer(middleware::from_fn(dal_serve_basic_auth_middleware));
277    }
278
279    app
280}

295/// Create router with runtime factory and optional scope writeback for shared state.
296pub fn create_router_with_options<F>(
297    server: HttpServer,
298    factory: F,
299    scope_writeback: Option<Arc<Box<dyn Fn(&crate::runtime::scope::Scope) + Send + Sync>>>,
300) -> Router
301where
302    F: Fn() -> crate::runtime::engine::Runtime + Send + Sync + 'static,
303{
304    let runtime_factory = Arc::new(
305        Box::new(factory) as Box<dyn Fn() -> crate::runtime::engine::Runtime + Send + Sync>
306    );
307
308    let state = Arc::new(RwLock::new(WebServerState {
309        server: server.clone(),
310        handlers: std::collections::HashMap::new(),
311        runtime_factory: Some(runtime_factory),
312        scope_writeback,
313    }));
314
315    let mut router = Router::new();
316
317    // Add routes from server configuration
318    for (route_key, route) in &server.routes {
319        // Parse route key (format: "METHOD:/path")
320        if let Some((method_str, path)) = route_key.split_once(':') {
321            let method = method_str.to_uppercase();
322            let handler_name = route.handler.clone();
323
324            // Create handler closure
325            let state_clone = state.clone();
326            let handler = move |request: Request| {
327                let state = state_clone.clone();
328                let handler_name = handler_name.clone();
329                async move { handle_with_middleware(State(state), request, &handler_name).await }
330            };
331
332            // Add route based on method
333            match method.as_str() {
334                "GET" => {
335                    router = router.route(path, get(handler));
336                }
337                "POST" => {
338                    router = router.route(path, post(handler));
339                }
340                "PUT" => {
341                    router = router.route(path, axum::routing::put(handler));
342                }
343                "DELETE" => {
344                    router = router.route(path, axum::routing::delete(handler));
345                }
346                _ => {
347                    // Default to GET
348                    router = router.route(path, get(handler));
349                }
350            }
351        }
352    }
353
354    // Add default routes if server has no routes configured
355    if server.routes.is_empty() {
356        router = router
357            .route("/", get(home_handler))
358            .route("/health", get(health_handler));
359    }
360
361    router.with_state(state)
362}
