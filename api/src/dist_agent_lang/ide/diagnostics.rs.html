9/// A single diagnostic (error or warning).
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct Diagnostic {
12    pub line: u32,
13    pub column: u32,
14    pub end_line: u32,
15    pub end_column: u32,
16    pub message: String,
17    pub severity: String, // "error" | "warning"
18}

20/// Run lexer and parser on source, return diagnostics.
21pub fn diagnostics_from_source(source: &str) -> Vec<Diagnostic> {
22    let mut diags = Vec::new();
23
24    let lexer = Lexer::new(source);
25    match lexer.tokenize_with_positions_immutable() {
26        Err(e) => {
27            let (line, col) = e.line_column().unwrap_or((1, 1));
28            diags.push(Diagnostic {
29                line: line as u32,
30                column: col as u32,
31                end_line: line as u32,
32                end_column: (col + 1) as u32,
33                message: e.to_string(),
34                severity: "error".to_string(),
35            });
36            return diags;
37        }
38        Ok(tokens_with_pos) => {
39            let mut parser = Parser::new_with_positions(tokens_with_pos);
40            if let Err(e) = parser.parse() {
41                let line = e.line_number().unwrap_or(1);
42                let col = e.column_number().unwrap_or(1);
43                diags.push(Diagnostic {
44                    line: line as u32,
45                    column: col as u32,
46                    end_line: line as u32,
47                    end_column: (col + 1) as u32,
48                    message: e.to_string(),
49                    severity: "error".to_string(),
50                });
51            }
52        }
53    }
54
55    diags
56}

58/// Get word at 0-based line and character.
59fn word_at_position(source: &str, line_0: u32, char_0: u32) -> Option<String> {
60    let lines: Vec<&str> = source.lines().collect();
61    let line_idx = line_0 as usize;
62    let line = lines.get(line_idx)?;
63    let chars: Vec<char> = line.chars().collect();
64    let char_idx = char_0 as usize;
65    if char_idx > chars.len() {
66        return None;
67    }
68    fn is_word_char(c: char) -> bool {
69        c.is_ascii_alphanumeric() || c == '_'
70    }
71    let start = chars[..char_idx]
72        .iter()
73        .rposition(|&c| is_word_char(c))
74        .map(|i| i + 1)
75        .unwrap_or(0);
76    let end = chars[char_idx..]
77        .iter()
78        .position(|&c| !is_word_char(c))
79        .map(|i| char_idx + i)
80        .unwrap_or(chars.len());
81    if start >= end {
82        return None;
83    }
84    Some(chars[start..end].iter().collect())
85}

87fn keyword_doc(word: &str) -> Option<&'static str> {
88    Some(match word {
89        "fn" => "Function declaration. `fn name(params) -> return_type { ... }`",
90        "let" => "Bind a value to a variable.",
91        "return" => "Return a value from a function.",
92        "if" | "else" => "Conditional branch.",
93        "while" | "for" | "loop" => "Loop construct.",
94        "service" => "Service declaration. `service Name { fields; fn methods() {} }`",
95        "struct" => "Struct type definition.",
96        "import" => "Import a module. `import \"path\";` or `import \"path\" as alias;`",
97        "match" | "case" | "default" => "Pattern matching.",
98        "try" | "catch" | "throw" => "Error handling.",
99        "spawn" | "agent" => "Agent / concurrent execution.",
100        "true" | "false" => "Boolean literal.",
101        _ => return None,
102    })
103}

118}

120fn collect_symbols_from_source(source: &str) -> Vec<(String, String)> {
121    let mut symbols = Vec::new();
122    let lexer = Lexer::new(source);
123    let tokens_with_pos = match lexer.tokenize_with_positions_immutable() {
124        Ok(t) => t,
125        Err(_) => return symbols,
126    };
127    let mut parser = Parser::new_with_positions(tokens_with_pos);
128    let program = match parser.parse() {
129        Ok(p) => p,
130        Err(_) => return symbols,
131    };
132    for stmt in &program.statements {
133        match stmt {
134            Statement::Service(s) => {
135                symbols.push((s.name.clone(), format!("service {}", s.name)));
136                for f in &s.fields {
137                    symbols.push((
138                        f.name.clone(),
139                        format!("field {}: {}", f.name, f.field_type),
140                    ));
141                }
142                for m in &s.methods {
143                    let params: Vec<String> = m.parameters.iter().map(|p| p.name.clone()).collect();
144                    let sig = format!("fn {}({})", m.name, params.join(", "));
145                    symbols.push((m.name.clone(), sig));
146                }
147            }
148            Statement::Function(f) => {
149                let params: Vec<String> = f.parameters.iter().map(|p| p.name.clone()).collect();
150                let sig = format!("fn {}({})", f.name, params.join(", "));
151                symbols.push((f.name.clone(), sig));
152            }
153            _ => {}
154        }
155    }
156    symbols
157}

159/// Hover content for a word at position.
160pub fn hover_at_position(source: &str, line_0: u32, char_0: u32) -> Option<String> {
161    let word = word_at_position(source, line_0, char_0)?;
162    let word = word.trim();
163    if word.is_empty() {
164        return None;
165    }
166    if let Some(doc) = keyword_doc(word) {
167        return Some(format!("**{}**\n\n{}", word, doc));
168    }
169    if let Some(doc) = stdlib_doc(word) {
170        return Some(format!("**{}** (stdlib)\n\n{}", word, doc));
171    }
172    for (name, detail) in collect_symbols_from_source(source) {
173        if name == word {
174            return Some(format!("**{}**\n\n`{}`", name, detail));
175        }
176    }
177    None
178}

180/// Completion item for HTTP API.
181#[derive(Debug, Clone, Serialize, Deserialize)]
182pub struct CompletionItem {
183    pub label: String,
184    pub kind: String, // "keyword" | "module" | "function"
185    #[serde(skip_serializing_if = "Option::is_none")]
186    pub detail: Option<String>,
187}
188
189/// Completion at position (prefix from source).
190pub fn completion_at_position(source: &str, line_0: u32, char_0: u32) -> Vec<CompletionItem> {
191    let prefix = word_at_position(source, line_0, char_0)
192        .unwrap_or_default()
193        .trim()
194        .to_string();
195
196    let mut items = Vec::new();
197
198    let keywords = [
199        "fn", "let", "return", "if", "else", "while", "for", "in", "loop", "break", "continue",
200        "service", "struct", "import", "match", "case", "default", "try", "catch", "throw",
201        "spawn", "agent", "true", "false", "impl", "pub", "self", "Ok", "Err", "Some", "None",
202    ];
203    for kw in keywords {
204        if prefix.is_empty() || kw.starts_with(&prefix) {
205            items.push(CompletionItem {
206                label: kw.to_string(),
207                kind: "keyword".to_string(),
208                detail: keyword_doc(kw).map(String::from),
209            });
210        }
211    }
212
213    let modules = [
214        "chain", "ai", "assist", "log", "auth", "config", "db", "crypto", "oracle", "agent",
215    ];
216    for m in modules {
217        if prefix.is_empty() || m.starts_with(&prefix) {
218            items.push(CompletionItem {
219                label: m.to_string(),
220                kind: "module".to_string(),
221                detail: stdlib_doc(m).map(String::from),
222            });
223        }
224    }
225
226    for (name, detail) in collect_symbols_from_source(source) {
227        if prefix.is_empty() || name.starts_with(&prefix) {
228            items.push(CompletionItem {
229                label: name,
230                kind: "function".to_string(),
231                detail: Some(detail),
232            });
233        }
234    }
235
236    items
237}
