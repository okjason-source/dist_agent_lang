1//! DAL document symbols and references for IDE (Functions & Invocations view).
2//! Used by POST /api/lsp/document_symbols and POST /api/lsp/references.

4use crate::lexer::Lexer;
5use crate::parser::ast::Statement;
6use crate::parser::Parser;
7use serde::{Deserialize, Serialize};

9/// One symbol (function, service, or method) with location.
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct DocumentSymbol {
12    pub name: String,
13    pub kind: String, // "function" | "service" | "method"
14    #[serde(skip_serializing_if = "Option::is_none")]
15    pub signature: Option<String>,
16    pub line: u32,
17    pub column: u32,
18}
19
20/// A reference (call site): line and column where the symbol is called.
21#[derive(Debug, Clone, Serialize, Deserialize)]
22pub struct ReferenceLocation {
23    pub line: u32,
24    pub column: u32,
25}

27/// Parse source and return document symbols with locations (from AST + statement_spans).
28pub fn document_symbols_from_source(source: &str) -> Vec<DocumentSymbol> {
29    let mut symbols = Vec::new();
30    let lexer = Lexer::new(source);
31    let tokens_with_pos = match lexer.tokenize_with_positions_immutable() {
32        Ok(t) => t,
33        Err(_) => return symbols,
34    };
35    let mut parser = Parser::new_with_positions(tokens_with_pos);
36    let program = match parser.parse() {
37        Ok(p) => p,
38        Err(_) => return symbols,
39    };
40    for (i, stmt) in program.statements.iter().enumerate() {
41        let span = program
42            .statement_spans
43            .get(i)
44            .and_then(|s| *s)
45            .unwrap_or(crate::parser::ast::Span { line: 1, column: 1 });
46        let line = span.line as u32;
47        let col = span.column as u32;
48        match stmt {
49            Statement::Function(f) => {
50                let sig = format!(
51                    "fn {}({})",
52                    f.name,
53                    f.parameters
54                        .iter()
55                        .map(|p| p.name.as_str())
56                        .collect::<Vec<_>>()
57                        .join(", ")
58                );
59                symbols.push(DocumentSymbol {
60                    name: f.name.clone(),
61                    kind: "function".to_string(),
62                    signature: Some(sig),
63                    line,
64                    column: col + 4, // after "fn "
65                });
66            }
67            Statement::Service(s) => {
68                symbols.push(DocumentSymbol {
69                    name: s.name.clone(),
70                    kind: "service".to_string(),
71                    signature: Some(format!("service {}", s.name)),
72                    line,
73                    column: col + 8, // after "service "
74                });
75                for m in &s.methods {
76                    let sig = format!(
77                        "fn {}({})",
78                        m.name,
79                        m.parameters
80                            .iter()
81                            .map(|p| p.name.as_str())
82                            .collect::<Vec<_>>()
83                            .join(", ")
84                    );
85                    symbols.push(DocumentSymbol {
86                        name: m.name.clone(),
87                        kind: "method".to_string(),
88                        signature: Some(sig),
89                        line,
90                        column: col + 8,
91                    });
92                }
93            }
94            _ => {}
95        }
96    }
97    symbols
98}

100/// Find call sites of `name` in source (text-based: identifier followed by `(`).
101pub fn references_in_source(source: &str, name: &str) -> Vec<ReferenceLocation> {
102    let mut refs = Vec::new();
103    if name.is_empty() {
104        return refs;
105    }
106    for (line_0, line) in source.lines().enumerate() {
107        let mut search_start = 0;
108        while let Some(idx) = line[search_start..].find(name) {
109            let pos = search_start + idx;
110            let after = line[pos + name.len()..].trim_start();
111            let word_bound_before = pos == 0
112                || !line[pos - 1..pos]
113                    .chars()
114                    .next()
115                    .is_some_and(|c| c.is_ascii_alphanumeric() || c == '_');
116            let word_bound_after = after.is_empty()
117                || !after
118                    .chars()
119                    .next()
120                    .is_some_and(|c| c.is_ascii_alphanumeric() || c == '_');
121            if word_bound_before && word_bound_after && after.starts_with('(') {
122                let column_0 = line[..pos].chars().count();
123                refs.push(ReferenceLocation {
124                    line: (line_0 + 1) as u32,
125                    column: (column_0 + 1) as u32,
126                });
127            }
128            search_start = pos + 1;
129        }
130    }
131    refs
132}
