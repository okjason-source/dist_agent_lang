1//! DAL package registry server. Production-ready: auth, file storage, index API.
2//!
3//! Endpoints:
4//! - GET /v1/packages/:name     -> index JSON (versions + url per version)
5//! - GET /v1/packages/:name/versions/:version/tarball -> tarball bytes
6//! - PUT /v1/packages/:name/versions/:version -> publish (body = tarball, `Authorization: Bearer` + registry token)
7//!
8//! Env: REGISTRY_STORAGE_PATH, REGISTRY_TOKEN (auth), REGISTRY_PUBLIC_URL (for index urls), PORT.
9

27fn storage_path() -> PathBuf {
28    std::env::var("REGISTRY_STORAGE_PATH")
29        .map(PathBuf::from)
30        .unwrap_or_else(|_| PathBuf::from("./registry-storage"))
31}
32
33fn auth_token() -> Option<String> {
34    std::env::var("REGISTRY_TOKEN")
35        .or_else(|_| std::env::var("DAL_REGISTRY_TOKEN"))
36        .ok()
37}
38
39fn public_url() -> String {
40    std::env::var("REGISTRY_PUBLIC_URL").unwrap_or_else(|_| "http://localhost:8787".to_string())
41}

43#[derive(Clone)]
44struct AppState {
45    storage: PathBuf,
46    public_url: String,
47    auth_token: Option<String>,
48}
49
50async fn get_package_index(
51    State(state): State<Arc<AppState>>,
52    Path(name): Path<String>,
53) -> Response {
54    if !sanitize_name(&name) {
55        return (StatusCode::BAD_REQUEST, "invalid package name").into_response();
56    }
57    let dir = name_to_storage_dir(&name);
58    let index_path = state.storage.join("packages").join(dir).join("index.json");
59    if !path_under_storage(&index_path, &state.storage) {
60        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
61    }
62    match tokio::fs::read_to_string(&index_path).await {
63        Ok(s) => {
64            let v: serde_json::Value = match serde_json::from_str(&s) {
65                Ok(j) => j,
66                Err(_) => {
67                    return (StatusCode::INTERNAL_SERVER_ERROR, "invalid index.json")
68                        .into_response()
69                }
70            };
71            (StatusCode::OK, Json(v)).into_response()
72        }
73        Err(e) if e.kind() == std::io::ErrorKind::NotFound => (
74            StatusCode::NOT_FOUND,
75            format!("package not found: {}", name),
76        )
77            .into_response(),
78        Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "read error").into_response(),
79    }
80}

82async fn get_tarball(
83    State(state): State<Arc<AppState>>,
84    Path((name, version)): Path<(String, String)>,
85) -> Response {
86    if !sanitize_name(&name) || !sanitize_version(&version) {
87        return (StatusCode::BAD_REQUEST, "invalid name or version").into_response();
88    }
89    let dir = name_to_storage_dir(&name);
90    let tgz_path = state
91        .storage
92        .join("packages")
93        .join(dir)
94        .join("versions")
95        .join(format!("{}.tgz", version));
96    if !path_under_storage(&tgz_path, &state.storage) {
97        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
98    }
99    match tokio::fs::read(&tgz_path).await {
100        Ok(bytes) => (
101            StatusCode::OK,
102            [("Content-Type", "application/gzip")],
103            bytes,
104        )
105            .into_response(),
106        Err(e) if e.kind() == std::io::ErrorKind::NotFound => (
107            StatusCode::NOT_FOUND,
108            format!("version not found: {}@{}", name, version),
109        )
110            .into_response(),
111        Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "read error").into_response(),
112    }
113}

115async fn put_version(
116    State(state): State<Arc<AppState>>,
117    Path((name, version)): Path<(String, String)>,
118    request: Request<Body>,
119) -> Response {
120    if !sanitize_name(&name) || !sanitize_version(&version) {
121        return (StatusCode::BAD_REQUEST, "invalid name or version").into_response();
122    }
123    if let Some(ref token) = state.auth_token {
124        let auth = request
125            .headers()
126            .get("Authorization")
127            .and_then(|v| v.to_str().ok())
128            .unwrap_or("");
129        if !auth.starts_with("Bearer ")
130            || auth.strip_prefix("Bearer ").unwrap_or("") != token.as_str()
131        {
132            return (StatusCode::UNAUTHORIZED, "invalid or missing token").into_response();
133        }
134    }
135    const MAX_TARBALL: usize = 50 * 1024 * 1024; // 50 MiB
136    let body = match axum::body::to_bytes(request.into_body(), MAX_TARBALL).await {
137        Ok(b) => b,
138        Err(_) => return (StatusCode::BAD_REQUEST, "body error").into_response(),
139    };
140    let dir = name_to_storage_dir(&name);
141    let pkg_dir = state.storage.join("packages").join(dir);
142    let versions_dir = pkg_dir.join("versions");
143    if !path_under_storage(&versions_dir, &state.storage) {
144        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
145    }
146    if let Err(e) = tokio::fs::create_dir_all(&versions_dir).await {
147        return (
148            StatusCode::INTERNAL_SERVER_ERROR,
149            format!("create dir: {}", e),
150        )
151            .into_response();
152    }
153    let tgz_path = versions_dir.join(format!("{}.tgz", version));
154    if !path_under_storage(&tgz_path, &state.storage) {
155        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
156    }
157    if let Err(e) = tokio::fs::write(&tgz_path, body.as_ref()).await {
158        return (StatusCode::INTERNAL_SERVER_ERROR, format!("write: {}", e)).into_response();
159    }
160    let index_path = pkg_dir.join("index.json");
161    if !path_under_storage(&index_path, &state.storage) {
162        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
163    }
164    let base = state.public_url.trim_end_matches('/');
165    let url = format!("{}/v1/packages/{}/versions/{}/tarball", base, name, version);
166    let mut index: serde_json::Map<String, serde_json::Value> =
167        match tokio::fs::read_to_string(&index_path).await {
168            Ok(s) => serde_json::from_str(&s).unwrap_or_else(|_| {
169                let mut m = serde_json::Map::new();
170                m.insert("name".to_string(), json!(&name));
171                m.insert("versions".to_string(), json!({}));
172                m
173            }),
174            Err(_) => {
175                let mut m = serde_json::Map::new();
176                m.insert("name".to_string(), json!(&name));
177                m.insert("versions".to_string(), json!({}));
178                m
179            }
180        };
181    let versions = index
182        .get_mut("versions")
183        .and_then(|v| v.as_object_mut())
184        .expect("versions object");
185    versions.insert(version.clone(), json!({ "url": url }));
186    let index_bytes = serde_json::to_vec_pretty(&index).expect("json");
187    if let Err(e) = tokio::fs::write(&index_path, index_bytes).await {
188        return (
189            StatusCode::INTERNAL_SERVER_ERROR,
190            format!("write index: {}", e),
191        )
192            .into_response();
193    }
194    (
195        StatusCode::CREATED,
196        Json(json!({ "ok": true, "name": name, "version": version })),
197    )
198        .into_response()
199}

201#[tokio::main]
202async fn main() {
203    dist_agent_lang::observability::init_tracing();
204
205    let storage = storage_path();
206    let public_url = public_url();
207    let auth_token = auth_token();
208    if auth_token.is_none() {
209        eprintln!(
210            "⚠️  REGISTRY_TOKEN (or DAL_REGISTRY_TOKEN) not set; publish is unauthenticated."
211        );
212    }
213    if let Err(e) = std::fs::create_dir_all(&storage) {
214        eprintln!("Failed to create storage dir {:?}: {}", storage, e);
215        std::process::exit(1);
216    }
217    let state = Arc::new(AppState {
218        storage,
219        public_url,
220        auth_token,
221    });
222    let app = Router::new()
223        .route("/v1/packages/:name", get(get_package_index))
224        .route(
225            "/v1/packages/:name/versions/:version/tarball",
226            get(get_tarball),
227        )
228        .route("/v1/packages/:name/versions/:version", put(put_version))
229        .route(
230            "/metrics",
231            get(dist_agent_lang::observability::metrics_http_response),
232        )
233        .layer(middleware::from_fn(
234            dist_agent_lang::observability::http_observability_middleware,
235        ))
236        .layer(CorsLayer::permissive())
237        .with_state(state);
238    let port: u16 = std::env::var("PORT")
239        .ok()
240        .and_then(|p| p.parse().ok())
241        .unwrap_or(8787);
242    let addr = std::net::SocketAddr::from(([0, 0, 0, 0], port));
243    println!(
244        "DAL registry listening on http://{} (storage: {:?})",
245        addr,
246        storage_path()
247    );
248    let listener = tokio::net::TcpListener::bind(addr).await.expect("bind");
249    axum::serve(listener, app).await.expect("serve");
250}
