1//! Agent local server: `dal agent serve [name] [--port N] [--mold path]`
2//! One agent per process; HTTP API for status, message, messages, task, tasks, health.
3//! See docs/development/AGENT_LOCAL_SERVER_DESIGN.md.

18/// Shared state for the agent server: the single agent's context.
19/// When prompt_only is true (no behavior script), the server responds to each message via the AI and posts the reply.
20struct AgentServeState {
21    ctx: AgentContext,
22    prompt_only: bool,
23}
24
25/// GET /status — agent id, name, type, status
26async fn handle_status(State(state): State<Arc<AgentServeState>>) -> impl IntoResponse {
27    let ctx = &state.ctx;
28    let body = serde_json::json!({
29        "id": ctx.agent_id,
30        "name": ctx.config.name,
31        "type": ctx.config.agent_type.to_string(),
32        "status": ctx.status.to_string(),
33    });
34    (StatusCode::OK, Json(body))
35}

37/// POST /message — body: { sender_id, content } or { sender_id?, message_type?, content }
38#[derive(Deserialize)]
39struct MessageBody {
40    sender_id: String,
41    #[serde(default)]
42    message_type: String,
43    content: String,
44}
45
46async fn handle_message(
47    State(state): State<Arc<AgentServeState>>,
48    Json(body): Json<MessageBody>,
49) -> impl IntoResponse {
50    let agent_id = &state.ctx.agent_id;
51    let message_type = if body.message_type.is_empty() {
52        "user".to_string()
53    } else {
54        body.message_type
55    };
56    let content_for_reply = body.content.clone();
57    let user_id_for_reply = body.sender_id.clone();
58    let message_id = format!(
59        "msg_{}",
60        SystemTime::now()
61            .duration_since(UNIX_EPOCH)
62            .unwrap_or_default()
63            .as_millis()
64    );
65    let msg = agent::create_agent_message(
66        message_id,
67        body.sender_id,
68        agent_id.clone(),
69        message_type,
70        dist_agent_lang::runtime::values::Value::String(body.content),
71    );
72    let sender = msg.sender_id.clone();
73    match agent::communicate(sender.as_str(), agent_id.as_str(), msg) {
74        Ok(_) => {
75            if state.prompt_only {
76                let agent_id = state.ctx.agent_id.clone();
77                let content = content_for_reply;
78                let user_id = user_id_for_reply;
79                tokio::task::spawn_blocking(move || {
80                    if let Ok(reply) = dist_agent_lang::stdlib::ai::generate_text(content) {
81                        let reply_msg = agent::create_agent_message(
82                            format!(
83                                "msg_reply_{}",
84                                SystemTime::now()
85                                    .duration_since(UNIX_EPOCH)
86                                    .unwrap_or_default()
87                                    .as_millis()
88                            ),
89                            agent_id.clone(),
90                            user_id.clone(),
91                            "assistant".to_string(),
92                            dist_agent_lang::runtime::values::Value::String(reply),
93                        );
94                        let _ = agent::communicate(agent_id.as_str(), user_id.as_str(), reply_msg);
95                    }
96                });
97            }
98            (StatusCode::OK, Json(serde_json::json!({ "ok": true })))
99        }
100        Err(e) => (
101            StatusCode::INTERNAL_SERVER_ERROR,
102            Json(serde_json::json!({ "error": e })),
103        ),
104    }
105}

107/// GET /messages — receive (and consume) messages for this agent
108async fn handle_messages(State(state): State<Arc<AgentServeState>>) -> impl IntoResponse {
109    let messages = agent::receive_messages(state.ctx.agent_id.as_str());
110    let arr: Vec<serde_json::Value> = messages
111        .iter()
112        .map(|m| {
113            serde_json::json!({
114                "message_id": m.message_id,
115                "sender_id": m.sender_id,
116                "message_type": m.message_type,
117                "content": value_to_json(&m.content),
118            })
119        })
120        .collect();
121    (StatusCode::OK, Json(arr))
122}

124/// POST /task — body: { task_id?, description, priority?, requester_id? }
125/// In prompt-only mode, requester_id (optional) is used to send the task result back as a message.
126#[derive(Deserialize)]
127struct TaskBody {
128    #[serde(default)]
129    task_id: String,
130    description: String,
131    #[serde(default)]
132    priority: String,
133    #[serde(default)]
134    requester_id: String,
135}
136
137fn task_priority_from_str(s: &str) -> &'static str {
138    match s.to_lowercase().as_str() {
139        "low" => "low",
140        "high" => "high",
141        "critical" => "critical",
142        "normal" => "medium",
143        _ => "medium",
144    }
145}
146
147async fn handle_task(
148    State(state): State<Arc<AgentServeState>>,
149    Json(body): Json<TaskBody>,
150) -> impl IntoResponse {
151    let agent_id = &state.ctx.agent_id;
152    let task_id = if body.task_id.is_empty() {
153        format!(
154            "task_{}",
155            SystemTime::now()
156                .duration_since(UNIX_EPOCH)
157                .unwrap_or_default()
158                .as_millis()
159        )
160    } else {
161        body.task_id
162    };
163    let priority = task_priority_from_str(&body.priority);
164    let task_opt = agent::create_agent_task(task_id.clone(), body.description.clone(), priority);
165    match task_opt {
166        Some(task) => {
167            if let Err(e) = agent::coordinate(agent_id.as_str(), task, "task_distribution") {
168                return (
169                    StatusCode::INTERNAL_SERVER_ERROR,
170                    Json(serde_json::json!({ "error": e })),
171                );
172            }
173            if state.prompt_only {
174                let agent_id = state.ctx.agent_id.clone();
175                let requester_id = if body.requester_id.trim().is_empty() {
176                    "task_requester".to_string()
177                } else {
178                    body.requester_id
179                };
180                tokio::task::spawn_blocking(move || {
181                    let pending = agent::receive_pending_tasks(agent_id.as_str());
182                    if let Some(t) = pending.into_iter().next() {
183                        let prompt = format!(
184                            "Complete the following task. Reply with the result or answer only.\n\nTask: {}",
185                            t.description
186                        );
187                        if let Ok(result) = dist_agent_lang::stdlib::ai::generate_text(prompt) {
188                            let reply_msg = agent::create_agent_message(
189                                format!(
190                                    "msg_task_{}",
191                                    SystemTime::now()
192                                        .duration_since(UNIX_EPOCH)
193                                        .unwrap_or_default()
194                                        .as_millis()
195                                ),
196                                agent_id.clone(),
197                                requester_id.clone(),
198                                "task_result".to_string(),
199                                dist_agent_lang::runtime::values::Value::String(result),
200                            );
201                            let _ = agent::communicate(
202                                agent_id.as_str(),
203                                requester_id.as_str(),
204                                reply_msg,
205                            );
206                        }
207                    }
208                });
209            }
210            (
211                StatusCode::OK,
212                Json(serde_json::json!({ "ok": true, "task_id": task_id })),
213            )
214        }
215        None => (
216            StatusCode::BAD_REQUEST,
217            Json(serde_json::json!({
218                "error": "Invalid priority; use low, normal, high, critical"
219            })),
220        ),
221    }
222}

224/// GET /tasks — receive (and consume) pending tasks for this agent
225async fn handle_tasks(State(state): State<Arc<AgentServeState>>) -> impl IntoResponse {
226    let tasks = agent::receive_pending_tasks(state.ctx.agent_id.as_str());
227    let arr: Vec<serde_json::Value> = tasks
228        .iter()
229        .map(|t| {
230            serde_json::json!({
231                "task_id": t.task_id,
232                "description": t.description,
233                "priority": format!("{:?}", t.priority).to_lowercase(),
234                "status": format!("{:?}", t.status).to_lowercase(),
235                "assigned_at": t.assigned_at,
236            })
237        })
238        .collect();
239    (StatusCode::OK, Json(arr))
240}

242/// GET /health — liveness
243async fn handle_health() -> impl IntoResponse {
244    (StatusCode::OK, Json(serde_json::json!({ "status": "ok" })))
245}

247fn build_router(state: Arc<AgentServeState>) -> Router {
248    Router::new()
249        .route("/status", get(handle_status))
250        .route("/message", post(handle_message))
251        .route("/messages", get(handle_messages))
252        .route("/task", post(handle_task))
253        .route("/tasks", get(handle_tasks))
254        .route("/health", get(handle_health))
255        .with_state(state)
256}

258/// Spawn agent: from mold if mold_path is Some, else default config with name and type "worker".
259fn spawn_agent(name: &str, mold_path: Option<&str>) -> Result<AgentContext, String> {
260    if let Some(source) = mold_path {
261        let base = std::env::current_dir().unwrap_or_else(|_| std::path::PathBuf::from("."));
262        dist_agent_lang::mold::create_from_mold_source(source, &base, Some(name))
263    } else {
264        let config =
265            agent::create_agent_config(name.to_string(), "worker", "Agent serve".to_string())
266                .ok_or_else(|| "Invalid agent type".to_string())?;
267        agent::spawn(config)
268    }
269}

271/// Run the agent HTTP server. Blocks until the server stops.
272/// If behavior_path is Some, run that DAL file first; the script must spawn an agent and call agent::set_serve_agent(agent_id). Otherwise spawn from name/mold.
273/// When behavior_path is None and prompt_only is true, the server responds to each message by calling the AI and posting the reply (prompt directions only, no DAL script).
274pub fn run_agent_serve(
275    name: &str,
276    port: u16,
277    mold_path: Option<&str>,
278    behavior_path: Option<&str>,
279    prompt_only: bool,
280) -> Result<(), String> {
281    let ctx = if let Some(path) = behavior_path {
282        dist_agent_lang::execute_dal_file(path)
283            .map_err(|e| format!("Behavior script error: {}", e))?;
284        agent::get_serve_agent_context().ok_or_else(|| {
285            "Behavior script did not set serve agent. In your script, spawn an agent and call agent::set_serve_agent(agent_id).".to_string()
286        })?
287    } else {
288        spawn_agent(name, mold_path)?
289    };
290    let display_name = ctx.config.name.clone();
291    let prompt_only = prompt_only && behavior_path.is_none();
292    let state = Arc::new(AgentServeState { ctx, prompt_only });
293    let app = build_router(state);
294
295    let addr = std::net::SocketAddr::from(([0, 0, 0, 0], port));
296    println!(
297        "Agent \"{}\" listening on http://localhost:{}/",
298        display_name.as_str(),
299        port
300    );
301    println!("  GET  /status   — agent id, name, type, status");
302    println!("  POST /message  — send message (body: sender_id, content)");
303    println!("  GET  /messages — receive messages");
304    println!("  POST /task     — assign task (body: description, optional task_id, priority)");
305    println!("  GET  /tasks    — receive pending tasks");
306    println!("  GET  /health   — liveness");
307    if prompt_only {
308        println!("  (prompt-only mode: replies via AI for each message)");
309    }
310    println!("Press Ctrl+C to stop");
311
312    let rt = tokio::runtime::Runtime::new().map_err(|e| e.to_string())?;
313    rt.block_on(async move {
314        let listener = tokio::net::TcpListener::bind(addr)
315            .await
316            .map_err(|e| format!("Failed to bind to port {}: {}", port, e))?;
317        axum::serve(listener, app)
318            .await
319            .map_err(|e| format!("Server error: {}", e))
320    })
321}
