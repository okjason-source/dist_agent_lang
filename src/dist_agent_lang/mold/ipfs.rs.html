3use std::path::Path;

5/// Default IPFS API base when DAL_IPFS_API is not set (local node).
6pub const DEFAULT_IPFS_API: &str = "http://127.0.0.1:5001";
7
8fn api_base() -> String {
9    std::env::var("DAL_IPFS_API").unwrap_or_else(|_| DEFAULT_IPFS_API.to_string())
10}
11
12/// Upload a file to IPFS via the HTTP API (POST /api/v0/add). Returns the CID (Hash).
13pub fn upload_file(api_base: &str, path: &Path) -> Result<String, String> {
14    let bytes = std::fs::read(path).map_err(|e| format!("read file: {}", e))?;
15    let name = path
16        .file_name()
17        .and_then(|n| n.to_str())
18        .unwrap_or("mold.json");
19    let part = reqwest::blocking::multipart::Part::bytes(bytes)
20        .file_name(name.to_string())
21        .mime_str("application/json")
22        .map_err(|e| format!("multipart: {}", e))?;
23    let form = reqwest::blocking::multipart::Form::new().part("file", part);
24    let url = format!("{}/api/v0/add", api_base.trim_end_matches('/'));
25    let client = reqwest::blocking::Client::builder()
26        .timeout(std::time::Duration::from_secs(60))
27        .build()
28        .map_err(|e| format!("http client: {}", e))?;
29    let resp = client
30        .post(&url)
31        .multipart(form)
32        .send()
33        .map_err(|e| format!("IPFS add request: {}", e))?;
34    if !resp.status().is_success() {
35        let status = resp.status();
36        let body = resp.text().unwrap_or_default();
37        return Err(format!("IPFS add failed {}: {}", status, body));
38    }
39    let json: serde_json::Value = resp
40        .json()
41        .map_err(|e| format!("IPFS add response: {}", e))?;
42    let hash = json
43        .get("Hash")
44        .and_then(|v: &serde_json::Value| v.as_str())
45        .ok_or_else(|| "IPFS response missing Hash".to_string())?;
46    Ok(hash.to_string())
47}

49/// Download content from IPFS by CID (POST /api/v0/cat?arg=<cid>).
50pub fn cat(api_base: &str, cid: &str) -> Result<String, String> {
51    let url = format!("{}/api/v0/cat?arg={}", api_base.trim_end_matches('/'), cid);
52    let client = reqwest::blocking::Client::builder()
53        .timeout(std::time::Duration::from_secs(60))
54        .build()
55        .map_err(|e| format!("http client: {}", e))?;
56    let resp = client
57        .post(&url)
58        .send()
59        .map_err(|e| format!("IPFS cat request: {}", e))?;
60    if !resp.status().is_success() {
61        let status = resp.status();
62        let body = resp.text().unwrap_or_default();
63        return Err(format!("IPFS cat failed {}: {}", status, body));
64    }
65    let body = resp.text().map_err(|e| format!("IPFS cat body: {}", e))?;
66    Ok(body)
67}

69/// Upload mold file at path; uses DAL_IPFS_API or localhost:5001.
70pub fn upload_mold_to_ipfs(path: &Path) -> Result<String, String> {
71    let base = api_base();
72    upload_file(&base, path)
73}

75/// Download mold content from IPFS by CID.
76pub fn download_mold_from_ipfs(cid: &str) -> Result<String, String> {
77    let base = api_base();
78    cat(&base, cid)
