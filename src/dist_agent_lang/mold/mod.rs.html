19/// Use on-chain mold (pay + cap check), then load from IPFS and spawn. Returns agent_id.
20/// Requires web3 feature. Env: DAL_PRIVATE_KEY, DAL_MOLD_REGISTRY_ADDRESS, DAL_RPC_URL.
21#[cfg(feature = "web3")]
22pub fn use_mold_and_spawn(
23    mold_id: u64,
24    base: &Path,
25    name_override: Option<&str>,
26) -> Result<crate::stdlib::agent::AgentContext, String> {
27    use crate::mold::web3;
28    let info = web3::get_mold_info(mold_id)?;
29    if !info.active {
30        return Err(format!("Mold {} is not active", mold_id));
31    }
32    web3::use_mold(mold_id, info.mint_fee)?;
33    let ipfs_source = format!("ipfs://{}", info.ipfs_hash.trim_start_matches("ipfs://"));
34    create_from_mold_source(&ipfs_source, base, name_override, None)
35}

58/// Substitute `{{key}}` in `s` with `params.get(key)`. Keys are case-sensitive.
59fn substitute_params(s: &str, params: &HashMap<String, String>) -> String {
60    let mut out = s.to_string();
61    for (k, v) in params {
62        let placeholder = format!("{{{{{}}}}}", k);
63        out = out.replace(&placeholder, v);
64    }
65    out
66}

68/// Build AgentConfig from a MoldConfig with optional principal overrides and create-time params.
69/// Principal owns trust: when `principal` is Some, trust_level is taken from principal only (never from mold).
70/// When `params` is Some, each k=v is merged into metadata and `{{key}}` in role/capabilities is substituted.
71/// Context path is always from process (evolve::get_path()); it is not stored in AgentConfig.
72/// Name override: if `name_override` is Some, use it as agent name; else use mold.name.
73pub fn mold_config_to_agent_config(
74    mold: &MoldConfig,
75    name_override: Option<&str>,
76    principal: Option<&PrincipalOverrides>,
77    params: Option<&HashMap<String, String>>,
78) -> Result<AgentConfig, String> {
79    let name = name_override
80        .map(String::from)
81        .unwrap_or_else(|| mold.name.clone());
82    let agent_type_str = mold.agent.agent_type.trim().to_lowercase();
83    let agent_type = AgentType::from_string(&agent_type_str)
84        .ok_or_else(|| format!("unknown agent type in mold: {}", mold.agent.agent_type))?;
85
86    let trust_level = principal
87        .map(|p| p.trust_level.as_str())
88        .unwrap_or("sandboxed")
89        .to_string();
90
91    let max_memory = MoldConfig::memory_limit_to_max_memory(&mold.agent.memory_limit);
92    let mut metadata: HashMap<String, Value> = mold
93        .metadata
94        .iter()
95        .map(|(k, v)| (k.clone(), json_value_to_runtime_value(v)))
96        .collect();
97    metadata.insert("mold_name".to_string(), Value::String(mold.name.clone()));
98    metadata.insert(
99        "mold_version".to_string(),
100        Value::String(mold.version.clone()),
101    );
102    if let Some(p) = params {
103        for (k, v) in p {
104            metadata.insert(k.clone(), Value::String(v.clone()));
105        }
106    }
107
108    let role = params
109        .map(|p| substitute_params(&mold.agent.role, p))
110        .unwrap_or_else(|| mold.agent.role.clone());
111    let capabilities: Vec<String> = params
112        .map(|p| {
113            mold.agent
114                .capabilities
115                .iter()
116                .map(|c| substitute_params(c, p))
117                .collect()
118        })
119        .unwrap_or_else(|| mold.agent.capabilities.clone());
120
121    let lifecycle = mold.lifecycle.as_ref().map(|l| LifecycleHooks {
122        on_create: l.on_create.clone(),
123        on_message: l.on_message.clone(),
124        on_evolve: l.on_evolve.clone(),
125        on_destroy: l.on_destroy.clone(),
126    });
127
128    let skills: Vec<String> = if mold.agent.skills.is_empty() {
129        crate::skills::DEFAULT_LEARNING_PATH_SKILLS
130            .iter()
131            .map(|s| (*s).to_string())
132            .collect()
133    } else {
134        mold.agent.skills.clone()
135    };
136
137    Ok(AgentConfig::new(name, agent_type)
138        .with_role(role)
139        .with_capabilities(capabilities)
140        .with_skills(skills)
141        .with_trust_level(trust_level)
142        .with_max_memory(max_memory)
143        .with_learning_enabled(mold.agent.learning)
144        .with_communication_enabled(mold.agent.communication)
145        .with_coordination_enabled(mold.agent.coordination)
146        .with_metadata(metadata)
147        .with_lifecycle(lifecycle))
148}

150fn json_value_to_runtime_value(v: &serde_json::Value) -> Value {
151    match v {
152        serde_json::Value::Null => Value::Null,
153        serde_json::Value::Bool(b) => Value::Bool(*b),
154        serde_json::Value::Number(n) => {
155            if let Some(i) = n.as_i64() {
156                Value::Int(i)
157            } else if let Some(f) = n.as_f64() {
158                Value::Float(f)
159            } else {
160                Value::Null
161            }
162        }
163        serde_json::Value::String(s) => Value::String(s.clone()),
164        serde_json::Value::Array(arr) => {
165            Value::List(arr.iter().map(json_value_to_runtime_value).collect())
166        }
167        serde_json::Value::Object(obj) => Value::Map(
168            obj.iter()
169                .map(|(k, v)| (k.clone(), json_value_to_runtime_value(v)))
170                .collect(),
171        ),
172    }
173}

175/// Load mold from path, convert to AgentConfig (with principal overrides for trust), and spawn.
176/// Evolve/context path is always from process (agent.toml / DAL_AGENT_CONTEXT_PATH).
177/// Create-time params are merged into metadata and substituted in role/capabilities ({{key}}).
178pub fn create_from_mold_path(
179    path: &Path,
180    name_override: Option<&str>,
181    params: Option<&HashMap<String, String>>,
182) -> Result<crate::stdlib::agent::AgentContext, String> {
183    let mold = load_mold_from_path(path)?;
184    let principal = load_principal_overrides();
185    let config = mold_config_to_agent_config(&mold, name_override, Some(&principal), params)?;
186    spawn(config)
187}

189/// Load mold from a source: local path/name (resolved with `base`) or `ipfs://<cid>` (requires http-interface).
190pub fn load_mold_from_source(source: &str, base: &Path) -> Result<MoldConfig, String> {
191    let source = source.trim();
192    if source.starts_with("ipfs://") {
193        #[cfg(feature = "http-interface")]
194        {
195            let cid = source.trim_start_matches("ipfs://").trim_end_matches('/');
196            let content = download_mold_from_ipfs(cid)?;
197            parse_mold_content(&content)
198        }
199        #[cfg(not(feature = "http-interface"))]
200        {
201            let _ = base;
202            Err("IPFS sources require building with http-interface feature (default)".to_string())
203        }
204    } else {
205        let path = resolve_mold_path(base, source)?;
206        load_mold_from_path(&path)
207    }
208}

225/// Discover local mold files under `base`: base, base/mold, base/mold/samples.
226/// Returns paths for files named `*.mold.dal` (canonical) or `*.mold.json` (legacy). .mold.dal listed first per stem.
227pub fn list_local_paths(base: &Path) -> Vec<PathBuf> {
228    let mut out = Vec::new();
229    let dirs = [
230        base.to_path_buf(),
231        base.join("mold"),
232        base.join("mold").join("samples"),
233    ];
234    for dir in &dirs {
235        if let Ok(entries) = std::fs::read_dir(dir) {
236            for e in entries.flatten() {
237                let p = e.path();
238                if p.is_file() {
239                    let name = p.file_name().and_then(|n| n.to_str()).unwrap_or("");
240                    if name.ends_with(".mold.dal") || name.ends_with(".mold.json") {
241                        out.push(p);
242                    }
243                }
244            }
245        }
246    }
247    out.sort_by(|a, b| {
248        let a_dal = a.to_string_lossy().ends_with(".mold.dal");
249        let b_dal = b.to_string_lossy().ends_with(".mold.dal");
250        match (a_dal, b_dal) {
251            (true, false) => std::cmp::Ordering::Less,
252            (false, true) => std::cmp::Ordering::Greater,
253            _ => a.cmp(b),
254        }
255    });
256    out.dedup();
257    out
258}

260/// Resolve path or name to a mold file path. If `path_or_name` is an existing path, return it; else try list_local and match by stem or mold name.
261pub fn resolve_mold_path(base: &Path, path_or_name: &str) -> Result<PathBuf, String> {
262    let p = Path::new(path_or_name);
263    if p.is_absolute() && p.exists() {
264        return Ok(p.to_path_buf());
265    }
266    let relative = base.join(path_or_name);
267    if relative.exists() {
268        return Ok(relative);
269    }
270    // Prefer .mold.dal (canonical), then .mold.json (legacy)
271    let with_dal = base.join(format!("{}.mold.dal", path_or_name));
272    if with_dal.exists() {
273        return Ok(with_dal);
274    }
275    let with_json = base.join(format!("{}.mold.json", path_or_name));
276    if with_json.exists() {
277        return Ok(with_json);
278    }
279    let list = list_local_paths(base);
280    for path in &list {
281        if path
282            .file_stem()
283            .and_then(|s| s.to_str())
284            .map(|s| s.trim_end_matches(".mold") == path_or_name)
285            .unwrap_or(false)
286        {
287            return Ok(path.clone());
288        }
289        if let Ok(m) = load_mold_from_path(path) {
290            if m.name == path_or_name {
291                return Ok(path.clone());
292            }
293        }
294    }
295    Err(format!(
296        "mold not found: {} (tried path, {}.mold.dal, {}.mold.json, and local list)",
297        path_or_name, path_or_name, path_or_name
298    ))
299}

301/// Scaffold a new .mold.dal file at `out_path` (canonical format).
302/// Includes default learning-path skills (development, creative, office, home) per AGENT_ASSISTANT_PLAN.
303pub fn scaffold_mold(name: &str, out_path: &Path) -> Result<(), String> {
304    let content = format!(
305        r#"mold "{}" "1.0"
306agent
307  type AI
308  role ""
309  capabilities
310  skills "development" "creative" "office" "home"
311  memory_limit "256MB"
312  learning true
313  communication true
314  coordination true
315"#,
316        name
317    );
318    std::fs::write(out_path, content).map_err(|e| format!("write {}: {}", out_path.display(), e))
319}
