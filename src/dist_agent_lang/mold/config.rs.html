44impl MoldConfig {
45    /// Memory limit string (e.g. "2GB") to approximate max_memory in bytes/pages.
46    pub fn memory_limit_to_max_memory(s: &str) -> usize {
47        let s = s.trim().to_uppercase();
48        if s.is_empty() {
49            return 1000;
50        }
51        let (num_part, unit) = if s.ends_with("GB") {
52            (s.trim_end_matches("GB"), 1024 * 1024 * 1024)
53        } else if s.ends_with("MB") {
54            (s.trim_end_matches("MB"), 1024 * 1024)
55        } else if s.ends_with("KB") {
56            (s.trim_end_matches("KB"), 1024)
57        } else {
58            (s.as_str(), 1)
59        };
60        let n: usize = num_part.trim().parse().unwrap_or(2);
61        (n * unit).min(usize::MAX) / 4096 // rough page count for max_memory
62    }
