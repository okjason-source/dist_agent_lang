1//! CLI design: branding, help layout, and AI-ready structure.
2//!
3//! Organized by use case for intuitive discovery across project types.
4//! AI-first: "AI & CODE ASSISTANCE" and "AGENTS & AUTOMATION" appear early.
5//! Future: `dal ask "<prompt>"` or `dal "<natural language>"` can be added
6//! as primary entry points as AI integration evolves.
7
8/// Brand tagline — used in banner and help (short form for banner)
9pub const TAGLINE: &str = "Web, Blockchain & AI";
10
11/// Full tagline for version output
12pub const TAGLINE_FULL: &str = "Unified language for Web, Blockchain & AI";
13
14/// Version string from Cargo
15pub fn version() -> &'static str {
16    env!("CARGO_PKG_VERSION")
17}
18
19/// Print branded banner (respects --no-banner, --quiet)
20pub fn print_banner(bin: &str, no_banner: bool, quiet: bool) {
21    if no_banner {
22        return;
23    }
24    if quiet {
25        println!("{} v{} — {}", bin, version(), TAGLINE);
26        return;
27    }
28    println!();
29    println!("┌──────────────────────────────────────────────────────────────────────┐");
30    println!("│  {} v{}  —  {}  │", bin, version(), TAGLINE_FULL);
31    println!("└──────────────────────────────────────────────────────────────────────┘");
32    println!();
33}
