15/// FFI Configuration
16#[derive(Debug, Clone)]
17pub struct FFIConfig {
18    pub interface_type: InterfaceType,
19    pub enable_http: bool,
20    pub enable_ffi: bool,
21    pub python_enabled: bool,
22    pub rust_enabled: bool,
23    pub c_enabled: bool,
24}
25
26impl Default for FFIConfig {
27    fn default() -> Self {
28        Self {
29            interface_type: InterfaceType::Both,
30            enable_http: true,
31            enable_ffi: true,
32            python_enabled: cfg!(feature = "python-ffi"),
33            rust_enabled: true,
34            c_enabled: cfg!(feature = "c-ffi"),
35        }
36    }
37}
38
39impl FFIConfig {
40    pub fn http_only() -> Self {
41        Self {
42            interface_type: InterfaceType::HTTP,
43            enable_http: true,
44            enable_ffi: false,
45            python_enabled: false,
46            rust_enabled: false,
47            c_enabled: false,
48        }
49    }
50
51    pub fn ffi_only() -> Self {
52        Self {
53            interface_type: InterfaceType::FFI,
54            enable_http: false,
55            enable_ffi: true,
56            python_enabled: cfg!(feature = "python-ffi"),
57            rust_enabled: true,
58            c_enabled: cfg!(feature = "c-ffi"),
59        }
60    }
61
62    pub fn both() -> Self {
63        Self::default()
64    }
65
66    /// Auto-detect configuration based on environment
67    pub fn auto_detect() -> Self {
68        // Check if we're in a local context (same process)
69        // If both are available, default to "both" with auto-selection
70        Self {
71            interface_type: InterfaceType::Both,
72            enable_http: true, // Always enable HTTP as fallback
73            enable_ffi: true,  // Enable FFI if available
74            python_enabled: cfg!(feature = "python-ffi"),
75            rust_enabled: true,
76            c_enabled: cfg!(feature = "c-ffi"),
77        }
78    }
79}
