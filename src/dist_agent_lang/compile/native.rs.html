8/// Map DAL type to Rust type for native (full Rust types).
9fn dal_type_to_rust(dal_type: &str) -> String {
10    let t = dal_type.trim();
11    if t.is_empty() || t == "int" {
12        return "i64".to_string();
13    }
14    if t == "string" {
15        return "String".to_string();
16    }
17    if t == "bool" {
18        return "bool".to_string();
19    }
20    if t.starts_with("list<") || t.starts_with("vector<") {
21        let inner = t
22            .strip_prefix("list<")
23            .or_else(|| t.strip_prefix("vector<"))
24            .and_then(|s| s.strip_suffix('>'))
25            .unwrap_or("i64");
26        return format!("Vec<{}>", dal_type_to_rust(inner));
27    }
28    if t.starts_with("map<") {
29        let inner = t
30            .strip_prefix("map<")
31            .and_then(|s| s.strip_suffix('>'))
32            .unwrap_or("String, i64");
33        let mut parts = inner.splitn(2, ',');
34        let k = parts.next().unwrap_or("String").trim();
35        let v = parts.next().unwrap_or("i64").trim();
36        return format!(
37            "std::collections::HashMap<{}, {}>",
38            dal_type_to_rust(k),
39            dal_type_to_rust(v)
40        );
41    }
42    if t == "any" || t == "float" {
43        return "f64".to_string();
44    }
45    t.to_string()
46}

48/// Emit Rust lib.rs for one or more DAL services (rlib for host).
49fn services_to_rust(services: &[&ServiceStatement]) -> String {
50    let mut out = String::new();
51    out.push_str("// Generated by DAL native backend. Compiles for host.\n");
52    out.push_str("#![allow(dead_code)]\n\n");
53
54    for service in services {
55        let mod_name = service.name.to_lowercase().replace('-', "_");
56        out.push_str(&format!("pub mod {} {{\n", mod_name));
57        if service.fields.is_empty() {
58            out.push_str("    pub struct State;\n");
59        } else {
60            out.push_str("    pub struct State {\n");
61            for field in &service.fields {
62                let rust_type = dal_type_to_rust(&field.field_type);
63                out.push_str(&format!("        pub {}: {},\n", field.name, rust_type));
64            }
65            out.push_str("    }\n");
66        }
67        for method in &service.methods {
68            let params_str: Vec<String> = method
69                .parameters
70                .iter()
71                .map(|p| {
72                    let ty = p.param_type.as_deref().unwrap_or("i64");
73                    format!("{}: {}", p.name, dal_type_to_rust(ty))
74                })
75                .collect();
76            let ret = method
77                .return_type
78                .as_deref()
79                .map(|r| dal_type_to_rust(r))
80                .unwrap_or_else(|| "()".to_string());
81            out.push_str(&format!(
82                "    pub fn {}({}) -> {} {{\n        unimplemented!(\"DAL native stub\")\n    }}\n",
83                method.name,
84                params_str.join(", "),
85                ret
86            ));
87        }
88        out.push_str("}\n\n");
89    }
90    out
91}

93/// Check that cargo/rustc is available.
94fn check_rust_available() -> bool {
95    Command::new("cargo")
96        .arg("--version")
97        .output()
98        .map(|o| o.status.success())
99        .unwrap_or(false)
100}

102/// Native backend: DAL → Rust → cargo build --release → .rlib (or binary).
103pub struct NativeBackend;
104
105impl CompileBackend for NativeBackend {
106    fn compile(
107        &self,
108        _program: &Program,
109        services: &[&ServiceStatement],
110        opts: &CompileOptions,
111    ) -> Result<CompileArtifacts, CompileError> {
112        if !check_rust_available() {
113            return Err(CompileError::CompilerNotFound {
114                target: "native".to_string(),
115                hint: "Install Rust: https://rustup.rs".to_string(),
116            });
117        }
118
119        std::fs::create_dir_all(&opts.output_dir).map_err(CompileError::Io)?;
120        let build_dir = opts.output_dir.join("native_build");
121        let src_dir = build_dir.join("src");
122        std::fs::create_dir_all(&src_dir).map_err(CompileError::Io)?;
123
124        let rust_code = services_to_rust(services);
125        let lib_path = src_dir.join("lib.rs");
126        std::fs::write(&lib_path, rust_code).map_err(CompileError::Io)?;
127
128        let cargo_toml = r#"
129[package]
130name = "dal_native"
131version = "0.1.0"
132edition = "2021"
133
134[lib]
135crate-type = ["rlib"]
136"#;
137        std::fs::write(build_dir.join("Cargo.toml"), cargo_toml).map_err(CompileError::Io)?;
138
139        let output = Command::new("cargo")
140            .current_dir(&build_dir)
141            .args(["build", "--release"])
142            .output()
143            .map_err(CompileError::Io)?;
144
145        if !output.status.success() {
146            let stderr = String::from_utf8_lossy(&output.stderr);
147            return Err(CompileError::Backend(format!(
148                "cargo native build failed: {}",
149                stderr.trim()
150            )));
151        }
152
153        let mut artifact_paths = Vec::new();
154        let rlib_name = "libdal_native.rlib";
155        let built_rlib = build_dir.join("target").join("release").join(rlib_name);
156        if built_rlib.exists() {
157            let out_rlib = opts.output_dir.join(rlib_name);
158            std::fs::copy(&built_rlib, &out_rlib).map_err(CompileError::Io)?;
159            artifact_paths.push(out_rlib);
160        }
161
162        let service_names: Vec<String> = services.iter().map(|s| s.name.clone()).collect();
163
164        Ok(CompileArtifacts {
165            target: "native".to_string(),
166            service_names,
167            artifact_paths,
168            stub: false,
169        })
170    }
171}
