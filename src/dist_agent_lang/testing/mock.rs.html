5/// Mock function definition
6#[derive(Debug, Clone)]
7pub struct MockFunction {
8    pub name: String,
9    pub namespace: Option<String>,
10    pub return_value: Option<Value>,
11    pub side_effects: Vec<MockSideEffect>,
12    pub call_count: usize,
13    pub expected_calls: Option<usize>,
14}
15
16impl MockFunction {
17    pub fn new(name: &str) -> Self {
18        Self {
19            name: name.to_string(),
20            namespace: None,
21            return_value: None,
22            side_effects: Vec::new(),
23            call_count: 0,
24            expected_calls: None,
25        }
26    }
27
28    pub fn with_namespace(mut self, namespace: &str) -> Self {
29        self.namespace = Some(namespace.to_string());
30        self
31    }
32
33    pub fn returns(mut self, value: Value) -> Self {
34        self.return_value = Some(value);
35        self
36    }
37
38    pub fn with_side_effect(mut self, side_effect: MockSideEffect) -> Self {
39        self.side_effects.push(side_effect);
40        self
41    }
42
43    pub fn expects_calls(mut self, count: usize) -> Self {
44        self.expected_calls = Some(count);
45        self
46    }
47
48    pub fn validates_arguments<F>(self, _validator: F) -> Self
49    where
50        F: Fn(&[Value]) -> Result<(), String> + 'static,
51    {
52        // Validator functionality removed for simplicity
53        self
54    }
55
56    pub fn call(&mut self, args: &[Value]) -> Result<Value, String> {
57        self.call_count += 1;
58
59        // Execute side effects
60        for side_effect in &self.side_effects {
61            side_effect.execute(args)?;
62        }
63
64        // Return mock value
65        Ok(self.return_value.clone().unwrap_or(Value::Null))
66    }
67
68    pub fn verify(&self) -> Result<(), String> {
69        if let Some(expected) = self.expected_calls {
70            if self.call_count != expected {
71                return Err(format!(
72                    "Mock function '{}' was called {} times, expected {} times",
73                    self.name, self.call_count, expected
74                ));
75            }
76        }
77        Ok(())
78    }
79}

81/// Side effects that can be triggered by mock functions
82#[derive(Debug, Clone)]
83pub enum MockSideEffect {
84    SetVariable(String, Value),
85    CallFunction(String, Vec<Value>),
86    LogMessage(String),
87    ThrowError(String),
88    Delay(std::time::Duration),
89}
90
91impl MockSideEffect {
92    pub fn execute(&self, _args: &[Value]) -> Result<(), String> {
93        match self {
94            MockSideEffect::SetVariable(_, _) => {
95                // In a real implementation, this would set a variable in the runtime
96                Ok(())
97            }
98            MockSideEffect::CallFunction(_, _) => {
99                // In a real implementation, this would call another function
100                Ok(())
101            }
102            MockSideEffect::LogMessage(message) => {
103                println!("[MOCK] {}", message);
104                Ok(())
105            }
106            MockSideEffect::ThrowError(error) => Err(error.clone()),
107            MockSideEffect::Delay(duration) => {
108                std::thread::sleep(*duration);
109                Ok(())
110            }
111        }
112    }
113}

115/// Mock registry for managing multiple mocks
116#[derive(Debug, Clone)]
117pub struct MockRegistry {
118    pub mocks: HashMap<String, MockFunction>,
119    pub enabled: bool,
120}
121
122impl MockRegistry {
123    pub fn new() -> Self {
124        Self {
125            mocks: HashMap::new(),
126            enabled: true,
127        }
128    }
129
130    pub fn register(&mut self, mock: MockFunction) {
131        let key = self.get_mock_key(&mock.name, mock.namespace.as_deref());
132        self.mocks.insert(key, mock);
133    }
134
135    pub fn get_mock(&mut self, name: &str, namespace: Option<&str>) -> Option<&mut MockFunction> {
136        let key = self.get_mock_key(name, namespace);
137        self.mocks.get_mut(&key)
138    }
139
140    pub fn call_mock(
141        &mut self,
142        name: &str,
143        namespace: Option<&str>,
144        args: &[Value],
145    ) -> Result<Value, String> {
146        if !self.enabled {
147            return Err("Mock registry is disabled".to_string());
148        }
149
150        if let Some(mock) = self.get_mock(name, namespace) {
151            mock.call(args)
152        } else {
153            Err(format!(
154                "No mock found for '{}{}'",
155                namespace.map(|ns| format!("{}::", ns)).unwrap_or_default(),
156                name
157            ))
158        }
159    }
160
161    pub fn verify_all(&self) -> Result<(), String> {
162        for (key, mock) in &self.mocks {
163            mock.verify()
164                .map_err(|e| format!("Mock '{}': {}", key, e))?;
165        }
166        Ok(())
167    }
168
169    pub fn clear(&mut self) {
170        self.mocks.clear();
171    }
172
173    pub fn enable(&mut self) {
174        self.enabled = true;
175    }
176
177    pub fn disable(&mut self) {
178        self.enabled = false;
179    }
180
181    fn get_mock_key(&self, name: &str, namespace: Option<&str>) -> String {
182        match namespace {
183            Some(ns) => format!("{}::{}", ns, name),
184            None => name.to_string(),
185        }
186    }
187}

189/// Mock builder for fluent API
190pub struct MockBuilder {

192    namespace: Option<String>,
193    return_value: Option<Value>,

195    expected_calls: Option<usize>,
196    arguments_validator: Option<Box<dyn Fn(&[Value]) -> Result<(), String>>>,
197}
198
199impl MockBuilder {
200    pub fn new(name: &str) -> Self {
201        Self {
202            name: name.to_string(),
203            namespace: None,
204            return_value: None,
205            side_effects: Vec::new(),
206            expected_calls: None,
207            arguments_validator: None,
208        }
209    }
210
211    pub fn in_namespace(mut self, namespace: &str) -> Self {
212        self.namespace = Some(namespace.to_string());
213        self
214    }
215
216    pub fn returns(mut self, value: Value) -> Self {
217        self.return_value = Some(value);

219    }

236    }

238    pub fn expects_calls(mut self, count: usize) -> Self {

241    }

244    where
245        F: Fn(&[Value]) -> Result<(), String> + 'static,

248        self
249    }
250
251    pub fn build(self) -> MockFunction {
252        MockFunction {
253            name: self.name,
254            namespace: self.namespace,
255            return_value: self.return_value,

337/// Test utilities for working with mocks
338pub trait MockTestUtils {
339    fn setup_mocks(&mut self) -> MockRegistry;
340    fn teardown_mocks(&mut self);
341    fn assert_mock_called(&self, name: &str, namespace: Option<&str>, expected_calls: usize);
342    fn assert_mock_not_called(&self, name: &str, namespace: Option<&str>);
343}
344
345impl MockTestUtils for MockRuntime {
346    fn setup_mocks(&mut self) -> MockRegistry {
347        self.mock_registry.enable();
348        self.mock_registry.clone()
349    }
350
351    fn teardown_mocks(&mut self) {
352        self.mock_registry.clear();
353        self.mock_registry.disable();
354    }
355
356    fn assert_mock_called(&self, name: &str, namespace: Option<&str>, expected_calls: usize) {
357        let key = self.mock_registry.get_mock_key(name, namespace);
358        if let Some(mock) = self.mock_registry.mocks.get(&key) {
359            assert_eq!(
360                mock.call_count, expected_calls,
361                "Mock '{}' was called {} times, expected {}",
362                key, mock.call_count, expected_calls
363            );
364        } else {
365            panic!("Mock '{}' not found", key);
366        }
367    }
368
369    fn assert_mock_not_called(&self, name: &str, namespace: Option<&str>) {
370        let key = self.mock_registry.get_mock_key(name, namespace);
371        if let Some(mock) = self.mock_registry.mocks.get(&key) {
372            assert_eq!(
373                mock.call_count, 0,
374                "Mock '{}' was called {} times, expected 0",
375                key, mock.call_count
376            );
377        }
378    }
