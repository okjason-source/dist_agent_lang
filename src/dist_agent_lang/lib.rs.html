92pub fn parse_source(source: &str) -> Result<ast::Program, Box<dyn std::error::Error>> {
93    // Phase 2: Input size limit - prevent DoS via oversized source code
94    const MAX_SOURCE_SIZE: usize = 10 * 1024 * 1024; // 10MB
95    if source.len() > MAX_SOURCE_SIZE {
96        return Err(format!(
97            "Source code too large: {} bytes (max: {} bytes)",
98            source.len(),
99            MAX_SOURCE_SIZE
100        )
101        .into());
102    }
103
104    let lexer = Lexer::new(source);
105    let tokens_with_pos = lexer
106        .tokenize_with_positions_immutable()
107        .map_err(|e| Box::new(e) as Box<dyn std::error::Error>)?;
108
109    // Phase 2: Token count limit - prevent DoS via excessive tokens
110    const MAX_TOKENS: usize = 1_000_000; // 1M tokens
111    if tokens_with_pos.len() > MAX_TOKENS {
112        return Err(format!(
113            "Too many tokens: {} (max: {})",
114            tokens_with_pos.len(),
115            MAX_TOKENS
116        )
117        .into());
118    }
119
120    let mut parser = Parser::new_with_positions(tokens_with_pos);
121    parser
122        .parse()
123        .map_err(|e| Box::new(e) as Box<dyn std::error::Error>)
124}

126pub fn execute_source(source: &str) -> Result<Value, Box<dyn std::error::Error>> {
127    let program = parse_source(source)?;
128    let mut runtime = Runtime::new();
129    let result = runtime
130        .execute_program(program, None)
131        .map_err(|e| Box::new(e) as Box<dyn std::error::Error>)?;
132    Ok(result.unwrap_or(Value::Null))
133}

135/// Execute DAL code with pre-set scope variables (e.g. agent_id for lifecycle hooks).
136pub fn execute_dal_with_scope(
137    vars: &std::collections::HashMap<String, Value>,
138    source: &str,
139) -> Result<Value, Box<dyn std::error::Error>> {
140    let program = parse_source(source)?;
141    let mut runtime = Runtime::new();
142    for (k, v) in vars {
143        runtime.set_variable(k.clone(), v.clone());
144    }
145    let result = runtime
146        .execute_program(program, None)
147        .map_err(|e| Box::new(e) as Box<dyn std::error::Error>)?;
148    Ok(result.unwrap_or(Value::Null))
149}

151/// Execute a DAL file (e.g. agent.dal for serve behavior). Resolves imports when present.
152/// Used by `dal agent serve --behavior path` so the script can spawn an agent and call agent::set_serve_agent(agent_id).
153pub fn execute_dal_file(path: &str) -> Result<(), String> {
154    use parser::ast::Statement;
155    let source =
156        std::fs::read_to_string(path).map_err(|e| format!("Failed to read {}: {}", path, e))?;
157    let program = parse_source(&source).map_err(|e| format!("Parse error: {}", e))?;
158    let has_imports = program
159        .statements
160        .iter()
161        .any(|s| matches!(s, Statement::Import(_)));
162    let mut runtime = Runtime::new();
163    if has_imports {
164        let entry_path = std::path::Path::new(path);
165        let entry_dir = entry_path
166            .parent()
167            .unwrap_or_else(|| std::path::Path::new("."));
168        let manifest_path = entry_dir.join("dal.toml");
169        let mut resolver =
170            module_resolver::ModuleResolver::new().with_root_dir(entry_dir.to_path_buf());
171        if manifest_path.exists() {
172            if let Ok(deps) = manifest::load_resolved_deps(&manifest_path) {
173                resolver = resolver.with_dependencies(deps);
174            }
175        }
176        let parse_fn = |s: &str| parse_source(s).map_err(|e| e.to_string());
177        let resolved = resolver
178            .resolve_program_with_cycles(&program, Some(entry_path), parse_fn)
179            .map_err(|e| e.to_string())?;
180        runtime
181            .execute_program(program, Some(&resolved))
182            .map_err(|e| format!("Runtime error: {}", e))?;
183    } else {
184        runtime
185            .execute_program(program, None)
186            .map_err(|e| format!("Runtime error: {}", e))?;
187    }
188    Ok(())
189}

191/// Execute DAL source and return (user_functions, scope) for HTTP server runtime factory.
192/// Use when you need to serve HTTP routes whose handlers are defined in DAL.
193/// The returned runtime state can be used with create_router_with_runtime_factory.
194pub fn execute_dal_and_extract_handlers(
195    source: &str,
196) -> Result<
197    (
198        std::collections::HashMap<String, runtime::engine::UserFunction>,
199        runtime::scope::Scope,
200    ),
201    Box<dyn std::error::Error>,
202> {
203    let program = parse_source(source)?;
204    let mut runtime = Runtime::new();
205    runtime
206        .execute_program(program, None)
207        .map_err(|e| Box::new(e) as Box<dyn std::error::Error>)?;
208    Ok((runtime.user_functions.clone(), runtime.scope.clone()))
209}
