16    /// Generate DAL source code from AST
17    pub fn generate(&self, ast: DALAST) -> Result<String, String> {
18        let mut code = String::new();
19
20        for service in ast.services {
21            code.push_str(&self.generate_service(service)?);
22            code.push_str("\n\n");
23        }
24
25        Ok(code)
26    }
27
28    fn generate_service(&self, service: Service) -> Result<String, String> {
29        let mut code = String::new();
30
31        // Generate attributes
32        for attr in &service.attributes {
33            code.push_str(attr);
34            code.push('\n');
35        }
36
37        // Generate service declaration
38        code.push_str(&format!("service {} {{\n", service.name));
39
40        // Generate fields
41        if !service.fields.is_empty() {
42            code.push_str("\n    // State variables\n");
43            for field in &service.fields {
44                code.push_str(&self.generate_field(field)?);
45            }
46        }
47
48        // Generate functions (pass field names for storage→self in body conversion)
49        let field_names: Vec<String> = service.fields.iter().map(|f| f.name.clone()).collect();
50        if !service.functions.is_empty() {
51            code.push_str("\n    // Functions\n");
52            for func in &service.functions {
53                code.push_str(&self.generate_function(func, &field_names)?);
54            }
55        }
56
57        // Generate events
58        if !service.events.is_empty() {
59            code.push_str("\n    // Events\n");
60            for event in &service.events {
61                code.push_str(&self.generate_event(event)?);
62            }
63        }
64
65        code.push_str("}\n");
66
67        Ok(code)
68    }

70    fn generate_field(&self, field: &Field) -> Result<String, String> {
71        let mut code = String::new();
72
73        code.push_str(&format!("    {}: {} = ", field.name, field.field_type));
74
75        if let Some(ref init) = field.initial_value {
76            code.push_str(init);
77        } else {
78            // Default values based on type
79            code.push_str(&self.default_value_for_type(&field.field_type));
80        }
81
82        code.push_str(",\n");
83
84        Ok(code)
85    }
86
87    fn default_value_for_type(&self, dal_type: &str) -> String {
88        match dal_type {
89            "int" => "0".to_string(),
90            "bool" => "false".to_string(),
91            "string" => "\"\"".to_string(),
92            "vector<u8>" => "[]".to_string(),
93            _ if dal_type.starts_with("vector<") => "[]".to_string(),
94            _ if dal_type.starts_with("map<") => "{}".to_string(),
95            _ => "null".to_string(),
96        }
97    }

99    fn generate_function(
100        &self,
101        func: &DALFunction,
102        field_names: &[String],
103    ) -> Result<String, String> {
104        let mut code = String::new();
105
106        // Generate attributes
107        for attr in &func.attributes {
108            code.push_str(&format!("    {}\n", attr));
109        }
110
111        // Generate function signature
112        code.push_str(&format!(
113            "    fn {}({})",
114            func.name,
115            self.generate_parameters(&func.parameters)?
116        ));
117
118        // Generate return type
119        if let Some(ref return_type) = func.return_type {
120            code.push_str(&format!(" -> {}", return_type));
121        }
122
123        code.push_str(" {\n");
124
125        if let Some(ref c) = func.comment {
126            code.push_str(&format!("        // {}\n", c));
127        }
128
129        // Generate function body (field_names used for storage→self in Solidity body)
130        if let Some(ref body) = func.body {
131            let dal_body = self.convert_solidity_body_to_dal(body, field_names);
132            code.push_str(&dal_body);
133        } else {
134            code.push_str("        // Function implementation\n");
135        }
136
137        code.push_str("    }\n\n");
138
139        Ok(code)
140    }

142    /// Solidity→DAL body conversion: storage→self; require/revert→if/throw; transfer/call→chain::; msg.sender→chain::caller(); assignments pass-through; rest as comments.
143    fn convert_solidity_body_to_dal(&self, body: &str, field_names: &[String]) -> String {
144        let body_with_self = Self::replace_state_vars_with_self(body, field_names);
145        let mut out = String::new();
146        for line in body_with_self.lines() {
147            let trimmed = line.trim();
148            let dal = self
149                .convert_require_line(trimmed)
150                .or_else(|| self.convert_revert_line(trimmed))
151                .or_else(|| self.convert_transfer_call_line(trimmed))
152                .or_else(|| self.convert_emit_line(trimmed))
153                .or_else(|| self.convert_control_flow_line(trimmed))
154                .or_else(|| self.convert_generic_line(trimmed));
155            match dal {
156                Some(s) => out.push_str(&format!("        {}\n", s)),
157                None => out.push_str(&format!("        // {}\n", line.trim())),
158            }
159        }
160        if out.is_empty() {
161            out.push_str("        // (no statements converted)\n");
162        }
163        out
164    }

166    /// Replace whole-word occurrences of each field name with self.<name> (Solidity storage → DAL self.field).
167    fn replace_state_vars_with_self(body: &str, field_names: &[String]) -> String {
168        let mut out = body.to_string();
169        for name in field_names {
170            if name.is_empty() {
171                continue;
172            }
173            out = Self::replace_word_boundary(&out, name, &format!("self.{}", name));
174        }
175        out
176    }

178    /// Replace whole-word occurrences of `word` with `replacement` (word boundary: not part of a longer identifier).
179    fn replace_word_boundary(s: &str, word: &str, replacement: &str) -> String {
180        let mut result = String::with_capacity(s.len());
181        let bytes = s.as_bytes();
182        let w = word.as_bytes();
183        let mut i = 0;
184        while i <= bytes.len() {
185            if i + w.len() <= bytes.len() && &bytes[i..i + w.len()] == w {
186                let word_start_ok =
187                    i == 0 || !bytes[i - 1].is_ascii_alphanumeric() && bytes[i - 1] != b'_';
188                let word_end_ok = i + w.len() == bytes.len()
189                    || !bytes[i + w.len()].is_ascii_alphanumeric() && bytes[i + w.len()] != b'_';
190                if word_start_ok && word_end_ok {
191                    result.push_str(replacement);
192                    i += w.len();
193                    continue;
194                }
195            }
196            if i < bytes.len() {
197                result.push(bytes[i] as char);
198            }
199            i += 1;
200        }
201        result
202    }

204    /// If line is require(cond); or require(cond, "msg"); emit DAL if (!cond) { throw ... }; else None.
205    fn convert_require_line(&self, line: &str) -> Option<String> {
206        let line = line.trim_end_matches(';').trim();
207        if !line.starts_with("require(") {
208            return None;
209        }
210        let inner = line["require(".len()..].trim();
211        let (cond, msg) = if let Some(comma_pos) = inner.find(", ") {
212            let (c, m) = inner.split_at(comma_pos);
213            (c.trim(), m[1..].trim().trim_matches('"'))
214        } else {
215            (
216                inner.trim_end_matches(';').trim_end_matches(')').trim(),
217                "require failed",
218            )
219        };
220        if cond.is_empty() {
221            return None;
222        }
223        Some(format!("if (!({})) {{ throw \"{}\"; }}", cond, msg))
224    }

226    /// If line is addr.transfer(amount) or addr.call(...) emit DAL chain::call(...); else None.
227    fn convert_transfer_call_line(&self, line: &str) -> Option<String> {
228        let line = line.trim_end_matches(';').trim();
229        if let Some(transfer_pos) = line.find(".transfer(") {
230            let receiver = line[..transfer_pos].trim();
231            let open_paren = transfer_pos + ".transfer(".len() - 1;
232            if let Some(close) = Self::find_matching_paren(line, open_paren) {
233                let arg = line[open_paren + 1..close].trim();
234                return Some(format!(
235                    "chain::call(1, {}, \"transfer\", {{\"amount\": {}}});  // chain_id=1; adjust if needed",
236                    receiver, arg
237                ));
238            }
239        }
240        if line.contains(".call(") || line.contains(".call{") {
241            let dot_call = line
242                .find(".call(")
243                .or_else(|| line.find(".call{"))
244                .unwrap_or(0);
245            let addr = line[..dot_call].trim();
246            return Some(format!(
247                "chain::call(1, {}, \"call\", {{}});  // Solidity .call - fill function name and args as needed",
248                addr
249            ));
250        }
251        None
252    }

254    /// Index of the closing ')' that matches the '(' at open_idx.
255    fn find_matching_paren(s: &str, open_idx: usize) -> Option<usize> {
256        let bytes = s.as_bytes();
257        if open_idx >= bytes.len() || bytes[open_idx] != b'(' {
258            return None;
259        }
260        let mut depth = 1u32;
261        for (i, &b) in bytes.iter().enumerate().skip(open_idx + 1) {
262            match b {
263                b'(' => depth += 1,
264                b')' => {
265                    depth -= 1;
266                    if depth == 0 {
267                        return Some(i);
268                    }
269                }
270                _ => {}
271            }
272        }
273        None
274    }

276    /// If line is emit EventName(...); pass through (DAL uses same emit syntax). Else None.
277    fn convert_emit_line(&self, line: &str) -> Option<String> {
278        let line = line.trim_end_matches(';').trim();
279        if line.starts_with("emit ") {
280            let mut s = line.to_string();
281            if !s.ends_with(';') {
282                s.push(';');
283            }
284            return Some(s);
285        }
286        None
287    }

289    /// Pass-through for control-flow: if/for/while/else, do-while, try/catch, unchecked/assembly, and lone `{`/`}` (body already has storage→self and msg.sender replaced).
290    fn convert_control_flow_line(&self, line: &str) -> Option<String> {
291        let trimmed = line.trim();
292        if trimmed.is_empty() {
293            return None;
294        }
295        if trimmed == "}" || trimmed == "{" {
296            return Some(trimmed.to_string());
297        }
298        if trimmed.starts_with("if ")
299            || trimmed.starts_with("if(")
300            || trimmed.starts_with("for ")
301            || trimmed.starts_with("for(")
302            || trimmed.starts_with("while ")
303            || trimmed.starts_with("while(")
304            || trimmed.starts_with("do ")
305            || trimmed.starts_with("} while(")
306            || trimmed.starts_with("else ")
307            || trimmed.starts_with("else{")
308            || trimmed.starts_with("else {")
309        {
310            return Some(trimmed.to_string());
311        }
312        if trimmed.starts_with("} else") {
313            return Some(trimmed.to_string());
314        }
315        if trimmed.starts_with("try ")
316            || trimmed.starts_with("} catch")
317            || trimmed.starts_with("catch ")
318            || trimmed.starts_with("catch(")
319        {
320            return Some(trimmed.to_string());
321        }
322        if trimmed.starts_with("unchecked ")
323            || trimmed.starts_with("unchecked{")
324            || trimmed.starts_with("unchecked {")
325        {
326            return Some(trimmed.to_string());
327        }
328        if trimmed.starts_with("assembly ")
329            || trimmed.starts_with("assembly{")
330            || trimmed.starts_with("assembly {")
331        {
332            return Some(trimmed.to_string());
333        }
334        None
335    }

337    /// If line is revert(); or revert("msg"); emit DAL throw "msg"; else None.
338    fn convert_revert_line(&self, line: &str) -> Option<String> {
339        let line = line.trim_end_matches(';').trim();
340        if !line.starts_with("revert(") {
341            return None;
342        }
343        let inner = line["revert(".len()..]
344            .trim_end_matches(')')
345            .trim()
346            .trim_matches('"');
347        let msg = if inner.is_empty() { "reverted" } else { inner };
348        Some(format!("throw \"{}\";", msg))
349    }

351    /// Pass-through for assignments and return; replace msg.sender with chain::caller().
352    /// Returns None to emit as comment if line is empty, only braces, or not a simple statement.
353    fn convert_generic_line(&self, line: &str) -> Option<String> {
354        let line = line.trim_end_matches(';').trim();
355        if line.is_empty()
356            || line == "{"
357            || line == "}"
358            || line.starts_with("//")
359            || line.starts_with("/*")
360        {
361            return None;
362        }
363        let mut dal = line.to_string();
364        dal = dal.replace("msg.sender", "chain::caller()");
365        let is_msg_sender = dal != line;
366        let is_return = line.starts_with("return ");
367        let is_assignment = Self::looks_like_assignment(line);
368        if is_msg_sender || is_return || is_assignment {
369            if !dal.ends_with(';') && !dal.ends_with('}') {
370                dal.push(';');
371            }
372            Some(dal)
373        } else {
374            None
375        }
376    }

378    /// True if line looks like a single assignment (lhs = rhs), not a comparison (==, !=, <=, >=).
379    fn looks_like_assignment(line: &str) -> bool {
380        if let Some(idx) = line.find(" = ") {
381            let before = line[..idx].trim_end();
382            let after = line[idx + 3..].trim_start();
383            !before.ends_with('=')
384                && !before.ends_with('!')
385                && !before.ends_with('<')
386                && !before.ends_with('>')
387                && !after.starts_with('=')
388        } else {
389            false
390        }
391    }

393    fn generate_parameters(&self, params: &[DALParameter]) -> Result<String, String> {
394        if params.is_empty() {
395            return Ok(String::new());
396        }
397
398        let param_strings: Vec<String> = params
399            .iter()
400            .map(|p| format!("{}: {}", p.name, p.param_type))
401            .collect();
402
403        Ok(param_strings.join(", "))
404    }

406    fn generate_event(&self, event: &DALEvent) -> Result<String, String> {
407        let mut code = String::new();
408
409        code.push_str(&format!(
410            "    event {}({});\n",
411            event.name,
412            self.generate_parameters(&event.parameters)?
413        ));
414
415        Ok(code)
416    }
