17/// Convert Solidity file to DAL
18pub fn convert_file(input_path: &Path, output_path: &Path) -> Result<String, String> {
19    // Read Solidity file
20    let solidity_code = std::fs::read_to_string(input_path)
21        .map_err(|e| format!("Failed to read input file: {}", e))?;
22
23    // Parse Solidity
24    let parser = SolidityParser::new();
25    let solidity_ast = parser.parse(&solidity_code)?;
26
27    // Convert to DAL
28    let converter = SolidityConverter::new();
29    let dal_ast = converter.convert(solidity_ast)?;
30
31    // Generate DAL code
32    let generator = generator::DALGenerator::new();
33    let dal_code = generator.generate(dal_ast)?;
34
35    // Write output
36    std::fs::write(output_path, &dal_code)
37        .map_err(|e| format!("Failed to write output file: {}", e))?;
38
39    Ok(dal_code)
40}
