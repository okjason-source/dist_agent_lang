6/// Analysis Report
7#[derive(Debug, Clone)]
8pub struct AnalysisReport {
9    pub compatibility_score: f64,
10    pub warnings: Vec<String>,
11    pub errors: Vec<String>,
12    pub suggestions: Vec<String>,
13    pub unsupported_features: Vec<String>,
14    /// Import paths / library identifiers found in the source (e.g. "openzeppelin/...", "@openzeppelin/...").
15    pub used_libraries: Vec<String>,
16}
17
18/// Conversion Analyzer
19pub struct ConversionAnalyzer {
20    type_mapper: TypeMapper,
21}
22
23impl ConversionAnalyzer {
24    pub fn new() -> Self {
25        Self {
26            type_mapper: TypeMapper::new(),
27        }
28    }

30    /// Analyze Solidity AST for conversion compatibility
31    pub fn analyze(&self, ast: SolidityAST) -> Result<AnalysisReport, String> {
32        let mut warnings = Vec::new();
33        let mut errors = Vec::new();
34        let mut suggestions = Vec::new();
35        let mut unsupported_features = Vec::new();
36
37        // Check imports for OpenZeppelin / known libraries
38        let has_openzeppelin_import = ast.imports.iter().any(|s| {
39            let lower = s.to_lowercase();
40            lower.contains("openzeppelin") || lower.contains("@openzeppelin")
41        });
42        if has_openzeppelin_import {
43            suggestions.push("File imports OpenZeppelin - DAL has built-in reentrancy protection; modifiers can be mapped to @secure".to_string());
44        }
45
46        for contract in &ast.contracts {
47            self.analyze_contract(
48                contract,
49                &mut warnings,
50                &mut errors,
51                &mut suggestions,
52                &mut unsupported_features,
53            );
54        }
55
56        // Calculate compatibility score
57        let compatibility_score =
58            self.calculate_compatibility_score(&warnings, &errors, &unsupported_features);
59
60        let used_libraries = ast.imports.clone();
61
62        Ok(AnalysisReport {
63            compatibility_score,
64            warnings,
65            errors,
66            suggestions,
67            unsupported_features,
68            used_libraries,
69        })
70    }
71
72    fn analyze_contract(
73        &self,
74        contract: &Contract,
75        warnings: &mut Vec<String>,
76        errors: &mut Vec<String>,
77        suggestions: &mut Vec<String>,
78        unsupported_features: &mut Vec<String>,
79    ) {
80        // Check contract kind
81        match contract.kind {
82            super::parser::ContractKind::Interface => {
83                warnings.push(format!(
84                    "Interface '{}' - will be converted to service",
85                    contract.name
86                ));
87            }
88            super::parser::ContractKind::Abstract => {
89                warnings.push(format!(
90                    "Abstract contract '{}' - abstract features may need manual conversion",
91                    contract.name
92                ));
93            }
94            super::parser::ContractKind::Library => {
95                unsupported_features.push(format!(
96                    "Library '{}' - libraries not directly supported",
97                    contract.name
98                ));
99            }
100            _ => {}
101        }
102
103        // Check inheritance
104        if !contract.inheritance.is_empty() {
105            warnings.push(format!(
106                "Contract '{}' uses inheritance - may need manual review",
107                contract.name
108            ));
109        }
110
111        // Analyze functions
112        for func in &contract.functions {
113            self.analyze_function(func, warnings, errors, suggestions);
114        }
115
116        // Check for common patterns (reentrancy: centralized in security module)
117        if super::security::SecurityConverter::new().has_reentrancy_risk(contract) {
118            suggestions.push(format!(
119                "Contract '{}' should use @secure attribute for reentrancy protection",
120                contract.name
121            ));
122        }
123
124        // Check for OpenZeppelin patterns
125        if self.uses_openzeppelin(contract) {
126            suggestions.push(format!(
127                "Contract '{}' uses OpenZeppelin - consider using DAL built-in security features",
128                contract.name
129            ));
130        }
131    }
132
133    fn analyze_function(
134        &self,
135        func: &Function,
136        warnings: &mut Vec<String>,
137        _errors: &mut Vec<String>,
138        suggestions: &mut Vec<String>,
139    ) {
140        // Check visibility
141        if matches!(func.visibility, Visibility::External) {
142            warnings.push(format!(
143                "Function '{}' is external - will be converted to @public",
144                func.name
145            ));
146        }
147
148        // Check mutability
149        if matches!(func.mutability, super::parser::Mutability::Payable) {
150            suggestions.push(format!(
151                "Function '{}' is payable - ensure proper handling in DAL",
152                func.name
153            ));
154        }
155
156        // Check modifiers
157        if !func.modifiers.is_empty() {
158            warnings.push(format!(
159                "Function '{}' uses modifiers - may need manual conversion",
160                func.name
161            ));
162        }
163
164        // Check return types
165        for ret in &func.returns {
166            if !self.type_mapper.is_supported(&ret.param_type) {
167                warnings.push(format!(
168                    "Function '{}' returns unsupported type '{}'",
169                    func.name, ret.param_type
170                ));
171            }
172        }
173
174        // Check parameters
175        for param in &func.parameters {
176            if !self.type_mapper.is_supported(&param.param_type) {
177                warnings.push(format!(
178                    "Function '{}' parameter '{}' has unsupported type '{}'",
179                    func.name, param.name, param.param_type
180                ));
181            }
182        }
183    }
184
185    fn uses_openzeppelin(&self, contract: &Contract) -> bool {
186        // Check modifiers for OpenZeppelin patterns; imports are checked in analyze() via ast.imports
187        contract.functions.iter().any(|f| {
188            f.modifiers.iter().any(|m| {
189                m.contains("onlyOwner") || m.contains("nonReentrant") || m.contains("whenNotPaused")
190            })
191        })
192    }
193
194    fn calculate_compatibility_score(
195        &self,
196        warnings: &[String],
197        errors: &[String],
198        unsupported: &[String],
199    ) -> f64 {
200        let mut score = 100.0;
201
202        // Deduct for errors
203        score -= (errors.len() as f64) * 20.0;
204
205        // Deduct for unsupported features
206        score -= (unsupported.len() as f64) * 15.0;
207
208        // Deduct for warnings
209        score -= (warnings.len() as f64) * 5.0;
210
211        // Ensure score is between 0 and 100
212        score.max(0.0).min(100.0)
213    }
214}
