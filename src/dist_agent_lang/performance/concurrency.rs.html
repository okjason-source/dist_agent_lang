18pub struct ThreadPool {
19    workers: Vec<Worker>,
20    sender: std::sync::mpsc::Sender<Message>,
21    stats: Arc<Mutex<ThreadPoolStats>>,
22}
23
24enum Message {
25    NewJob(Box<dyn FnOnce() + Send + 'static>),
26    Terminate,
27}
28
29struct Worker {
30    #[allow(dead_code)]
31    id: usize,
32    thread: Option<thread::JoinHandle<()>>,
33}
34
35impl ThreadPool {
36    pub fn new(size: usize) -> ThreadPool {
37        assert!(size > 0);
38
39        let (sender, receiver) = std::sync::mpsc::channel();
40        let receiver = Arc::new(Mutex::new(receiver));
41        let stats = Arc::new(Mutex::new(ThreadPoolStats {
42            total_threads: size,
43            active_threads: 0,
44            idle_threads: size,
45            queued_tasks: 0,
46            completed_tasks: 0,
47            failed_tasks: 0,
48            average_task_duration: Duration::ZERO,
49        }));
50
51        let mut workers = Vec::with_capacity(size);
52
53        for id in 0..size {
54            workers.push(Worker::new(id, Arc::clone(&receiver), Arc::clone(&stats)));
55        }
56
57        ThreadPool {
58            workers,
59            sender,
60            stats,
61        }
62    }
63
64    pub fn execute<F>(&self, f: F)
65    where
66        F: FnOnce() + Send + 'static,
67    {
68        let job = Box::new(f);
69        self.sender.send(Message::NewJob(job)).unwrap();
70
71        // Update stats
72        if let Ok(mut stats) = self.stats.lock() {
73            stats.queued_tasks += 1;
74        }
75    }
76
77    pub fn get_stats(&self) -> ThreadPoolStats {
78        if let Ok(stats) = self.stats.lock() {
79            stats.clone()
80        } else {
81            ThreadPoolStats {
82                total_threads: 0,
83                active_threads: 0,
84                idle_threads: 0,
85                queued_tasks: 0,
86                completed_tasks: 0,
87                failed_tasks: 0,
88                average_task_duration: Duration::ZERO,
89            }
90        }
91    }
92}

94impl Drop for ThreadPool {
95    fn drop(&mut self) {
96        for _ in &self.workers {
97            self.sender.send(Message::Terminate).unwrap();
98        }
99
100        for worker in &mut self.workers {
101            if let Some(thread) = worker.thread.take() {
102                thread.join().unwrap();
103            }
104        }
105    }
106}
107
108impl Worker {
109    fn new(
110        id: usize,
111        receiver: Arc<Mutex<std::sync::mpsc::Receiver<Message>>>,
112        stats: Arc<Mutex<ThreadPoolStats>>,
113    ) -> Worker {
114        let thread = thread::spawn(move || {
115            loop {
116                let message = receiver.lock().unwrap().recv().unwrap();
117
118                match message {
119                    Message::NewJob(job) => {
120                        // Update stats: worker becomes active
121                        if let Ok(mut stats) = stats.lock() {
122                            stats.active_threads += 1;
123                            stats.idle_threads = stats.idle_threads.saturating_sub(1);
124                        }
125
126                        let start_time = Instant::now();
127                        let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(job));
128                        let duration = start_time.elapsed();
129
130                        // Update stats: job completed
131                        if let Ok(mut stats) = stats.lock() {
132                            stats.active_threads = stats.active_threads.saturating_sub(1);
133                            stats.idle_threads += 1;
134                            stats.queued_tasks = stats.queued_tasks.saturating_sub(1);
135
136                            if result.is_ok() {
137                                stats.completed_tasks += 1;
138                            } else {
139                                stats.failed_tasks += 1;
140                            }
141
142                            // Update average duration
143                            let total_duration =
144                                stats.average_task_duration * stats.completed_tasks as u32;
145                            let new_total = total_duration + duration;
146                            stats.average_task_duration =
147                                new_total / (stats.completed_tasks + 1) as u32;
148                        }
149                    }
150                    Message::Terminate => {
151                        break;
152                    }
153                }
154            }
155        });
156
157        Worker {
158            id,
159            thread: Some(thread),
160        }
161    }
162}

165pub struct AsyncScheduler {

167    task_queue: Arc<Mutex<VecDeque<AsyncTask>>>,
168    running_tasks: Arc<Mutex<Vec<RunningTask>>>,
169}

171pub struct AsyncTask {
172    pub id: usize,
173    pub name: String,
174    pub priority: TaskPriority,
175    pub task: Box<dyn FnOnce() -> Result<(), String> + Send + 'static>,
176    pub created_at: Instant,
177}
178
179#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
180pub enum TaskPriority {
181    Low = 0,
182    Normal = 1,
183    High = 2,
184    Critical = 3,
185}
186
187#[derive(Debug)]
188struct RunningTask {
189    id: usize,
190    #[allow(dead_code)]
191    start_time: Instant,
192    #[allow(dead_code)]
193    thread_id: usize,
194}
195
196impl AsyncScheduler {
197    pub fn new(thread_count: usize) -> Self {
198        Self {
199            thread_pool: ThreadPool::new(thread_count),
200            task_queue: Arc::new(Mutex::new(VecDeque::new())),
201            running_tasks: Arc::new(Mutex::new(Vec::new())),
202        }
203    }
204
205    pub fn schedule<F>(&self, name: &str, priority: TaskPriority, task: F) -> usize
206    where
207        F: FnOnce() -> Result<(), String> + Send + 'static,
208    {
209        static NEXT_ID: AtomicUsize = AtomicUsize::new(0);
210        let task_id = NEXT_ID.fetch_add(1, Ordering::SeqCst);
211
212        let async_task = AsyncTask {
213            id: task_id,
214            name: name.to_string(),
215            priority,
216            task: Box::new(task),
217            created_at: Instant::now(),
218        };
219
220        if let Ok(mut queue) = self.task_queue.lock() {
221            // Insert based on priority (higher priority first)
222            let mut insert_index = None;
223            for (i, existing_task) in queue.iter().enumerate() {
224                if priority > existing_task.priority {
225                    insert_index = Some(i);
226                    break;
227                }
228            }
229
230            if let Some(index) = insert_index {
231                queue.insert(index, async_task);
232            } else {
233                queue.push_back(async_task);
234            }
235        }
236
237        self.process_queue();
238        task_id
239    }
240
241    fn process_queue(&self) {
242        if let Ok(mut queue) = self.task_queue.lock() {
243            if let Some(task) = queue.pop_front() {
244                let task_id = task.id;
245                let task_name = task.name.clone();
246
247                let running_tasks = self.running_tasks.clone();
248                self.thread_pool.execute(move || {
249                    let start_time = Instant::now();
250
251                    // Add to running tasks
252                    if let Ok(mut running) = running_tasks.lock() {
253                        running.push(RunningTask {
254                            id: task_id,
255                            start_time,
256                            thread_id: std::time::Instant::now().elapsed().as_nanos() as usize,
257                        });
258                    }
259
260                    // Execute task
261                    let result = (task.task)();
262
263                    // Remove from running tasks
264                    if let Ok(mut running) = running_tasks.lock() {
265                        running.retain(|t| t.id != task_id);
266                    }
267
268                    match result {
269                        Ok(_) => println!("Task '{}' completed successfully", task_name),
270                        Err(e) => println!("Task '{}' failed: {}", task_name, e),
271                    }
272                });
273            }
274        }
275    }
276
277    pub fn get_task_status(&self, task_id: usize) -> TaskStatus {
278        // Check if task is running
279        if let Ok(running) = self.running_tasks.lock() {
280            if running.iter().any(|t| t.id == task_id) {
281                return TaskStatus::Running;
282            }
283        }
284
285        // Check if task is queued
286        if let Ok(queue) = self.task_queue.lock() {
287            if queue.iter().any(|t| t.id == task_id) {
288                return TaskStatus::Queued;
289            }
290        }
291
292        TaskStatus::Completed
293    }
294
295    pub fn cancel_task(&self, task_id: usize) -> bool {
296        // Remove from queue
297        if let Ok(mut queue) = self.task_queue.lock() {
298            queue.retain(|t| t.id != task_id);
299        }
300
301        // Note: Can't cancel running tasks in this simple implementation
302        true
303    }
304}

307pub enum TaskStatus {
308    Queued,
309    Running,
310    Completed,
311    Failed,
312    Cancelled,
313}

315// Parallel execution utilities
316pub struct ParallelExecutor;
317
318impl ParallelExecutor {
319    pub fn map_parallel<T, R, F>(items: Vec<T>, f: F, num_threads: usize) -> Vec<R>
320    where
321        T: Send + Sync + Clone + 'static,
322        R: Send + Sync + 'static,
323        F: Fn(T) -> R + Send + Sync + 'static,
324    {
325        let thread_pool = ThreadPool::new(num_threads);
326        let (sender, receiver) = std::sync::mpsc::channel();
327        let items = Arc::new(items);
328        let f = Arc::new(f);
329
330        // Submit tasks
331        for i in 0..items.len() {
332            let items = Arc::clone(&items);
333            let f = Arc::clone(&f);
334            let sender = sender.clone();
335
336            thread_pool.execute(move || {
337                let item = items[i].clone();
338                let result = f(item);
339                sender.send((i, result)).unwrap();
340            });
341        }
342
343        // Collect results
344        let mut results = Vec::new();
345        results.resize_with(items.len(), || unsafe { std::mem::zeroed() });
346
347        for _ in 0..items.len() {
348            if let Ok((index, result)) = receiver.recv() {
349                results[index] = result;
350            }
351        }
352
353        results
354    }
355
356    pub fn reduce_parallel<T, F>(items: Vec<T>, f: F, num_threads: usize) -> T
357    where
358        T: Send + Sync + Clone + 'static,
359        F: Fn(T, T) -> T + Send + Sync + Clone + 'static,
360    {
361        if items.is_empty() {
362            panic!("Cannot reduce empty vector");
363        }
364
365        if items.len() == 1 {
366            return items[0].clone();
367        }
368
369        // Divide and conquer approach
370        let chunk_size = items.len().div_ceil(num_threads);
371        let chunks: Vec<Vec<T>> = items
372            .chunks(chunk_size)
373            .map(|chunk| chunk.to_vec())
374            .collect();
375
376        let thread_pool = ThreadPool::new(num_threads);
377        let (sender, receiver) = std::sync::mpsc::channel();
378
379        let chunks_len = chunks.len();
380        // Process each chunk
381        for chunk in chunks {
382            let sender = sender.clone();
383            let f_clone = f.clone();
384            thread_pool.execute(move || {
385                let result = chunk.into_iter().reduce(f_clone).unwrap();
386                sender.send(result).unwrap();
387            });
388        }
389
390        // Combine results
391        let mut final_result = receiver.recv().unwrap();
392        for _ in 1..chunks_len {
393            if let Ok(result) = receiver.recv() {
394                final_result = f(final_result, result);
395            }
396        }
397
398        final_result
399    }

402// Performance monitoring for concurrency
403pub struct ConcurrencyProfiler {
404    thread_pool_stats: Arc<Mutex<Vec<ThreadPoolStats>>>,
405    task_execution_times: Arc<Mutex<Vec<Duration>>>,
406}
407
408impl ConcurrencyProfiler {
409    pub fn new() -> Self {
410        Self {
411            thread_pool_stats: Arc::new(Mutex::new(Vec::new())),
412            task_execution_times: Arc::new(Mutex::new(Vec::new())),
413        }
414    }
415
416    pub fn record_thread_pool_stats(&self, stats: ThreadPoolStats) {
417        if let Ok(mut stats_vec) = self.thread_pool_stats.lock() {
418            stats_vec.push(stats);
419            // Keep only last 100 entries
420            if stats_vec.len() > 100 {
421                stats_vec.remove(0);
422            }
423        }
424    }
425
426    pub fn record_task_execution_time(&self, duration: Duration) {
427        if let Ok(mut times) = self.task_execution_times.lock() {
428            times.push(duration);
429            // Keep only last 1000 entries
430            if times.len() > 1000 {
431                times.remove(0);
432            }
433        }
434    }
435
436    pub fn generate_report(&self) -> String {
437        let mut report = String::new();
438        report.push_str("Concurrency Performance Report\n");
439        report.push_str("=============================\n\n");
440
441        // Thread pool statistics
442        if let Ok(stats_vec) = self.thread_pool_stats.lock() {
443            if let Some(latest_stats) = stats_vec.last() {
444                report.push_str("Thread Pool Status:\n");
445                report.push_str(&format!(
446                    "  Total Threads: {}\n",
447                    latest_stats.total_threads
448                ));
449                report.push_str(&format!(
450                    "  Active Threads: {}\n",
451                    latest_stats.active_threads
452                ));
453                report.push_str(&format!("  Idle Threads: {}\n", latest_stats.idle_threads));
454                report.push_str(&format!("  Queued Tasks: {}\n", latest_stats.queued_tasks));
455                report.push_str(&format!(
456                    "  Completed Tasks: {}\n",
457                    latest_stats.completed_tasks
458                ));
459                report.push_str(&format!("  Failed Tasks: {}\n", latest_stats.failed_tasks));
460                report.push_str(&format!(
461                    "  Average Task Duration: {:?}\n",
462                    latest_stats.average_task_duration
463                ));
464            }
465        }
466
467        // Task execution time statistics
468        if let Ok(times) = self.task_execution_times.lock() {
469            if !times.is_empty() {
470                let total_time: Duration = times.iter().sum();
471                let avg_time = total_time / times.len() as u32;
472                let min_time = times.iter().min().unwrap();
473                let max_time = times.iter().max().unwrap();
474
475                report.push_str("\nTask Execution Times:\n");
476                report.push_str(&format!("  Total Tasks: {}\n", times.len()));
477                report.push_str(&format!("  Average Time: {:?}\n", avg_time));
478                report.push_str(&format!("  Min Time: {:?}\n", min_time));
479                report.push_str(&format!("  Max Time: {:?}\n", max_time));
480            }
481        }
482
483        report
484    }
