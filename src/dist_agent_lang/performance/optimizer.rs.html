46    pub fn optimize(&self, ast: Program) -> OptimizationResult {
47        let mut current_ast = ast.clone();
48        let mut applied_optimizations = Vec::new();
49
50        for pass in &self.passes {
51            if pass.enabled && self.should_apply_pass(pass) {
52                let (optimized_ast, optimizations) = self.apply_pass(pass, &current_ast);
53                current_ast = optimized_ast;
54                applied_optimizations.extend(optimizations);
55            }
56        }
57
58        let performance_improvement = self.estimate_improvement(&applied_optimizations);
59
60        OptimizationResult {
61            original_ast: ast,
62            optimized_ast: current_ast,
63            optimizations_applied: applied_optimizations,
64            performance_improvement,
65        }
66    }

103    fn should_apply_pass(&self, pass: &OptimizationPass) -> bool {
104        match self.optimization_level {
105            OptimizationLevel::None => false,
106            OptimizationLevel::Basic => matches!(
107                pass.name.as_str(),
108                "constant_folding" | "dead_code_elimination"
109            ),
110            OptimizationLevel::Aggressive => true,
111            OptimizationLevel::Maximum => true,
112        }

115    fn apply_pass(&self, pass: &OptimizationPass, ast: &Program) -> (Program, Vec<String>) {
116        match pass.name.as_str() {
117            "constant_folding" => self.constant_folding(ast),
118            "dead_code_elimination" => self.dead_code_elimination(ast),
119            "function_inlining" => self.function_inlining(ast),
120            "common_subexpression_elimination" => self.common_subexpression_elimination(ast),
121            "loop_optimization" => self.loop_optimization(ast),
122            "strength_reduction" => self.strength_reduction(ast),
123            _ => (ast.clone(), vec![]),
124        }
125    }

127    fn constant_folding(&self, ast: &Program) -> (Program, Vec<String>) {
128        let mut optimized_statements = Vec::new();
129        let mut optimizations = Vec::new();
130
131        for statement in &ast.statements {
132            match statement {
133                Statement::Let(let_stmt) => {
134                    if let Some(folded_value) = self.fold_constant_expression(&let_stmt.value) {
135                        optimized_statements.push(Statement::Let(LetStatement {
136                            name: let_stmt.name.clone(),
137                            value: folded_value,
138                            line: let_stmt.line,
139                        }));
140                        optimizations.push(format!("Constant folded: {}", let_stmt.name));
141                    } else {
142                        optimized_statements.push(statement.clone());
143                    }
144                }
145                Statement::Expression(expr) => {
146                    if let Some(folded_value) = self.fold_constant_expression(expr) {
147                        optimized_statements.push(Statement::Expression(folded_value));
148                        optimizations.push("Constant folded expression".to_string());
149                    } else {
150                        optimized_statements.push(statement.clone());
151                    }
152                }
153                Statement::Function(func_stmt) => {
154                    let optimized_body = self.optimize_block(&func_stmt.body);
155                    optimized_statements.push(Statement::Function(FunctionStatement {
156                        name: func_stmt.name.clone(),
157                        parameters: func_stmt.parameters.clone(),
158                        return_type: func_stmt.return_type.clone(),
159                        body: optimized_body,
160                        attributes: func_stmt.attributes.clone(),
161                        is_async: func_stmt.is_async,
162                    }));
163                }
164                _ => optimized_statements.push(statement.clone()),
165            }
166        }
167
168        let len = optimized_statements.len();
169        (
170            Program {
171                statements: optimized_statements,
172                statement_spans: vec![None; len],
173            },
174            optimizations,
175        )
176    }

178    fn fold_constant_expression(&self, expr: &Expression) -> Option<Expression> {
179        match expr {
180            Expression::BinaryOp(left, op, right) => {
181                let left_folded = self.fold_constant_expression(left)?;
182                let right_folded = self.fold_constant_expression(right)?;
183
184                if let (Expression::Literal(lit1), Expression::Literal(lit2)) =
185                    (&left_folded, &right_folded)
186                {
187                    match (lit1, lit2, op) {
188                        (Literal::Int(a), Literal::Int(b), Operator::Plus) => {
189                            Some(Expression::Literal(Literal::Int(a + b)))
190                        }
191                        (Literal::Int(a), Literal::Int(b), Operator::Minus) => {
192                            Some(Expression::Literal(Literal::Int(a - b)))
193                        }
194                        (Literal::Int(a), Literal::Int(b), Operator::Star) => {
195                            Some(Expression::Literal(Literal::Int(a * b)))
196                        }
197                        (Literal::Int(a), Literal::Int(b), Operator::Slash) => {
198                            if *b != 0 {
199                                Some(Expression::Literal(Literal::Int(a / b)))
200                            } else {
201                                None
202                            }
203                        }
204                        _ => None,
205                    }
206                } else {
207                    None
208                }
209            }
210            Expression::UnaryOp(op, expr) => {
211                let folded = self.fold_constant_expression(expr)?;
212                if let Expression::Literal(lit) = &folded {
213                    match (lit, op) {
214                        (Literal::Int(a), Operator::Minus) => {
215                            Some(Expression::Literal(Literal::Int(-a)))
216                        }
217                        _ => None,
218                    }
219                } else {
220                    None
221                }
222            }
223            Expression::Literal(_) => Some(expr.clone()),
224            _ => None,
225        }
226    }

228    fn dead_code_elimination(&self, ast: &Program) -> (Program, Vec<String>) {
229        let mut optimized_statements = Vec::new();
230        let mut optimizations = Vec::new();
231
232        for statement in &ast.statements {
233            match statement {
234                Statement::If(if_stmt) => {
235                    // Check if condition is always true/false
236                    if let Some(constant_condition) =
237                        self.evaluate_boolean_expression(&if_stmt.condition)
238                    {
239                        if constant_condition {
240                            optimized_statements
241                                .push(Statement::Block(if_stmt.consequence.clone()));
242                            optimizations.push("Removed unreachable else branch".to_string());
243                        } else {
244                            if let Some(alternative) = &if_stmt.alternative {
245                                optimized_statements.push(Statement::Block(alternative.clone()));
246                            }
247                            optimizations.push("Removed unreachable if branch".to_string());
248                        }
249                    } else {
250                        optimized_statements.push(statement.clone());
251                    }
252                }
253                Statement::Expression(Expression::Literal(Literal::Null)) => {
254                    // Remove null expressions
255                    optimizations.push("Removed null expression".to_string());
256                }
257                _ => optimized_statements.push(statement.clone()),
258            }
259        }
260
261        let len = optimized_statements.len();
262        (
263            Program {
264                statements: optimized_statements,
265                statement_spans: vec![None; len],
266            },
267            optimizations,
268        )
269    }

271    fn evaluate_boolean_expression(&self, expr: &Expression) -> Option<bool> {
272        match expr {
273            Expression::Literal(Literal::Bool(b)) => Some(*b),
274            Expression::BinaryOp(left, op, right) => {
275                let left_val = self.evaluate_boolean_expression(left)?;
276                let right_val = self.evaluate_boolean_expression(right)?;
277
278                match op {
279                    Operator::And => Some(left_val && right_val),
280                    Operator::Or => Some(left_val || right_val),
281                    _ => None,
282                }
283            }
284            _ => None,
285        }
286    }

288    fn function_inlining(&self, ast: &Program) -> (Program, Vec<String>) {
289        // Simple function inlining for small functions
290        let mut optimized_statements = Vec::new();
291        let optimizations = Vec::new();
292        let mut function_map = HashMap::new();
293
294        // First pass: collect small functions
295        for statement in &ast.statements {
296            if let Statement::Function(func_stmt) = statement {
297                if self.should_inline_function(func_stmt) {
298                    function_map.insert(func_stmt.name.clone(), func_stmt.clone());
299                }
300            }
301        }
302
303        // Second pass: inline function calls
304        for statement in &ast.statements {
305            match statement {
306                Statement::Function(func_stmt) => {
307                    if !function_map.contains_key(&func_stmt.name) {
308                        optimized_statements.push(statement.clone());
309                    }
310                }
311                Statement::Expression(expr) => {
312                    let inlined_expr = self.inline_function_calls(expr, &function_map);
313                    optimized_statements.push(Statement::Expression(inlined_expr));
314                }
315                _ => optimized_statements.push(statement.clone()),
316            }
317        }
318
319        let len = optimized_statements.len();
320        (
321            Program {
322                statements: optimized_statements,
323                statement_spans: vec![None; len],
324            },
325            optimizations,
326        )
327    }

329    fn should_inline_function(&self, func_stmt: &FunctionStatement) -> bool {
330        // Inline functions with 3 or fewer statements and no parameters
331        func_stmt.parameters.is_empty() && func_stmt.body.statements.len() <= 3
332    }

334    fn inline_function_calls(
335        &self,
336        expr: &Expression,
337        function_map: &HashMap<String, FunctionStatement>,
338    ) -> Expression {
339        match expr {
340            Expression::FunctionCall(call) => {
341                if let Some(func) = function_map.get(&call.name) {
342                    // Simple inlining: replace function call with function body
343                    if call.arguments.is_empty() {
344                        // For now, just return the first statement's expression if it's a return
345                        if let Some(Statement::Return(ret_stmt)) = func.body.statements.first() {
346                            if let Some(value) = &ret_stmt.value {
347                                return value.clone();
348                            }
349                        }
350                    }
351                }
352                expr.clone()
353            }
354            _ => expr.clone(),
355        }
356    }

358    fn common_subexpression_elimination(&self, ast: &Program) -> (Program, Vec<String>) {
359        // Simple CSE: find identical expressions and replace with variables
360        let mut optimized_statements = Vec::new();
361        let mut optimizations = Vec::new();
362        let mut expression_cache: HashMap<String, String> = HashMap::new();
363
364        for statement in &ast.statements {
365            match statement {
366                Statement::Let(let_stmt) => {
367                    let cache_key = self.expression_to_string(&let_stmt.value);
368                    if let Some(existing_var) = expression_cache.get(&cache_key) {
369                        // Replace with existing variable
370                        optimized_statements.push(Statement::Let(LetStatement {
371                            name: let_stmt.name.clone(),
372                            value: Expression::Identifier(existing_var.clone()),
373                            line: let_stmt.line,
374                        }));
375                        optimizations.push(format!("CSE: {} = {}", let_stmt.name, existing_var));
376                    } else {
377                        expression_cache.insert(cache_key, let_stmt.name.clone());
378                        optimized_statements.push(statement.clone());
379                    }
380                }
381                _ => optimized_statements.push(statement.clone()),
382            }
383        }
384
385        let len = optimized_statements.len();
386        (
387            Program {
388                statements: optimized_statements,
389                statement_spans: vec![None; len],
390            },
391            optimizations,
392        )
393    }

395    fn expression_to_string(&self, expr: &Expression) -> String {
396        match expr {
397            Expression::Literal(lit) => format!("{:?}", lit),
398            Expression::Identifier(id) => id.clone(),
399            Expression::BinaryOp(left, op, right) => {
400                format!("({:?} {:?} {:?})", left, op, right)
401            }
402            _ => format!("{:?}", expr),
403        }
404    }

415    fn strength_reduction(&self, ast: &Program) -> (Program, Vec<String>) {
416        let mut optimized_statements = Vec::new();
417        let optimizations = Vec::new();
418
419        for statement in &ast.statements {
420            match statement {
421                Statement::Expression(expr) => {
422                    let reduced_expr = self.reduce_expression_strength(expr);
423                    optimized_statements.push(Statement::Expression(reduced_expr));
424                }
425                _ => optimized_statements.push(statement.clone()),
426            }
427        }
428
429        let len = optimized_statements.len();
430        (
431            Program {
432                statements: optimized_statements,
433                statement_spans: vec![None; len],
434            },
435            optimizations,
436        )
437    }

439    fn reduce_expression_strength(&self, expr: &Expression) -> Expression {
440        match expr {
441            Expression::BinaryOp(left, op, right) => {
442                match op {
443                    Operator::Star => {
444                        // Replace multiplication by 2 with left shift
445                        if let Expression::Literal(Literal::Int(2)) = **right {
446                            Expression::BinaryOp(
447                                left.clone(),
448                                Operator::Star,
449                                Box::new(Expression::Literal(Literal::Int(2))),
450                            )
451                        } else if let Expression::Literal(Literal::Int(2)) = **left {
452                            Expression::BinaryOp(
453                                right.clone(),
454                                Operator::Star,
455                                Box::new(Expression::Literal(Literal::Int(2))),
456                            )
457                        } else {
458                            expr.clone()
459                        }
460                    }
461                    _ => expr.clone(),
462                }
463            }
464            _ => expr.clone(),
465        }
466    }

468    fn optimize_block(&self, block: &BlockStatement) -> BlockStatement {
469        let mut optimized_statements = Vec::new();
470
471        for statement in &block.statements {
472            match statement {
473                Statement::Let(let_stmt) => {
474                    if let Some(folded_value) = self.fold_constant_expression(&let_stmt.value) {
475                        optimized_statements.push(Statement::Let(LetStatement {
476                            name: let_stmt.name.clone(),
477                            value: folded_value,
478                            line: let_stmt.line,
479                        }));
480                    } else {
481                        optimized_statements.push(statement.clone());
482                    }
483                }
484                _ => optimized_statements.push(statement.clone()),
485            }
486        }
487
488        BlockStatement {
489            statements: optimized_statements,
490        }
491    }

493    fn estimate_improvement(&self, optimizations: &[String]) -> f64 {
494        let mut improvement: f64 = 0.0;
495
496        for opt in optimizations {
497            if opt.contains("Constant folded") {
498                improvement += 5.0; // 5% improvement for constant folding
499            } else if opt.contains("Removed unreachable") {
500                improvement += 3.0; // 3% improvement for dead code elimination
501            } else if opt.contains("CSE") {
502                improvement += 2.0; // 2% improvement for common subexpression elimination
503            } else if opt.contains("inlined") {
504                improvement += 8.0; // 8% improvement for function inlining
505            }
506        }
507
508        improvement.min(50.0_f64) // Cap at 50% improvement
509    }

515impl OptimizationUtils {
516    pub fn analyze_complexity(ast: &Program) -> usize {
517        let mut complexity = 0;
518
519        for statement in &ast.statements {
520            complexity += Self::statement_complexity(statement);
521        }
522
523        complexity
524    }
525
526    fn statement_complexity(statement: &Statement) -> usize {
527        match statement {
528            Statement::Function(func_stmt) => 1 + Self::block_complexity(&func_stmt.body),
529            Statement::If(if_stmt) => {
530                2 + Self::block_complexity(&if_stmt.consequence)
531                    + if_stmt
532                        .alternative
533                        .as_ref()
534                        .map_or(0, Self::block_complexity)
535            }
536            Statement::Let(_) => 1,
537            Statement::Return(_) => 1,
538            Statement::Expression(_) => 1,
539            Statement::Block(block) => Self::block_complexity(block),
540            _ => 1,
541        }
542    }
543
544    fn block_complexity(block: &BlockStatement) -> usize {
545        block
546            .statements
547            .iter()
548            .map(Self::statement_complexity)
549            .sum()
550    }
551
552    pub fn estimate_optimization_potential(ast: &Program) -> f64 {
553        let complexity = Self::analyze_complexity(ast);
554        let constant_expressions = Self::count_constant_expressions(ast);
555        let function_calls = Self::count_function_calls(ast);
556
557        let mut potential = 0.0;
558
559        // More complex code has more optimization potential
560        potential += (complexity as f64 * 0.5).min(20.0);
561
562        // Constant expressions can be folded
563        potential += constant_expressions as f64 * 3.0;
564
565        // Function calls can be inlined
566        potential += function_calls as f64 * 2.0;
567
568        potential.min(50.0) // Cap at 50%
569    }
570
571    fn count_constant_expressions(ast: &Program) -> usize {
572        let mut count = 0;
573
574        for statement in &ast.statements {
575            count += Self::count_constants_in_statement(statement);
576        }
577
578        count
579    }
580
581    fn count_constants_in_statement(statement: &Statement) -> usize {
582        match statement {
583            Statement::Let(let_stmt) => Self::count_constants_in_expression(&let_stmt.value),
584            Statement::Expression(expr) => Self::count_constants_in_expression(expr),
585            Statement::Function(func_stmt) => Self::block_complexity(&func_stmt.body),
586            _ => 0,
587        }
588    }
589
590    fn count_constants_in_expression(expr: &Expression) -> usize {
591        match expr {
592            Expression::Literal(_) => 1,
593            Expression::BinaryOp(left, _, right) => {
594                Self::count_constants_in_expression(left)
595                    + Self::count_constants_in_expression(right)
596            }
597            Expression::UnaryOp(_, expr) => Self::count_constants_in_expression(expr),
598            _ => 0,
599        }
600    }
601
602    fn count_function_calls(ast: &Program) -> usize {
603        let mut count = 0;
604
605        for statement in &ast.statements {
606            count += Self::count_calls_in_statement(statement);
607        }
608
609        count
610    }
611
612    fn count_calls_in_statement(statement: &Statement) -> usize {
613        match statement {
614            Statement::Expression(expr) => Self::count_calls_in_expression(expr),
615            Statement::Function(func_stmt) => Self::count_calls_in_block(&func_stmt.body),
616            _ => 0,
617        }
618    }
619
620    fn count_calls_in_expression(expr: &Expression) -> usize {
621        match expr {
622            Expression::FunctionCall(_) => 1,
623            Expression::BinaryOp(left, _, right) => {
624                Self::count_calls_in_expression(left) + Self::count_calls_in_expression(right)
625            }
626            _ => 0,
627        }
628    }
629
630    fn count_calls_in_block(block: &BlockStatement) -> usize {
631        block
632            .statements
633            .iter()
634            .map(Self::count_calls_in_statement)
635            .sum()
636    }
