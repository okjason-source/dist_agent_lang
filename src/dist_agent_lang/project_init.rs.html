1//! Project init: run `dal init` logic from code (CLI and agent dal_init tool).
2//! Templates: general (main.dal), chain (chain.dal), iot (iot.dal), agent (agent.dal + evolve).
3
4use std::path::Path;
5
6/// Run init for the given template in `dir`. Additive; does not overwrite existing files.
7/// Returns a short success message or an error.
8/// Templates: "general" | "dal" (main.dal), "chain" (chain.dal), "iot" (iot.dal), "agent" (agent.dal + evolve).
9pub fn run_init(template: &str, dir: &Path) -> Result<String, String> {
10    let template = template.trim().to_lowercase();
11    let template = template.as_str();
12    match template {
13        "general" | "dal" | "" => run_init_general(dir),
14        "chain" => run_init_chain(dir),
15        "iot" => run_init_iot(dir),
16        "agent" => run_init_agent(dir),
17        _ => Err(format!(
18            "Unknown template '{}'. Use: general, chain, iot, agent (e.g. dal init --template chain)",
19            template
20        )),
21    }
22}

24fn write_if_missing(dir: &Path, name: &str, contents: &str) -> std::io::Result<()> {
25    let p = dir.join(name);
26    if !p.exists() {
27        std::fs::write(p, contents)?;
28    }
29    Ok(())
30}

32fn ensure_gitignore(dir: &Path) -> std::io::Result<()> {
33    let p = dir.join(".gitignore");
34    if p.exists() {
35        if let Ok(content) = std::fs::read_to_string(&p) {
36            if !content.lines().any(|l| l.trim() == ".env") {
37                let mut f = std::fs::OpenOptions::new().append(true).open(&p)?;
38                use std::io::Write;
39                writeln!(f, "\n# Local env (secrets)\n.env")?;
40            }
41        }
42    } else {
43        std::fs::write(p, "# Local env (do not commit)\n.env\n")?;
44    }
45    Ok(())
46}

48fn run_init_general(dir: &Path) -> Result<String, String> {
49    if dir.join("dal.toml").exists() {
50        return Err("Project already initialized (dal.toml exists)".to_string());
51    }
52    let dal_toml = r#"[package]
53name = "my-project"
54version = "0.1.0"
55authors = []
56
57[dependencies]
58# Add dependencies here
59"#;
60    write_if_missing(dir, "dal.toml", dal_toml).map_err(|e| e.to_string())?;
61
62    let main_dal = r#"// Main entry point
63
64fn main() {
65    print("Hello from dist_agent_lang!");
66}
67
68main();
69"#;
70    write_if_missing(dir, "main.dal", main_dal).map_err(|e| e.to_string())?;
71
72    write_if_missing(
73        dir,
74        "README.md",
75        "# My DAL Project\n\nA dist_agent_lang project.\n\nSet env as needed (e.g. from `.env`); see `.env.example`.\n",
76    )
77    .map_err(|e| e.to_string())?;
78
79    write_if_missing(
80        dir,
81        ".env.example",
82        r#"# DAL project env (copy to .env and set values; do not commit .env)
83# Optional: for ai:: / assist
84# OPENAI_API_KEY=
85# ANTHROPIC_API_KEY=
86"#,
87    )
88    .map_err(|e| e.to_string())?;
89
90    write_if_missing(
91        dir,
92        ".env",
93        "# Local env — set values and do not commit (add .env to .gitignore)\n",
94    )
95    .map_err(|e| e.to_string())?;
96
97    ensure_gitignore(dir).map_err(|e| e.to_string())?;
98    Ok("Initialized DAL project (main.dal, dal.toml).".to_string())
99}

175fn run_init_agent(dir: &Path) -> Result<String, String> {
176    if !dir.join("dal.toml").exists() {
177        let dal_toml = r#"[package]
178name = "my-agent"
179version = "0.1.0"
180authors = []
181
182[dependencies]
183"#;
184        std::fs::write(dir.join("dal.toml"), dal_toml).map_err(|e| e.to_string())?;
185    }
186
187    let agent_toml = r#"# Agent project config (see docs/guides/AGENT_SETUP_AND_USAGE.md)
188
189[agent.sh]
190trust = "sandboxed"
191
192# Evolve: wired by default. To disable: comment out the line below.
193[agent]
194context_path = "./evolve.md"
195"#;
196    write_if_missing(dir, "agent.toml", agent_toml).map_err(|e| e.to_string())?;
197
198    let agent_dal = r#"// Agent behavior entry (run with: dal run agent.dal or dal agent serve)
199// Evolve: context_path in agent.toml. Use sh::run(cmd) for shell (respects [agent.sh] trust).
200
201use agent;
202
203fn main() {
204    let agent_id = agent::spawn({
205        "name": "my-agent",
206        "type": "worker",
207        "role": "Agent serve"
208    });
209    agent::set_serve_agent(agent_id);
210}
211
212main();
213"#;
214    write_if_missing(dir, "agent.dal", agent_dal).map_err(|e| e.to_string())?;
215
216    write_if_missing(
217        dir,
218        "README.md",
219        "# Agent project\n\n- agent.dal, agent.toml, evolve.md\n- dal run agent.dal | dal agent serve\n",
220    )
221    .map_err(|e| e.to_string())?;
222
223    if !dir.join("evolve.md").exists() {
224        let header = "# Agent context\n\n## Conversation\n\n## Action log\n\n| Time | Action | Detail | Result |\n|------|--------|--------|--------|\n";
225        std::fs::write(dir.join("evolve.md"), header).map_err(|e| e.to_string())?;
226    }
227
228    write_if_missing(
229        dir,
230        ".env.example",
231        r#"# Agent project env
232DAL_AGENT_SHELL_TRUST=sandboxed
233DAL_AGENT_CONTEXT_PATH=./evolve.md
234# OPENAI_API_KEY=
235# ANTHROPIC_API_KEY=
236"#,
237    )
238    .map_err(|e| e.to_string())?;
239
240    write_if_missing(
241        dir,
242        ".env",
243        "# Local env\nDAL_AGENT_SHELL_TRUST=sandboxed\nDAL_AGENT_CONTEXT_PATH=./evolve.md\n",
244    )
245    .map_err(|e| e.to_string())?;
246
247    ensure_gitignore(dir).map_err(|e| e.to_string())?;
248    Ok("Initialized agent project (agent.dal, agent.toml, evolve.md).".to_string())
249}
