8/// Best-effort guard that a path is contained by root.
9/// For existing paths, compares canonical paths to account for symlinks.
10/// For non-existing paths, falls back to lexical prefix check.
11fn path_is_within_root(root: &Path, path: &Path) -> bool {
12    if let (Ok(canon_root), Ok(canon_path)) = (root.canonicalize(), path.canonicalize()) {
13        return canon_path.starts_with(&canon_root);
14    }
15    path.starts_with(root)
16}

91/// Discover DAL projects and run configs in a workspace.
92/// Uses canonical form of workspace_root for path operations to avoid path traversal.
93pub fn discover_workspace(workspace_root: &Path) -> OrchestrationResponse {
94    let workspace_root = workspace_root
95        .canonicalize()
96        .unwrap_or_else(|_| workspace_root.to_path_buf());
97    let workspace_root = workspace_root.as_path();
98
99    let mut projects = Vec::new();
100    let mut run_configs = Vec::new();
101    let mut scripts = Vec::new();
102    let mut workflows = None;
103    let mut agent_evolve = None;
104
105    // Find dal.toml at root and in subfolders
106    let manifest_paths = find_manifests(workspace_root);
107
108    for manifest_path in manifest_paths {
109        let project_root = manifest_path
110            .parent()
111            .unwrap_or_else(|| Path::new("."))
112            .to_path_buf();
113
114        let project_name = crate::manifest::parse_package_info(&manifest_path)
115            .ok()
116            .map(|p| p.name);
117
118        let root_str = project_root.to_string_lossy().to_string();
119        let manifest_str = manifest_path.to_string_lossy().to_string();
120
121        projects.push(ProjectInfo {
122            r#type: "dal".to_string(),
123            root: root_str.clone(),
124            manifest_path: manifest_str.clone(),
125            name: project_name.clone(),
126        });
127
128        // Run configs: common entrypoints
129        let entrypoints = discover_entrypoints(&project_root);
130        for (path, label, is_serve) in entrypoints {
131            let rel = path
132                .strip_prefix(&project_root)
133                .unwrap_or(&path)
134                .to_string_lossy()
135                .to_string();
136            let id = format!("{}:{}", root_str, rel);
137            run_configs.push(RunConfig {
138                id: id.clone(),
139                label: label.clone(),
140                r#type: if is_serve {
141                    RunConfigType::Serve
142                } else {
143                    RunConfigType::Run
144                },
145                command: "dal".to_string(),
146                args: Some(if is_serve {
147                    vec![
148                        "serve".to_string(),
149                        rel.clone(),
150                        "--port".to_string(),
151                        "4040".to_string(),
152                    ]
153                } else {
154                    vec!["run".to_string(), rel.clone()]
155                }),
156                cwd: Some(root_str.clone()),
157            });
158            scripts.push(ScriptEntry {
159                path: path.to_string_lossy().to_string(),
160                label,
161            });
162        }
163
164        // Test config: one "Run tests" per project (runs `dal test`)
165        run_configs.push(RunConfig {
166            id: format!("{}:__tests__", root_str),
167            label: "Run tests".to_string(),
168            r#type: RunConfigType::Test,
169            command: "dal".to_string(),
170            args: Some(vec!["test".to_string()]),
171            cwd: Some(root_str.clone()),
172        });
173
174        // Agent/evolve: check agent.toml
175        let agent_toml = project_root.join("agent.toml");
176        if agent_toml.exists() {
177            if let Ok(content) = std::fs::read_to_string(&agent_toml) {
178                let evolve_path = parse_agent_context_path(&content, &project_root);
179                let context_len = evolve_path
180                    .as_ref()
181                    .and_then(|p| std::fs::read_to_string(p).ok())
182                    .map(|s| s.len());
183                agent_evolve = Some(AgentEvolveInfo {
184                    evolve_path,
185                    context_length: context_len,
186                });
187            }
188        }
189
190        // Workflows: check for workflows.dal and add run config per workflow
191        let workflows_dal = project_root.join("workflows.dal");
192        if workflows_dal.exists() {
193            if let Some(wfs) = parse_workflows_from_dal(&workflows_dal) {
194                workflows = Some(wfs.clone());
195                for wf in wfs {
196                    run_configs.push(RunConfig {
197                        id: format!("workflow:{}:{}", root_str, wf.id),
198                        label: format!("Workflow: {}", wf.name),
199                        r#type: RunConfigType::Workflow,
200                        command: "dal".to_string(),
201                        args: Some(vec!["run".to_string(), "workflows.dal".to_string()]),
202                        cwd: Some(root_str.clone()),
203                    });
204                }
205            }
206        }
207    }
208
209    // Cargo projects: discover Cargo.toml at workspace root and in direct subdirs
210    let cargo_roots = find_cargo_roots(workspace_root);
211    for cargo_root in cargo_roots {
212        let root_str = cargo_root.to_string_lossy().to_string();
213        for (suffix, label, args) in [
214            ("run", "Cargo: run", vec!["run".to_string()]),
215            ("test", "Cargo: test", vec!["test".to_string()]),
216            ("build", "Cargo: build", vec!["build".to_string()]),
217        ] {
218            run_configs.push(RunConfig {
219                id: format!("cargo:{}:{}", root_str, suffix),
220                label: label.to_string(),
221                r#type: RunConfigType::Cargo,
222                command: "cargo".to_string(),
223                args: Some(args),
224                cwd: Some(root_str.clone()),
225            });
226        }
227    }
228
229    OrchestrationResponse {
230        projects,
231        run_configs,
232        scripts: Some(scripts),
233        workflows,
234        agent_evolve,
235    }
236}

238/// Find directories that contain Cargo.toml (workspace root + immediate subdirs).
239fn find_cargo_roots(workspace_root: &Path) -> Vec<PathBuf> {
240    let mut out = Vec::new();
241    let root_cargo = workspace_root.join("Cargo.toml");
242    if path_is_within_root(workspace_root, &root_cargo) && root_cargo.exists() {
243        out.push(workspace_root.to_path_buf());
244    }
245    if path_is_within_root(workspace_root, workspace_root) {
246        if let Ok(entries) = std::fs::read_dir(workspace_root) {
247            for entry in entries.flatten() {
248                let path = entry.path();
249                if path.is_dir() {
250                    let manifest = path.join("Cargo.toml");
251                    if path_is_within_root(workspace_root, &manifest) && manifest.exists() {
252                        out.push(path);
253                    }
254                }
255            }
256        }
257    }
258    out
259}

261/// Find manifest paths recursively (root + immediate subdirs).
262fn find_manifests(workspace_root: &Path) -> Vec<PathBuf> {
263    let mut out = Vec::new();
264    let root_manifest = workspace_root.join("dal.toml");
265    if path_is_within_root(workspace_root, &root_manifest) && root_manifest.exists() {
266        out.push(root_manifest);
267    }
268    if path_is_within_root(workspace_root, workspace_root) {
269        if let Ok(entries) = std::fs::read_dir(workspace_root) {
270            for entry in entries.flatten() {
271                let path = entry.path();
272                if path.is_dir() {
273                    let manifest = path.join("dal.toml");
274                    if path_is_within_root(workspace_root, &manifest) && manifest.exists() {
275                        out.push(manifest);
276                    }
277                }
278            }
279        }
280    }
281    out
282}

284/// Discover entrypoint .dal files.
285fn discover_entrypoints(project_root: &Path) -> Vec<(PathBuf, String, bool)> {
286    let mut out = Vec::new();
287    let candidates = [
288        ("server.dal", "Serve server.dal", true),
289        ("main.dal", "Run main.dal", false),
290        ("agent.dal", "Run agent.dal", false),
291    ];
292    for (name, label, is_serve) in candidates {
293        let p = project_root.join(name);
294        if p.exists() {
295            out.push((p, label.to_string(), is_serve));
296        }
297    }
298    // Also scan for .dal files in root (excluding already discovered)
299    let seen: HashSet<&str> = candidates.iter().map(|(n, _, _)| *n).collect();
300    if let Ok(entries) = std::fs::read_dir(project_root) {
301        for entry in entries.flatten() {
302            let path = entry.path();
303            if path.is_file() {
304                if let Some(name) = path.file_name().and_then(|n| n.to_str()) {
305                    if name.ends_with(".dal") && !seen.contains(name) {
306                        let label = format!("Run {}", name);
307                        out.push((path.clone(), label, false));
308                    }
309                }
310            }
311        }
312    }
313    out
314}

316/// Parse context_path from agent.toml [agent] section.
317fn parse_agent_context_path(content: &str, project_root: &Path) -> Option<String> {
318    for line in content.lines() {
319        let line = line.trim();
320        if line.starts_with("context_path") {
321            if let Some(v) = line.split('=').nth(1) {
322                let path = v.trim().trim_matches('"').trim();
323                if !path.is_empty() {
324                    let resolved = project_root.join(path);
325                    return Some(resolved.to_string_lossy().to_string());
326                }
327            }
328        }
329    }
330    None
331}

333/// Parse workflow definitions from workflows.dal (simple heuristic).
334fn parse_workflows_from_dal(path: &Path) -> Option<Vec<WorkflowInfo>> {
335    let content = std::fs::read_to_string(path).ok()?;
336    let mut workflows = Vec::new();
337    for line in content.lines() {
338        let line = line.trim();
339        if line.starts_with("// workflow:") || line.starts_with("# workflow:") {
340            let rest = line
341                .trim_start_matches("// workflow:")
342                .trim_start_matches("# workflow:")
343                .trim();
344            if let Some((name, desc)) = rest.split_once(' ') {
345                workflows.push(WorkflowInfo {
346                    id: name.to_string().to_lowercase().replace(' ', "_"),
347                    name: name.to_string(),
348                    description: Some(desc.to_string()),
349                    steps: None,
350                });
351            } else if !rest.is_empty() {
352                workflows.push(WorkflowInfo {
353                    id: rest.to_lowercase().replace(' ', "_"),
354                    name: rest.to_string(),
355                    description: None,
356                    steps: None,
357                });
358            }
359        }
360    }
361    if workflows.is_empty() {
362        None
363    } else {
364        Some(workflows)
365    }
