1//! WebSocket bridge for `dal lsp` stdio. Allows the browser to use full LSP (monaco-languageclient)
2//! by connecting to GET /api/lsp/stream, which spawns `dal lsp` and forwards LSP messages.
3
4use axum::extract::ws::{Message, WebSocket};
5use futures_util::{SinkExt, StreamExt};
6use std::io::{BufRead, BufReader, Read, Write};
7use std::path::Path;
8use std::process::{Child, Command, Stdio};
9use std::sync::mpsc;
10use std::thread;
11use tokio::sync::mpsc as tokio_mpsc;

13/// Read one LSP message (Content-Length: n \r\n \r\n body) from reader.
14fn read_lsp_message<R: Read>(reader: &mut BufReader<R>) -> Option<String> {
15    let mut header = String::new();
16    if reader.read_line(&mut header).ok()? == 0 {
17        return None;
18    }
19    let mut content_length: Option<usize> = None;
20    loop {
21        let line = header.trim_end_matches("\r\n").trim_end_matches('\n');
22        if line.is_empty() {
23            break;
24        }
25        if let Some(stripped) = line.strip_prefix("Content-Length:") {
26            if let Ok(n) = stripped.trim().parse::<usize>() {
27                content_length = Some(n);
28            }
29        }
30        header.clear();
31        if reader.read_line(&mut header).ok()? == 0 {
32            return None;
33        }
34    }
35    let n = content_length?;
36    let mut buf = vec![0u8; n];
37    let mut read = 0;
38    while read < n {
39        let got = reader.get_mut().read(&mut buf[read..]).ok()?;
40        if got == 0 {
41            return None;
42        }
43        read += got;
44    }
45    String::from_utf8(buf).ok()
46}

48/// Write one LSP message to writer.
49fn write_lsp_message<W: Write>(writer: &mut W, body: &str) -> std::io::Result<()> {
50    let bytes = body.as_bytes();
51    write!(writer, "Content-Length: {}\r\n\r\n", bytes.len())?;
52    writer.write_all(bytes)?;
53    writer.flush()?;
54    Ok(())
55}

57/// Spawn `dal lsp` with workspace as cwd. Returns (child, stdout_reader) for bridging.
58fn spawn_dal_lsp(workspace_root: &Path) -> Option<(Child, BufReader<std::process::ChildStdout>)> {
59    let mut cmd = Command::new("dal");
60    cmd.arg("lsp")
61        .current_dir(workspace_root)
62        .stdin(Stdio::piped())
63        .stdout(Stdio::piped())
64        .stderr(Stdio::null());
65    let mut child = cmd.spawn().ok()?;
66    let stdout = child.stdout.take()?;
67    let reader = BufReader::new(stdout);
68    Some((child, reader))
69}

71/// Run the LSP bridge: spawn `dal lsp`, forward between WebSocket and process stdio.
72/// WebSocket sends/receives raw JSON-RPC body (no Content-Length header); we add/strip header for stdio.
73pub async fn run_lsp_bridge(socket: WebSocket, workspace_root: std::path::PathBuf) {
74    if let Some((mut child, mut stdout_reader)) = spawn_dal_lsp(&workspace_root) {
75        let mut stdin = match child.stdin.take() {
76            Some(s) => s,
77            None => return,
78        };
79        let (tx_out, mut rx_out) = tokio_mpsc::unbounded_channel::<String>();
80        let (tx_in, rx_in) = mpsc::sync_channel::<String>(64);
81
82        // Thread: read from LSP stdout, send body to async via tx_out
83        thread::spawn(move || {
84            while let Some(body) = read_lsp_message(&mut stdout_reader) {
85                if tx_out.send(body).is_err() {
86                    break;
87                }
88            }
89        });
90
91        // Thread: receive from async via rx_in, write to LSP stdin
92        thread::spawn(move || {
93            while let Ok(body) = rx_in.recv() {
94                if write_lsp_message(&mut stdin, &body).is_err() {
95                    break;
96                }
97            }
98        });
99
100        // Async: forward between WebSocket and channels
101        let (mut ws_sender, mut ws_receiver) = socket.split();
102        loop {
103            tokio::select! {
104                Some(body) = rx_out.recv() => {
105                    if ws_sender.send(Message::Text(body)).await.is_err() {
106                        break;
107                    }
108                }
109                msg = ws_receiver.next() => {
110                    match msg {
111                        Some(Ok(Message::Text(text))) => {
112                            let _ = tx_in.send(text);
113                        }
114                        Some(Ok(Message::Close(_))) | None => break,
115                        _ => {}
116                    }
117                }
118            }
119        }
120        let _ = child.kill();
121    }
122}
