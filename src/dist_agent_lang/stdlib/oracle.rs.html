21#[cfg(feature = "http-interface")]
22fn json_to_value(j: &serde_json::Value) -> Value {
23    match j {
24        serde_json::Value::Null => Value::Null,
25        serde_json::Value::Bool(b) => Value::Bool(*b),
26        serde_json::Value::Number(n) => {
27            if let Some(i) = n.as_i64() {
28                Value::Int(i)
29            } else if let Some(f) = n.as_f64() {
30                Value::Float(f)
31            } else {
32                Value::String(n.to_string())
33            }
34        }
35        serde_json::Value::String(s) => Value::String(s.clone()),
36        serde_json::Value::Array(arr) => Value::List(arr.iter().map(json_to_value).collect()),
37        serde_json::Value::Object(obj) => {
38            let m: HashMap<String, Value> = obj
39                .iter()
40                .map(|(k, v)| (k.clone(), json_to_value(v)))
41                .collect();
42            Value::Map(m)
43        }
44    }
45}

47/// Oracle ABI - Interface for external data integration with security hardening
48///
49/// This provides a namespace-based approach to oracle operations:
50/// - oracle::fetch(source, query) - Fetch data from external source
51/// - oracle::verify(data, signature) - Verify data authenticity
52/// - oracle::stream(source, callback) - Stream real-time data
53///
54/// Security Features:
55/// - Signed data feeds with cryptographic verification
56/// - Multi-source validation and consensus
57/// - Timestamp validation and replay protection
58/// - Rate limiting per source
59/// - Trusted source allowlisting
60

189impl OracleSecurityManager {
190    pub fn new() -> Self {
191        Self {
192            trusted_sources: HashMap::new(),
193            response_cache: HashMap::new(),
194            max_age_seconds: 300, // Default: 5 minutes
195        }
196    }
197
198    /// Register a trusted oracle source
199    pub fn register_trusted_source(&mut self, source: OracleSource) {
200        if source.trusted && source.public_key.is_some() {
201            self.trusted_sources.insert(source.name.clone(), source);
202        }
203    }
204
205    /// Security: Verify oracle response signature
206    pub fn verify_response(&self, response: &OracleResponse) -> bool {
207        if let Some(signature) = &response.signature {
208            if let Some(source) = self.trusted_sources.get(&response.source) {
209                if let Some(public_key) = &source.public_key {
210                    // In production, use actual cryptographic signature verification
211                    // For now, implement basic validation
212                    return verify_signature(&response.data, signature, public_key);
213                }
214            }
215        }
216        false
217    }
218
219    /// Security: Validate response timestamp (prevent replay attacks)
220    pub fn validate_timestamp(&self, timestamp: u64) -> bool {
221        let now = get_current_timestamp();
222        let age = now.saturating_sub(timestamp);
223
224        // Reject responses older than max_age
225        age <= self.max_age_seconds * 1000 // Convert to ms
226    }
227
228    /// Security: Get cached response if valid
229    pub fn get_cached(&self, cache_key: &str) -> Option<OracleResponse> {
230        if let Some((response, cached_time)) = self.response_cache.get(cache_key) {
231            if self.validate_timestamp(*cached_time) {
232                return Some(response.clone());
233            }
234        }
235        None
236    }

239impl OracleConsensus {
240    pub fn new(sources: Vec<String>, threshold: f64) -> Self {
241        Self {
242            sources,
243            responses: Vec::new(),
244            consensus_threshold: threshold,
245        }
246    }
247
248    /// Add a response from a source
249    pub fn add_response(&mut self, response: OracleResponse) {
250        self.responses.push(response);
251    }
252
253    /// Security: Determine consensus from multiple sources
254    pub fn get_consensus(&self) -> Option<OracleResponse> {
255        if self.responses.is_empty() {
256            return None;
257        }
258
259        // Find the most common value
260        let mut value_counts: HashMap<String, (usize, OracleResponse)> = HashMap::new();
261
262        for response in &self.responses {
263            let key = format!("{:?}", response.data); // Simplified comparison
264            value_counts
265                .entry(key)
266                .and_modify(|(count, _)| *count += 1)
267                .or_insert((1, response.clone()));
268        }
269
270        // Find majority
271        let total = self.responses.len();
272        for (count, response) in value_counts.values() {
273            let agreement = *count as f64 / total as f64;
274            if agreement >= self.consensus_threshold {
275                let mut consensus_response = response.clone();
276                consensus_response.confidence_score = agreement;
277                return Some(consensus_response);
278            }
279        }
280
281        None
282    }
283}

285/// Security: Fetch data from external oracle source with validation. When http-interface and source is a URL, GETs and parses response; optional X-Signature header.
286pub fn fetch(source: &str, query: OracleQuery) -> Result<OracleResponse, String> {
287    let timestamp = get_current_timestamp();
288
289    #[cfg(feature = "http-interface")]
290    if source.starts_with("http://") || source.starts_with("https://") {
291        let client = reqwest::blocking::Client::builder()
292            .timeout(std::time::Duration::from_secs(
293                query.timeout.unwrap_or(30) as u64
294            ))
295            .build()
296            .map_err(|e| e.to_string())?;
297        let resp = client.get(source).send().map_err(|e| e.to_string())?;
298        if !resp.status().is_success() {
299            return Err(format!("Oracle fetch failed: {}", resp.status()));
300        }
301        let signature = resp
302            .headers()
303            .get("X-Signature")
304            .and_then(|v| v.to_str().ok())
305            .map(String::from);
306        let body: serde_json::Value = resp.json().map_err(|e| e.to_string())?;
307        let data = json_to_value(&body);
308        if query.require_signature && signature.is_none() {
309            return Err("Signature required but not provided".to_string());
310        }
311        return Ok(OracleResponse {
312            data,
313            timestamp,
314            source: source.to_string(),
315            signature,
316            verified: false,
317            confidence_score: 1.0,
318        });
319    }
320
321    // Fallback: mock by source name
322    let (mock_data, signature) = match source {
323        "price_feed" => {
324            let data = match query.query_type.as_str() {
325                "btc_price" => Value::Int(45000),
326                "eth_price" => Value::Int(3200),
327                "sol_price" => Value::Int(98),
328                _ => Value::String("unknown_query".to_string()),
329            };
330            let sig = generate_signature(&data, "oracle_key_12345");
331            (data, Some(sig))
332        }
333        "weather" => {
334            let data = match query.query_type.as_str() {
335                "temperature" => Value::Int(72),
336                "humidity" => Value::Int(65),
337                "forecast" => Value::String("sunny".to_string()),
338                _ => Value::String("unknown_query".to_string()),
339            };
340            let sig = generate_signature(&data, "weather_key_67890");
341            (data, Some(sig))
342        }
343        _ => return Err(format!("Unknown oracle source: {}", source)),
344    };
345
346    let response = OracleResponse {
347        data: mock_data,
348        timestamp,
349        source: source.to_string(),
350        signature,
351        verified: false,
352        confidence_score: 1.0,
353    };
354
355    if query.require_signature && response.signature.is_none() {
356        return Err("Signature required but not provided".to_string());
357    }
358
359    Ok(response)
360}

362/// Security: Fetch data from multiple sources and validate consensus
363pub fn fetch_with_consensus(
364    sources: Vec<&str>,
365    query: OracleQuery,
366    threshold: f64,
367) -> Result<OracleResponse, String> {
368    let mut consensus =
369        OracleConsensus::new(sources.iter().map(|s| s.to_string()).collect(), threshold);
370
371    // Fetch from all sources
372    for source in sources {
373        match fetch(source, query.clone()) {
374            Ok(response) => consensus.add_response(response),
375            Err(e) => eprintln!("Oracle source {} failed: {}", source, e),
376        }
377    }
378
379    // Get consensus
380    consensus.get_consensus().ok_or_else(|| {
381        format!(
382            "Failed to reach consensus (threshold: {:.1}%)",
383            threshold * 100.0
384        )
385    })
386}

395/// Stream real-time data from oracle source with security.
396///
397/// A real implementation would establish WebSocket connections (e.g. via
398/// tokio-tungstenite). When `source` is a `ws://` or `wss://` URL, it is
399/// accepted and a stream id is returned; actual connection logic could be
400/// added behind a Cargo feature (e.g. `oracle-websocket`) that enables
401/// tokio-tungstenite. Named sources (price_feed, weather, events) return
402/// mock stream ids for testing.
403///
404/// **App developers** can use WebSockets today: call `stream(ws_url, …)`
405/// to get a `stream_id`, then implement their own WebSocket client (in-app,
406/// frontend, or another service) and use `stream_id` with `get_stream` /
407/// `close_stream` for correlation; they are not blocked on the crate
408/// adding a built-in WebSocket connection.
409pub fn stream(source: &str, _callback: &str) -> Result<String, String> {
410    // Security: Validate source before streaming
411    let trusted_named = ["price_feed", "weather", "events"];
412
413    let created_at = get_current_timestamp();
414
415    // Real-world: accept ws/wss URLs; real connection would use WebSocket (e.g. tokio-tungstenite)
416    if source.starts_with("ws://") || source.starts_with("wss://") {
417        let stream_id = format!("stream_ws_{:x}", {
418            let mut h = Sha256::new();
419            h.update(source.as_bytes());
420            h.finalize()[0..8]
421                .iter()
422                .fold(0u64, |a, &b| (a << 8) | u64::from(b))
423        });
424        get_stream_registry().insert(
425            stream_id.clone(),
426            StreamEntry {
427                source: source.to_string(),
428                created_at,
429            },
430        );
431        return Ok(stream_id);
432    }
433
434    if !trusted_named.contains(&source) {
435        return Err(format!("Untrusted source for streaming: {}", source));
436    }
437
438    let stream_id = match source {
439        "price_feed" => "stream_id_price_123".to_string(),
440        "weather" => "stream_id_weather_456".to_string(),
441        "events" => "stream_id_events_789".to_string(),
442        _ => return Err(format!("Streaming not supported for source: {}", source)),
443    };
444    get_stream_registry().insert(
445        stream_id.clone(),
446        StreamEntry {
447            source: source.to_string(),
448            created_at,
449        },
450    );
451    Ok(stream_id)
452}

487/// Security: Generate signature for data
488fn generate_signature(data: &Value, private_key: &str) -> String {
489    let data_string = format!("{:?}", data); // Simplified serialization
490    let mut hasher = Sha256::new();
491    hasher.update(data_string.as_bytes());
492    hasher.update(private_key.as_bytes());
493    format!("{:x}", hasher.finalize())
494}
495
496/// Security: Verify signature against data and public key
497fn verify_signature(data: &Value, signature: &str, public_key: &str) -> bool {
498    // In production, this would use actual asymmetric crypto (ECDSA/EdDSA)
499    // For now, implement basic verification using hash comparison
500    let expected_sig = generate_signature(data, public_key);
501
502    // Constant-time comparison to prevent timing attacks
503    if signature.len() != expected_sig.len() {
504        return false;
505    }
506
507    let mut matches = true;
508    for (a, b) in signature.bytes().zip(expected_sig.bytes()) {
509        if a != b {
510            matches = false;
511        }
512    }
513    matches
514}
