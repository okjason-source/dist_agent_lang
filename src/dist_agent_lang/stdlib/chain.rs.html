14    // Shared reqwest client to avoid concurrent initialization panics
15    // Reqwest's blocking client runtime can only be initialized once per process
16    lazy_static! {
17        static ref CLIENT: Arc<Mutex<Option<Arc<reqwest::blocking::Client>>>> =
18            Arc::new(Mutex::new(None));
19    }
20
21    fn get_client() -> Result<Arc<reqwest::blocking::Client>, String> {
22        let mut client_guard = CLIENT
23            .lock()
24            .map_err(|e| format!("Mutex poisoned: {}", e))?;
25
26        if let Some(ref client) = *client_guard {
27            // Return a clone of the Arc (cheap, shares the underlying client)
28            return Ok(Arc::clone(client));
29        }
30
31        // Create new client and store it
32        let client = reqwest::blocking::Client::builder()
33            .timeout(std::time::Duration::from_secs(30))
34            .build()
35            .map_err(|e| e.to_string())?;
36
37        let client_arc = Arc::new(client);
38        *client_guard = Some(Arc::clone(&client_arc));
39        Ok(client_arc)
40    }

42    /// Perform a JSON-RPC 2.0 request to the chain RPC endpoint.
43    /// Returns the "result" field on success, or an error string.
44    pub(super) fn rpc_request(
45        rpc_url: &str,
46        method: &str,
47        params: Vec<JsonValue>,
48    ) -> Result<JsonValue, String> {
49        let body = json!({
50            "jsonrpc": "2.0",
51            "method": method,
52            "params": params,
53            "id": 1
54        });
55        let client = get_client()?;
56        let resp = client
57            .post(rpc_url)
58            .json(&body)
59            .send()
60            .map_err(|e| e.to_string())?;
61        let status = resp.status();
62        let json: JsonValue = resp.json().map_err(|e| e.to_string())?;
63        if let Some(err) = json.get("error") {
64            let msg = err
65                .get("message")
66                .and_then(|m| m.as_str())
67                .unwrap_or("RPC error");
68            return Err(msg.to_string());
69        }
70        if !status.is_success() {
71            return Err(format!("RPC HTTP {}", status));
72        }
73        json.get("result")
74            .cloned()
75            .ok_or_else(|| "Missing result".to_string())
76    }

78    /// Parse hex string (with or without 0x) to u128, saturating to i64 for balance/amounts.
79    pub(super) fn hex_to_i64(hex_str: &str) -> i64 {
80        let s = hex_str.strip_prefix("0x").unwrap_or(hex_str);
81        let s = s.trim_start_matches('0');
82        if s.is_empty() {
83            return 0;
84        }
85        let mut value: u128 = 0;
86        for c in s.chars() {
87            let d = c.to_digit(16).unwrap_or(0) as u128;
88            value = value.saturating_mul(16).saturating_add(d);
89        }
90        value.min(i64::MAX as u128) as i64
91    }
92
93    /// Parse hex gas price to gwei (f64).
94    pub(super) fn hex_gas_price_to_gwei(hex_str: &str) -> f64 {
95        let raw = hex_to_i64(hex_str);
96        if raw <= 0 {
97            return 0.0;
98        }
99        raw as f64 / 1_000_000_000.0
100    }

249    // Get chain config

314#[cfg(feature = "http-interface")]
315fn deploy_via_raw_transaction(rpc_url: &str, raw_tx_hex: &str) -> Result<String, String> {
316    use serde_json::json;
317    let tx_hex = raw_tx_hex.strip_prefix("0x").unwrap_or(raw_tx_hex);
318    let result = rpc::rpc_request(
319        rpc_url,
320        "eth_sendRawTransaction",
321        vec![json!(format!("0x{}", tx_hex))],
322    )?;
323    let tx_hash = result.as_str().ok_or("expected tx hash string")?;
324    let receipt = wait_for_receipt(rpc_url, tx_hash, 30)?;
325    let addr = receipt
326        .get("contractAddress")
327        .and_then(|v| v.as_str())
328        .ok_or("no contractAddress in receipt")?
329        .to_string();
330    Ok(addr)
331}

333#[cfg(feature = "http-interface")]
334fn wait_for_receipt(
335    rpc_url: &str,
336    tx_hash: &str,
337    max_attempts: u32,
338) -> Result<serde_json::Value, String> {
339    use serde_json::json;
340    for _ in 0..max_attempts {
341        let result = rpc::rpc_request(rpc_url, "eth_getTransactionReceipt", vec![json!(tx_hash)])?;
342        if !result.is_null() {
343            return Ok(result);
344        }
345        std::thread::sleep(std::time::Duration::from_secs(2));
346    }
347    Err("timeout waiting for transaction receipt".to_string())
348}

351///
352/// # Arguments
353/// * `chain_id` - The target chain ID
354/// * `operation` - The operation to estimate gas for
355///
356/// # Returns
357/// * `i64` - Estimated gas cost
358///
359/// # Example
360/// ```rust
361/// use dist_agent_lang::stdlib::chain;
362/// let gas = chain::estimate_gas(1, "transfer".to_string());
363/// ```
364pub fn estimate_gas(chain_id: i64, operation: String) -> i64 {
365    let chain_config = match get_chain_config(chain_id) {
366        Some(c) => c,
367        None => return 0,
368    };
369
370    #[cfg(feature = "http-interface")]
371    {
372        use serde_json::json;
373        let params = vec![json!({ "to": null, "data": "0x" })];
374        if let Ok(result) = rpc::rpc_request(&chain_config.rpc_url, "eth_estimateGas", params) {
375            if let Some(hex_str) = result.as_str() {
376                let gas = rpc::hex_to_i64(hex_str);
377                if gas > 0 {
378                    return gas;
379                }
380            }
381        }
382    }
383
384    // Fallback: mock gas estimation when RPC unavailable or not configured
385    let base_gas = match operation.as_str() {
386        "transfer" => 21000,
387        "mint" => 50000,
388        "burn" => 30000,
389        "approve" => 46000,
390        "deploy" => 200000,
391        _ => 21000,
392    };
393    let adjusted_gas = if chain_config.is_testnet {
394        base_gas / 2
395    } else {
396        base_gas
397    };
398    adjusted_gas
399}

432        .duration_since(std::time::UNIX_EPOCH)
433        .unwrap_or_default()
434        .as_secs()
435        % 100) as f64
436        / 100.0;
437    chain_config.gas_price + variation

464///
465/// # Arguments
466/// * `chain_id` - The target chain ID
467/// * `tx_hash` - Transaction hash
468///
469/// # Returns
470/// * `String` - Transaction status
471///
472/// # Example
473/// ```rust
474/// use dist_agent_lang::stdlib::chain;
475/// let status = chain::get_transaction_status(1, "0x1234...".to_string());
476/// ```
477pub fn get_transaction_status(chain_id: i64, tx_hash: String) -> String {
478    let chain_config = match get_chain_config(chain_id) {
479        Some(c) => c,
480        None => {
481            return if tx_hash.starts_with("0x") && tx_hash.len() > 10 {
482                "confirmed".to_string()
483            } else {
484                "pending".to_string()
485            };
486        }
487    };
488
489    #[cfg(feature = "http-interface")]
490    {
491        use serde_json::json;
492        let hash = if tx_hash.starts_with("0x") {
493            tx_hash.clone()
494        } else {
495            format!("0x{}", tx_hash)
496        };
497        if let Ok(result) = rpc::rpc_request(
498            &chain_config.rpc_url,
499            "eth_getTransactionReceipt",
500            vec![json!(hash)],
501        ) {
502            if result.is_null() {
503                return "pending".to_string();
504            }
505            if let Some(status) = result.get("status").and_then(|v| v.as_str()) {
506                return match status {
507                    "0x1" => "confirmed".to_string(),
508                    "0x0" => "failed".to_string(),
509                    _ => "pending".to_string(),
510                };
511            }
512        }
513    }
514
515    if tx_hash.starts_with("0x") && tx_hash.len() > 10 {
516        "confirmed".to_string()
517    } else {
518        "pending".to_string()
519    }
520}

523///

534/// let balance = chain::get_balance(1, "0x1234...".to_string());
535/// ```
536pub fn get_balance(chain_id: i64, address: String) -> i64 {
537    let chain_config = get_chain_config(chain_id);
538
539    #[cfg(feature = "http-interface")]
540    if let Some(ref config) = chain_config {
541        use serde_json::json;
542        let addr = if address.starts_with("0x") {
543            address.clone()
544        } else {
545            format!("0x{}", address)
546        };
547        if let Ok(result) = rpc::rpc_request(
548            &config.rpc_url,
549            "eth_getBalance",
550            vec![json!(addr), json!("latest")],
551        ) {
552            if let Some(hex_str) = result.as_str() {
553                return rpc::hex_to_i64(hex_str);
554            }
555        }
556    }
557
558    // Fallback: mock balance when RPC unavailable or chain not supported
559    if let Some(_) = chain_config {
560        if address.starts_with("0x") {
561            let hash_sum: i64 = address
562                .chars()
563                .filter(|c| c.is_ascii_hexdigit())
564                .map(|c| c.to_digit(16).unwrap_or(0) as i64)
565                .sum();
566            let limited_sum = hash_sum % 1000;
567            let wei_per_unit: i64 = 1000000000000000;
568            return limited_sum.checked_mul(wei_per_unit).unwrap_or(0);
569        }
570    }
571    0
572}

617            );

620        },
621        Some("chain"),
622    );
623
624    #[cfg(feature = "http-interface")]
625    if let Some(call_data) = args.get("data").or_else(|| args.get("calldata")) {

643            }
644        }
645    }
646
647    format!(
648        "success: {} called on {} at {}",
649        function_name, contract_address, chain_config.name

653/// Mint a new asset or token
654///
655/// # Arguments
656/// * `name` - The name of the asset
657/// * `metadata` - Additional metadata for the asset
658///
659/// # Returns
660/// * `i64` - The generated ID for the minted asset
661///
662/// # Example
663/// ```rust
664/// use dist_agent_lang::stdlib::chain;
665/// use std::collections::HashMap;
666/// let mut metadata = HashMap::new();
667/// metadata.insert("description".to_string(), "A unique NFT".to_string());
668/// metadata.insert("image".to_string(), "ipfs://...".to_string());
669/// let asset_id = chain::mint("MyNFT".to_string(), metadata);
670/// ```
671pub fn mint(name: String, metadata: HashMap<String, String>) -> i64 {
672    // Log the mint operation for audit purposes
673    crate::stdlib::log::audit(
674        "mint",
675        {
676            let mut data = std::collections::HashMap::new();
677            data.insert("name".to_string(), Value::String(name.clone()));
678            data.insert(
679                "metadata".to_string(),
680                Value::String(format!("{:?}", metadata)),
681            );
682            data
683        },
684        Some("chain"),
685    );
686
687    // Unique ID: hash(name + metadata + timestamp) for uniqueness; fit to i64
688    let timestamp = std::time::SystemTime::now()
689        .duration_since(std::time::UNIX_EPOCH)
690        .unwrap_or_default()
691        .as_secs() as i64;
692    let mut combined = name.as_bytes().to_vec();
693    combined.extend(format!("{:?}", metadata).as_bytes());
694    combined.extend(timestamp.to_be_bytes());
695    let hash = md5::compute(&combined);
696    let hash_val: i64 = (hash[0] as i64)
697        .wrapping_shl(24)
698        .wrapping_add((hash[1] as i64).wrapping_shl(16))
699        .wrapping_add((hash[2] as i64).wrapping_shl(8))
700        .wrapping_add(hash[3] as i64);
701    let asset_id = timestamp
702        .abs()
703        .wrapping_mul(10000)
704        .wrapping_add(hash_val & 0x7FFF_FFFF);
705
706    // Log success
707    crate::stdlib::log::info(
708        "mint_success",
709        {
710            let mut data = std::collections::HashMap::new();
711            data.insert("asset_id".to_string(), Value::Int(asset_id));
712            data
713        },
714        None,
715    );
716
717    asset_id
718}

720/// Update an existing asset or token
721///
722/// # Arguments
723/// * `asset_id` - The ID of the asset to update
724/// * `updates` - The updates to apply to the asset
725///
726/// # Returns
727/// * `bool` - True if update was successful, false otherwise
728///
729/// # Example
730/// ```rust
731/// use dist_agent_lang::stdlib::chain;
732/// use std::collections::HashMap;
733/// let mut updates = HashMap::new();
734/// updates.insert("description".to_string(), "Updated description".to_string());
735/// let success = chain::update(12345, updates);
736/// ```
737pub fn update(asset_id: i64, updates: HashMap<String, String>) -> bool {
738    crate::stdlib::log::audit(
739        "update",
740        {
741            let mut data = std::collections::HashMap::new();
742            data.insert("asset_id".to_string(), Value::Int(asset_id));
743            data.insert(
744                "updates".to_string(),
745                Value::String(format!("{:?}", updates)),
746            );
747            data
748        },
749        Some("chain"),
750    );
751
752    #[cfg(feature = "http-interface")]
753    if let Some(raw_hex) = updates
754        .get("raw_transaction")
755        .or_else(|| updates.get("signed_tx"))
756    {
757        let chain_id = updates
758            .get("chain_id")
759            .and_then(|s| s.trim().parse::<i64>().ok())
760            .or_else(|| {
761                env::var("CHAIN_ASSET_CHAIN_ID")
762                    .ok()
763                    .and_then(|s| s.trim().parse().ok())
764            });
765        if let Some(chain_id) = chain_id {
766            if let Some(config) = get_chain_config(chain_id) {
767                let tx_hex = if raw_hex.starts_with("0x") {
768                    raw_hex.clone()
769                } else {
770                    format!("0x{}", raw_hex)
771                };
772                if let Ok(result) = rpc::rpc_request(
773                    &config.rpc_url,
774                    "eth_sendRawTransaction",
775                    vec![serde_json::json!(tx_hex)],
776                ) {
777                    if let Some(s) = result.as_str() {
778                        if s.len() > 2 {
779                            return true;
780                        }
781                    }
782                }
783            }
784        }
785    }
786
787    let success = asset_id > 0;
788
789    if success {
790        // Log success
791        crate::stdlib::log::info(
792            "update_success",
793            {
794                let mut data = std::collections::HashMap::new();
795                data.insert("asset_id".to_string(), Value::Int(asset_id));
796                data
797            },
798            None,
799        );
800    } else {
801        // Log failure
802        crate::stdlib::log::info(
803            "update_failed",
804            {
805                let mut data = std::collections::HashMap::new();
806                data.insert("asset_id".to_string(), Value::Int(asset_id));
807                data.insert(
808                    "reason".to_string(),
809                    Value::String("Invalid asset ID".to_string()),
810                );
811                data
812            },
813            None,
814        );
815    }
816
817    success
818}

820/// Get asset information. When CHAIN_ASSET_CHAIN_ID and CHAIN_ASSET_CONTRACT are set and
821/// http-interface is enabled, calls tokenURI(asset_id) on the contract and adds token_uri to metadata.
822pub fn get(asset_id: i64) -> HashMap<String, String> {
823    let mut asset_info = HashMap::new();
824    asset_info.insert("id".to_string(), asset_id.to_string());
825    asset_info.insert("name".to_string(), format!("Asset_{}", asset_id));
826    asset_info.insert("created_at".to_string(), (asset_id / 10000).to_string());
827    asset_info.insert("status".to_string(), "active".to_string());
828
829    #[cfg(feature = "http-interface")]
830    if let (Ok(chain_id_str), Ok(contract)) = (
831        env::var("CHAIN_ASSET_CHAIN_ID"),
832        env::var("CHAIN_ASSET_CONTRACT"),
833    ) {
834        if let Ok(chain_id) = chain_id_str.trim().parse::<i64>() {
835            if let Some(config) = get_chain_config(chain_id) {
836                let token_id = if asset_id < 0 { 0u64 } else { asset_id as u64 };
837                let data = format!("0xc87b56dd{:064x}", token_id); // tokenURI(uint256)
838                let to = if contract.starts_with("0x") {
839                    contract.clone()
840                } else {
841                    format!("0x{}", contract)
842                };
843                let tx = serde_json::json!({ "to": to, "data": data });
844                if let Ok(result) = rpc::rpc_request(
845                    &config.rpc_url,
846                    "eth_call",
847                    vec![tx, serde_json::json!("latest")],
848                ) {
849                    if let Some(hex_str) = result.as_str() {
850                        asset_info.insert("token_uri_result".to_string(), hex_str.to_string());
851                    }
852                }
853            }
854        }
855    }
856
857    let mut metadata = HashMap::new();
858    metadata.insert("description".to_string(), "A blockchain asset".to_string());
859    metadata.insert("version".to_string(), "1".to_string());
860    asset_info.insert("metadata".to_string(), format!("{:?}", metadata));
861    asset_info
862}

864/// Check if an asset exists. When CHAIN_ASSET_CHAIN_ID and CHAIN_ASSET_CONTRACT are set and
865/// http-interface is enabled, calls ownerOf(asset_id); if non-zero address, returns true.
866pub fn exists(asset_id: i64) -> bool {
867    #[cfg(feature = "http-interface")]
868    if let (Ok(chain_id_str), Ok(contract)) = (
869        env::var("CHAIN_ASSET_CHAIN_ID"),
870        env::var("CHAIN_ASSET_CONTRACT"),
871    ) {
872        if let Ok(chain_id) = chain_id_str.trim().parse::<i64>() {
873            if let Some(config) = get_chain_config(chain_id) {
874                let token_id = if asset_id < 0 { 0u64 } else { asset_id as u64 };
875                let data = format!("0x6352211e{:064x}", token_id); // ownerOf(uint256)
876                let to = if contract.starts_with("0x") {
877                    contract.clone()
878                } else {
879                    format!("0x{}", contract)
880                };
881                let tx = serde_json::json!({ "to": to, "data": data });
882                if let Ok(result) = rpc::rpc_request(
883                    &config.rpc_url,
884                    "eth_call",
885                    vec![tx, serde_json::json!("latest")],
886                ) {
887                    if let Some(hex_str) = result.as_str() {
888                        let s = hex_str
889                            .strip_prefix("0x")
890                            .unwrap_or(hex_str)
891                            .trim_start_matches('0');
892                        if !s.is_empty() && s != "0" {
893                            return true;
894                        }
895                    }
896                }
897                return false;
898            }
899        }
900    }
901    asset_id > 0
902}
