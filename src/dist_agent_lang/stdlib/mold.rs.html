1// Mold stdlib — Expose mold load/spawn to DAL code.
2//
3// Aligns with mold/STDLIB_PROPOSAL.md and mold/DEFINITION.md.
4// - mold::load(source) — Load mold from path or name, return config map
5// - mold::spawn_from(source, name_override?, params?) — Load mold + spawn agent, return agent_id; params optional map for create-time overrides
6// - mold::list() — List local mold paths
7// - mold::get_info(mold_id) — Query on-chain mold info (requires web3)
8// - mold::use_mold(mold_id, name_override?) — Pay + spawn from on-chain mold (requires web3)

15/// Load mold from source (path, name, or ipfs://cid). Returns config as map for DAL.
16/// Base is typically std::env::current_dir() or script directory.
17pub fn load(source: &str, base: &Path) -> Result<Value, String> {
18    let mold = load_mold_from_source(source, base)?;
19    Ok(mold_config_to_value(&mold))
20}

22/// Create agent from mold source. Returns agent_id. Name override and params optional.
23/// Params: optional map of string keys to string values; merged into metadata and substituted in role/capabilities ({{key}}).
24pub fn spawn_from(
25    source: &str,
26    base: &Path,
27    name_override: Option<&str>,
28    params: Option<Value>,
29) -> Result<String, String> {
30    let params_map = value_map_to_string_params(params.as_ref());
31    let ctx = create_from_mold_source(source, base, name_override, params_map.as_ref())?;
32    Ok(ctx.agent_id)
33}

35/// Convert Value::Map with string values to HashMap<String, String>. Non-map or non-string values are skipped.
36fn value_map_to_string_params(v: Option<&Value>) -> Option<HashMap<String, String>> {
37    let map = match v {
38        Some(Value::Map(m)) => m,
39        _ => return None,
40    };
41    let mut out = HashMap::new();
42    for (k, val) in map {
43        if let Value::String(s) = val {
44            out.insert(k.clone(), s.clone());
45        }
46    }
47    if out.is_empty() {
48        None
49    } else {
50        Some(out)
51    }
52}

54/// List local mold file paths (base, mold/, mold/samples). Returns list of path strings.
55pub fn list(base: &Path) -> Value {
56    let paths = crate::mold::list_local_paths(base);
57    Value::List(
58        paths
59            .into_iter()
60            .map(|p| Value::String(p.display().to_string()))
61            .collect(),
62    )
63}

93/// Convert MoldInfo to Value::Map for DAL.
94#[cfg(feature = "web3")]
95fn mold_info_to_value(info: &crate::mold::MoldInfo) -> Value {
96    let mut map = HashMap::new();
97    map.insert(
98        "creator".to_string(),
99        Value::String(format!("{:?}", info.creator)),
100    );
101    map.insert(
102        "ipfs_hash".to_string(),
103        Value::String(info.ipfs_hash.clone()),
104    );
105    map.insert("mint_fee".to_string(), Value::Int(info.mint_fee as i64));
106    map.insert("mint_count".to_string(), Value::Int(info.mint_count as i64));
107    map.insert(
108        "max_use_count".to_string(),
109        Value::Int(info.max_use_count as i64),
110    );
111    map.insert("active".to_string(), Value::Bool(info.active));
112    map.insert("created_at".to_string(), Value::Int(info.created_at as i64));
113    map.insert("updated_at".to_string(), Value::Int(info.updated_at as i64));
114    Value::Map(map)
115}

117/// Convert MoldConfig to Value::Map for DAL.
118fn mold_config_to_value(mold: &MoldConfig) -> Value {
119    let mut map = HashMap::new();
120    map.insert("name".to_string(), Value::String(mold.name.clone()));
121    map.insert("version".to_string(), Value::String(mold.version.clone()));
122
123    let mut agent_map = HashMap::new();
124    agent_map.insert(
125        "type".to_string(),
126        Value::String(mold.agent.agent_type.clone()),
127    );
128    agent_map.insert("role".to_string(), Value::String(mold.agent.role.clone()));
129    agent_map.insert(
130        "capabilities".to_string(),
131        Value::List(
132            mold.agent
133                .capabilities
134                .iter()
135                .map(|s| Value::String(s.clone()))
136                .collect(),
137        ),
138    );
139    agent_map.insert(
140        "trust_level".to_string(),
141        Value::String(mold.agent.trust_level.clone()),
142    );
143    agent_map.insert("learning".to_string(), Value::Bool(mold.agent.learning));
144    agent_map.insert(
145        "communication".to_string(),
146        Value::Bool(mold.agent.communication),
147    );
148    agent_map.insert(
149        "coordination".to_string(),
150        Value::Bool(mold.agent.coordination),
151    );
152    agent_map.insert(
153        "memory_limit".to_string(),
154        Value::String(mold.agent.memory_limit.clone()),
155    );
156    map.insert("agent".to_string(), Value::Map(agent_map));
157
158    if !mold.parameters.is_empty() {
159        let params: HashMap<String, Value> = mold
160            .parameters
161            .iter()
162            .map(|(k, v)| (k.clone(), Value::String(v.clone())))
163            .collect();
164        map.insert("parameters".to_string(), Value::Map(params));
165    }
166
167    if !mold.dependencies.is_empty() {
168        map.insert(
169            "dependencies".to_string(),
170            Value::List(
171                mold.dependencies
172                    .iter()
173                    .map(|s| Value::String(s.clone()))
174                    .collect(),
175            ),
176        );
177    }
178
179    Value::Map(map)
180}
