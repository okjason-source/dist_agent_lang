1/// CloudAdmin ABI - Interface for CloudAdmin operations
2///
3/// Authorization and policy are delegated to the shared trust policy engine (trust::),
4/// so admin permissions and policy rules are unified across trust and cloudadmin.
5///
6/// - cloudadmin::authorize(admin_id, operation, resource) - Authorize admin operation
7/// - cloudadmin::enforce_policy(policy_name, context) - Enforce admin policy
8/// - cloudadmin::validate_hybrid_trust(admin_trust, user_trust) - Validate hybrid trust
9/// - cloudadmin::bridge_trusts(centralized_trust, decentralized_trust) - Bridge trust models
10pub use crate::stdlib::trust::{create_admin_context, AdminContext, AdminLevel, AdminPolicy};
11
12/// Authorize admin operation (delegates to trust policy engine: key registry, admin registry, env).
13pub fn authorize(admin_id: &str, operation: &str, resource: &str) -> bool {
14    crate::stdlib::trust::authorize(admin_id, operation, resource)
15}
16
17/// Enforce admin policy (delegates to trust policy engine).
18pub fn enforce_policy(policy_name: &str, context: AdminContext) -> Result<bool, String> {
19    crate::stdlib::trust::enforce_policy(policy_name, context)
20}
21
22/// Validate hybrid trust between admin and user
23pub fn validate_hybrid_trust(admin_trust: &str, user_trust: &str) -> bool {
24    // Hybrid trust requires both admin and user trust to be valid
25    admin_trust == "valid" && user_trust == "valid"
26}
27
28/// Bridge centralized admin trust with decentralized user trust
29pub fn bridge_trusts(centralized_trust: &str, decentralized_trust: &str) -> bool {
30    // Bridge requires both trust models to be compatible
31    centralized_trust == "admin" && decentralized_trust == "user"
32}
