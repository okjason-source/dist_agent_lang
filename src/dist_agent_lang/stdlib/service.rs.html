106        if let Ok(text) = call_llm_api(
107            base,

113        ) {
114            return Ok(text);
115        }
116    }
117
118    // Fallback: mock responses when no API key or HTTP unavailable
119    match service.model.as_str() {
120        "gpt-4" => {
121            let response = match prompt {

123                "Explain smart contracts" => "Smart contracts are self-executing contracts...",

128        }
129        "claude" => {
130            let response = match prompt {
131                "What is blockchain?" => "Blockchain represents a paradigm shift...",
132                "Explain smart contracts" => "Smart contracts automate agreement execution...",
133                "How does DeFi work?" => "DeFi removes intermediaries from financial services...",
134                _ => "I can help explain blockchain concepts...",

136            Ok(response.to_string())

142#[cfg(feature = "http-interface")]
143fn call_llm_api(
144    base_url: &str,
145    api_key: &str,
146    model: &str,
147    prompt: &str,
148    temperature: f64,
149    max_tokens: Option<i64>,
150) -> Result<String, String> {
151    let url = format!("{}/chat/completions", base_url.trim_end_matches('/'));
152    let body = serde_json::json!({
153        "model": model,
154        "messages": [{"role": "user", "content": prompt}],
155        "temperature": temperature,
156        "max_tokens": max_tokens.unwrap_or(1024),
157    });
158    let client = reqwest::blocking::Client::builder()
159        .timeout(std::time::Duration::from_secs(60))
160        .build()
161        .map_err(|e| e.to_string())?;
162    let resp = client
163        .post(&url)
164        .header("Authorization", format!("Bearer {}", api_key))
165        .header("Content-Type", "application/json")
166        .json(&body)
167        .send()
168        .map_err(|e| e.to_string())?;
169    if !resp.status().is_success() {
170        return Err(format!("API error: {}", resp.status()));
171    }
172    let json: serde_json::Value = resp.json().map_err(|e| e.to_string())?;
173    let text = json
174        .get("choices")
175        .and_then(|c| c.get(0))
176        .and_then(|c| c.get("message"))
177        .and_then(|m| m.get("content"))
178        .and_then(|c| c.as_str())
179        .ok_or_else(|| "Invalid API response".to_string())?;
180    Ok(text.to_string())
181}

198#[cfg(feature = "http-interface")]
199fn call_embeddings_api(
200    base_url: &str,
201    api_key: &str,
202    model: &str,
203    input: &str,
204) -> Result<Vec<f64>, String> {
205    let url = format!("{}/embeddings", base_url.trim_end_matches('/'));
206    let body = serde_json::json!({
207        "model": model,
208        "input": input,
209    });
210    let client = reqwest::blocking::Client::builder()
211        .timeout(std::time::Duration::from_secs(30))
212        .build()
213        .map_err(|e| e.to_string())?;
214    let resp = client
215        .post(&url)
216        .header("Authorization", format!("Bearer {}", api_key))
217        .header("Content-Type", "application/json")
218        .json(&body)
219        .send()
220        .map_err(|e| e.to_string())?;
221    if !resp.status().is_success() {
222        return Err(format!("Embeddings API error: {}", resp.status()));
223    }
224    let json: serde_json::Value = resp.json().map_err(|e| e.to_string())?;
225    let data = json
226        .get("data")
227        .and_then(|d| d.get(0))
228        .and_then(|e| e.get("embedding"))
229        .and_then(|e| e.as_array())
230        .ok_or_else(|| "Invalid embeddings response".to_string())?;
231    let vec: Result<Vec<f64>, _> = data.iter().map(|v| v.as_f64().ok_or("non-f64")).collect();
232    vec.map_err(|_| "Embedding values must be f64".to_string())
233}

262fn call_vision_api(
263    base_url: &str,
264    api_key: &str,
265    model: &str,
266    prompt: &str,
267    image_url: Option<&str>,
268    image_base64: Option<&str>,
269) -> Result<String, String> {
270    let url = format!("{}/chat/completions", base_url.trim_end_matches('/'));
271    let image_content = if let Some(u) = image_url {
272        serde_json::json!({ "type": "image_url", "image_url": { "url": u } })
273    } else if let Some(b64) = image_base64 {
274        let data_url = if b64.starts_with("data:") {
275            b64.to_string()
276        } else {
277            format!("data:image/jpeg;base64,{}", b64)
278        };
279        serde_json::json!({ "type": "image_url", "image_url": { "url": data_url } })
280    } else {
281        return Err("vision_analyze requires image_url or image_base64".to_string());
282    };
283    let body = serde_json::json!({
284        "model": model,
285        "messages": [{
286            "role": "user",
287            "content": [
288                { "type": "text", "text": prompt },
289                image_content
290            ]
291        }],
292        "max_tokens": 1024
293    });
294    let client = reqwest::blocking::Client::builder()
295        .timeout(std::time::Duration::from_secs(60))
296        .build()
297        .map_err(|e| e.to_string())?;
298    let resp = client
299        .post(&url)
300        .header("Authorization", format!("Bearer {}", api_key))
301        .header("Content-Type", "application/json")
302        .json(&body)
303        .send()
304        .map_err(|e| e.to_string())?;
305    if !resp.status().is_success() {
306        return Err(format!("Vision API error: {}", resp.status()));
307    }
308    let json: serde_json::Value = resp.json().map_err(|e| e.to_string())?;
309    let text = json
310        .get("choices")
311        .and_then(|c| c.get(0))
312        .and_then(|c| c.get("message"))
313        .and_then(|m| m.get("content"))
314        .and_then(|c| c.as_str())
315        .ok_or_else(|| "Invalid vision API response".to_string())?;
316    Ok(text.to_string())
317}

363        .get("data")

376pub fn call(service: ServiceCall) -> Result<Value, String> {
377    #[cfg(feature = "http-interface")]
378    if let Some(ref base_url) = service.base_url {
379        if let Ok(v) = call_http_service(
380            base_url,
381            &service.method,
382            &service.parameters,
383            service.timeout,
384        ) {
385            return Ok(v);
386        }
387    }
388
389    // Fallback: mock when no base_url or HTTP unavailable
390    match service.service_name.as_str() {
391        "payment" => match service.method.as_str() {
392            "process" => Ok(Value::String("payment_processed_12345".to_string())),
393            "refund" => Ok(Value::String("refund_issued_67890".to_string())),
394            "status" => Ok(Value::String("completed".to_string())),
395            _ => Err(format!("Unknown payment method: {}", service.method)),
396        },
397        "email" => match service.method.as_str() {
398            "send" => Ok(Value::String("email_sent_abc123".to_string())),
399            "template" => Ok(Value::String("welcome_email".to_string())),
400            "verify" => Ok(Value::Bool(true)),
401            _ => Err(format!("Unknown email method: {}", service.method)),
402        },
403        "sms" => match service.method.as_str() {

409    }

428#[cfg(feature = "http-interface")]
429fn value_to_json(v: &Value) -> serde_json::Value {
430    match v {
431        Value::Int(n) => serde_json::json!(n),
432        Value::Float(f) => serde_json::json!(f),
433        Value::String(s) => serde_json::json!(s),
434        Value::Bool(b) => serde_json::json!(b),
435        Value::Null => serde_json::Value::Null,
436        Value::List(arr) => serde_json::Value::Array(arr.iter().map(value_to_json).collect()),
437        Value::Map(m) => serde_json::Value::Object(
438            m.iter()
439                .map(|(k, v)| (k.clone(), value_to_json(v)))
440                .collect(),
441        ),
442        Value::Struct(_, m) => serde_json::Value::Object(
443            m.iter()
444                .map(|(k, v)| (k.clone(), value_to_json(v)))
445                .collect(),
446        ),
447        Value::Array(arr) => serde_json::Value::Array(arr.iter().map(value_to_json).collect()),
448        _ => serde_json::Value::String(v.to_string()),
449    }
450}

452#[cfg(feature = "http-interface")]
453fn call_http_service(
454    base_url: &str,
455    method: &str,
456    parameters: &HashMap<String, Value>,
457    timeout_secs: Option<i64>,
458) -> Result<Value, String> {
459    let url = format!(
460        "{}/{}",
461        base_url.trim_end_matches('/'),
462        method.trim_start_matches('/')
463    );
464    let body: serde_json::Value = serde_json::Value::Object(
465        parameters
466            .iter()
467            .map(|(k, v)| (k.clone(), value_to_json(v)))
468            .collect(),
469    );
470    let mut client_builder = reqwest::blocking::Client::builder();
471    if let Some(secs) = timeout_secs {
472        client_builder = client_builder.timeout(std::time::Duration::from_secs(secs as u64));
473    }
474    let client = client_builder.build().map_err(|e| e.to_string())?;
475    let resp = client
476        .post(&url)
477        .json(&body)
478        .send()
479        .map_err(|e| e.to_string())?;
480    if !resp.status().is_success() {
481        return Err(format!(
482            "HTTP {}: {}",
483            resp.status(),
484            resp.text().unwrap_or_default()
485        ));
486    }
487    let json: serde_json::Value = resp.json().map_err(|e| e.to_string())?;
488    Ok(Value::String(json.to_string()))
489}

491#[cfg(feature = "http-interface")]
492fn webhook_http_post(
493    url: &str,
494    method: &str,
495    headers: &HashMap<String, String>,
496    data: &HashMap<String, Value>,
497) -> Result<String, String> {
498    let body: serde_json::Value = serde_json::Value::Object(
499        data.iter()
500            .map(|(k, v)| (k.clone(), value_to_json(v)))
501            .collect(),
502    );
503    let client = reqwest::blocking::Client::builder()
504        .timeout(std::time::Duration::from_secs(30))
505        .build()
506        .map_err(|e| e.to_string())?;
507    let mut req = match method.to_uppercase().as_str() {
508        "GET" => client.get(url),
509        "POST" => client.post(url),
510        "PUT" => client.put(url),
511        "PATCH" => client.patch(url),
512        _ => client.post(url),
513    };
514    for (k, v) in headers {
515        req = req.header(k.as_str(), v.as_str());
516    }
517    let resp = req.json(&body).send().map_err(|e| e.to_string())?;
518    if !resp.status().is_success() {
519        return Err(format!("Webhook HTTP {}", resp.status()));
520    }
521    Ok(format!("webhook_ok_{}", url.len()))
522}
