19/// Nonce manager to prevent replay attacks.
20/// "Last seen nonce" per key is stored; when a key has no entry, the next valid nonce is 1 (no literal used as nonce).
21fn default_last_seen_for_replay_check() -> u64 {
22    u64::MIN
23}
24
25/// Nonce manager to prevent replay attacks
26#[derive(Debug, Clone)]
27pub struct NonceManager {
28    nonces: HashMap<String, u64>, // (address, chain_id) -> last_nonce
29}
30
31impl NonceManager {
32    pub fn new() -> Self {
33        Self {
34            nonces: HashMap::new(),
35        }
36    }
37
38    /// Check if nonce is valid (must be greater than last seen nonce)
39    pub fn check_nonce(&mut self, key: &str, nonce: u64) -> Result<bool, RuntimeError> {
40        let last_seen = self
41            .nonces
42            .get(key)
43            .copied()
44            .unwrap_or_else(default_last_seen_for_replay_check);
45
46        if nonce <= last_seen {
47            return Ok(false); // Replay attack detected
48        }
49
50        // Update last seen nonce
51        self.nonces.insert(key.to_string(), nonce);
52        Ok(true)
53    }
54
55    /// Get next expected nonce for an address (1 when none seen yet; otherwise last_seen + 1).
56    pub fn get_next_nonce(&self, key: &str) -> u64 {
57        self.nonces.get(key).map(|&n| n + 1).unwrap_or(1)
58    }
59}

61/// ECDSA signature verification and signing (Ethereum-style with secp256k1)
62pub struct ECDSASignatureVerifier;
63
64impl ECDSASignatureVerifier {
65    /// Verify an ECDSA signature using k256
66    ///
67    /// # Arguments
68    /// * `message` - The message that was signed
69    /// * `signature` - Hex-encoded signature (64 or 65 bytes: r + s or r + s + v)
70    /// * `public_key` - Hex-encoded compressed or uncompressed public key (33 or 65 bytes)
71    pub fn verify(message: &[u8], signature: &str, public_key: &str) -> Result<bool, RuntimeError> {
72        // Hash the message with SHA256
73        let mut hasher = Sha256::new();
74        hasher.update(message);
75        let message_hash = hasher.finalize();
76
77        // Decode the signature from hex
78        let signature_bytes = hex::decode(signature)
79            .map_err(|e| RuntimeError::General(format!("Invalid hex signature: {}", e)))?;
80
81        // Signature should be 64 bytes (r + s) or 65 bytes (r + s + v for Ethereum)
82        let sig_bytes = if signature_bytes.len() == 65 {
83            // Remove recovery ID (v) if present
84            &signature_bytes[0..64]
85        } else if signature_bytes.len() == 64 {
86            &signature_bytes[..]
87        } else {
88            return Err(RuntimeError::General(format!(
89                "Invalid signature length: expected 64 or 65 bytes, got {}",
90                signature_bytes.len()
91            )));
92        };
93
94        // Parse the signature
95        let signature = K256Signature::from_slice(sig_bytes)
96            .map_err(|e| RuntimeError::General(format!("Invalid ECDSA signature: {}", e)))?;
97
98        // Decode the public key from hex
99        let pubkey_bytes = hex::decode(public_key)
100            .map_err(|e| RuntimeError::General(format!("Invalid hex public key: {}", e)))?;
101
102        // Parse the public key (supports both compressed 33 bytes and uncompressed 65 bytes)
103        let verifying_key = VerifyingKey::from_sec1_bytes(&pubkey_bytes)
104            .map_err(|e| RuntimeError::General(format!("Invalid public key: {}", e)))?;
105
106        // Verify the signature
107        match verifying_key.verify(&message_hash, &signature) {
108            Ok(_) => Ok(true),
109            Err(_) => Ok(false),
110        }
111    }
112
113    /// Sign a message with ECDSA using k256
114    ///
115    /// # Arguments
116    /// * `message` - The message to sign
117    /// * `private_key` - Hex-encoded private key (32 bytes)
118    ///
119    /// # Returns
120    /// * Hex-encoded signature (64 bytes: r + s)
121    pub fn sign(message: &[u8], private_key: &str) -> Result<String, RuntimeError> {
122        // Hash the message with SHA256
123        let mut hasher = Sha256::new();
124        hasher.update(message);
125        let message_hash = hasher.finalize();
126
127        // Decode the private key from hex
128        let privkey_bytes = hex::decode(private_key)
129            .map_err(|e| RuntimeError::General(format!("Invalid hex private key: {}", e)))?;
130
131        // Private key should be 32 bytes
132        if privkey_bytes.len() != 32 {
133            return Err(RuntimeError::General(format!(
134                "Invalid private key length: expected 32 bytes, got {}",
135                privkey_bytes.len()
136            )));
137        }
138
139        // Parse the private key
140        let secret_key = SecretKey::from_slice(&privkey_bytes)
141            .map_err(|e| RuntimeError::General(format!("Invalid private key: {}", e)))?;
142
143        let signing_key = SigningKey::from(secret_key);
144
145        // Sign the message
146        let signature: K256Signature = signing_key.sign(&message_hash);
147
148        // Return hex-encoded signature (r + s, 64 bytes)
149        Ok(hex::encode(signature.to_bytes()))
150    }
151
152    /// Generate a new ECDSA keypair
153    ///
154    /// # Returns
155    /// * (private_key_hex, public_key_hex)
156    pub fn generate_keypair() -> Result<(String, String), RuntimeError> {
157        use rand::rngs::OsRng;
158
159        // Generate random private key
160        let secret_key = SecretKey::random(&mut OsRng);
161
162        // Get private key bytes before moving secret_key
163        let private_key_bytes = secret_key.to_bytes();
164
165        // Create signing key (moves secret_key)
166        let signing_key = SigningKey::from(secret_key);
167
168        // Get public key
169        let verifying_key = signing_key.verifying_key();
170
171        // Encode to hex
172        let private_key_hex = hex::encode(private_key_bytes);
173        let public_key_hex = hex::encode(verifying_key.to_encoded_point(true).as_bytes()); // Compressed
174
175        Ok((private_key_hex, public_key_hex))
176    }
177}

189/// EdDSA signature verification (Solana-style with Ed25519)
190pub struct EdDSASignatureVerifier;
191
192impl EdDSASignatureVerifier {
193    /// Verify an EdDSA (Ed25519) signature using ed25519-dalek
194    ///
195    /// # Arguments
196    /// * `message` - The message that was signed (NOT pre-hashed)
197    /// * `signature` - Hex-encoded or base64-encoded EdDSA signature (64 bytes)
198    /// * `public_key` - Hex-encoded or base64-encoded EdDSA public key (32 bytes)
199    pub fn verify(message: &[u8], signature: &str, public_key: &str) -> Result<bool, RuntimeError> {
200        // Try to decode signature (try hex first, then base64)
201        let signature_bytes = if let Ok(bytes) = hex::decode(signature) {
202            bytes
203        } else if let Ok(bytes) = general_purpose::STANDARD.decode(signature) {
204            bytes
205        } else {
206            return Err(RuntimeError::General(
207                "Signature must be hex or base64 encoded".to_string(),
208            ));
209        };
210
211        // Ed25519 signature should be exactly 64 bytes
212        if signature_bytes.len() != 64 {
213            return Err(RuntimeError::General(format!(
214                "Invalid Ed25519 signature length: expected 64 bytes, got {}",
215                signature_bytes.len()
216            )));
217        }
218
219        // Parse the signature
220        let signature = Ed25519Signature::from_slice(&signature_bytes)
221            .map_err(|e| RuntimeError::General(format!("Invalid Ed25519 signature: {}", e)))?;
222
223        // Try to decode public key (try hex first, then base64)
224        let pubkey_bytes = if let Ok(bytes) = hex::decode(public_key) {
225            bytes
226        } else if let Ok(bytes) = general_purpose::STANDARD.decode(public_key) {
227            bytes
228        } else {
229            return Err(RuntimeError::General(
230                "Public key must be hex or base64 encoded".to_string(),
231            ));
232        };
233
234        // Ed25519 public key should be exactly 32 bytes
235        if pubkey_bytes.len() != 32 {
236            return Err(RuntimeError::General(format!(
237                "Invalid Ed25519 public key length: expected 32 bytes, got {}",
238                pubkey_bytes.len()
239            )));
240        }
241
242        // Parse the public key
243        let verifying_key =
244            Ed25519VerifyingKey::from_bytes(pubkey_bytes.as_slice().try_into().map_err(|_| {
245                RuntimeError::General("Failed to convert public key bytes".to_string())
246            })?)
247            .map_err(|e| RuntimeError::General(format!("Invalid Ed25519 public key: {}", e)))?;
248
249        // Verify the signature
250        // Ed25519 signs the raw message, not a hash
251        match verifying_key.verify(message, &signature) {
252            Ok(_) => Ok(true),
253            Err(_) => Ok(false),
254        }
255    }
256
257    /// Verify an EdDSA signature with hex encoding (convenience method)
258    pub fn verify_hex(
259        message: &[u8],
260        signature_hex: &str,
261        pubkey_hex: &str,
262    ) -> Result<bool, RuntimeError> {
263        Self::verify(message, signature_hex, pubkey_hex)
264    }
265
266    /// Verify an EdDSA signature with base64 encoding (convenience method)  
267    pub fn verify_base64(
268        message: &[u8],
269        signature_b64: &str,
270        pubkey_b64: &str,
271    ) -> Result<bool, RuntimeError> {
272        Self::verify(message, signature_b64, pubkey_b64)
273    }
274
275    /// Sign a message with Ed25519. Message is signed raw (not pre-hashed).
276    /// * `message` - The message to sign
277    /// * `private_key` - Hex-encoded or base64-encoded private key (32 bytes)
278    /// Returns hex-encoded signature (64 bytes).
279    pub fn sign(message: &[u8], private_key: &str) -> Result<String, RuntimeError> {
280        let privkey_bytes = if let Ok(bytes) = hex::decode(private_key) {
281            bytes
282        } else if let Ok(bytes) = general_purpose::STANDARD.decode(private_key) {
283            bytes
284        } else {
285            return Err(RuntimeError::General(
286                "Private key must be hex or base64 encoded".to_string(),
287            ));
288        };
289        if privkey_bytes.len() != 32 {
290            return Err(RuntimeError::General(format!(
291                "Invalid Ed25519 private key length: expected 32 bytes, got {}",
292                privkey_bytes.len()
293            )));
294        }
295        let signing_key =
296            Ed25519SigningKey::from_bytes(privkey_bytes.as_slice().try_into().map_err(|_| {
297                RuntimeError::General("Failed to convert private key bytes".to_string())
298            })?);
299        let signature = signing_key.sign(message);
300        Ok(hex::encode(signature.to_bytes()))
301    }
302
303    /// Generate a new Ed25519 keypair.
304    /// Returns (private_key_hex, public_key_hex).
305    pub fn generate_keypair() -> Result<(String, String), RuntimeError> {
306        let mut seed = [0u8; 32];
307        rand::RngCore::fill_bytes(&mut rand::rngs::OsRng, &mut seed);
308        let signing_key = Ed25519SigningKey::from_bytes(&seed);
309        let verifying_key = signing_key.verifying_key();
310        let private_key_hex = hex::encode(signing_key.to_bytes());
311        let public_key_hex = hex::encode(verifying_key.to_bytes());
312        Ok((private_key_hex, public_key_hex))
313    }

316/// Cryptographic signature verifier with replay protection
317#[derive(Debug, Clone)]
318pub struct SecureSignatureVerifier {
319    nonce_manager: NonceManager,
320}
321
322impl SecureSignatureVerifier {
323    pub fn new() -> Self {
324        Self {
325            nonce_manager: NonceManager::new(),
326        }
327    }
328
329    /// Verify a signature with nonce-based replay protection
330    ///
331    /// # Arguments
332    /// * `message` - The message that was signed
333    /// * `signature` - The signature
334    /// * `public_key` - The public key or address
335    /// * `nonce` - Nonce to prevent replay attacks
336    /// * `signer_key` - Key for nonce tracking (address + chain_id)
337    /// * `scheme` - Signature scheme (ECDSA or EdDSA)
338    pub fn verify_with_nonce(
339        &mut self,
340        message: &[u8],
341        signature: &str,
342        public_key: &str,
343        nonce: u64,
344        signer_key: &str,
345        scheme: &str,
346    ) -> Result<bool, RuntimeError> {
347        // First, check nonce to prevent replay attacks
348        if !self.nonce_manager.check_nonce(signer_key, nonce)? {
349            return Err(RuntimeError::General(format!(
350                "Replay attack detected: nonce {} already used",
351                nonce
352            )));
353        }
354
355        // Include nonce in message hash to bind signature to nonce
356        let mut hasher = Sha256::new();
357        hasher.update(message);
358        hasher.update(&nonce.to_be_bytes());
359        let message_with_nonce = hasher.finalize();
360
361        // Verify signature based on scheme
362        let is_valid = match scheme.to_lowercase().as_str() {
363            "ecdsa" | "ethereum" => {
364                ECDSASignatureVerifier::verify(&message_with_nonce, signature, public_key)?
365            }
366            "eddsa" | "solana" => {
367                EdDSASignatureVerifier::verify(&message_with_nonce, signature, public_key)?
368            }
369            _ => {
370                return Err(RuntimeError::General(format!(
371                    "Unsupported signature scheme: {}",
372                    scheme
373                )));
374            }
375        };
376
377        Ok(is_valid)
378    }
379
380    /// Get next expected nonce for an address
381    pub fn get_next_nonce(&self, signer_key: &str) -> u64 {
382        self.nonce_manager.get_next_nonce(signer_key)
383    }
384}
