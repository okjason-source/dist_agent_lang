1//! Evolve: dynamic markdown file for conversation history and action log (Phase 3).
2//! Path from DAL_AGENT_CONTEXT_PATH or \[agent\] context_path in agent.toml / dal.toml; default ./evolve.md.
3
4use std::env;
5use std::fs::OpenOptions;
6use std::io::Write;
7use std::path::PathBuf;
8
9const DEFAULT_CONTEXT_PATH: &str = "./evolve.md";
10const DEFAULT_AGENT_NAME: &str = "Agent";
11
12fn get_context_path() -> PathBuf {
13    if let Ok(p) = env::var("DAL_AGENT_CONTEXT_PATH") {
14        return PathBuf::from(p);
15    }
16    let cwd = env::current_dir().unwrap_or_else(|_| PathBuf::from("."));
17    for name in &["agent.toml", "dal.toml"] {
18        let path = cwd.join(name);
19        if let Ok(content) = std::fs::read_to_string(&path) {
20            if let Ok(table) = content.parse::<toml::Table>() {
21                if let Some(ctx_path) = table
22                    .get("agent")
23                    .and_then(|v| v.as_table())
24                    .and_then(|t| t.get("context_path"))
25                    .and_then(|v| v.as_str())
26                {
27                    let mut buf = PathBuf::from(ctx_path);
28                    if !buf.is_absolute() {
29                        buf = cwd.join(buf);
30                    }
31                    return buf;
32                }
33            }
34        }
35    }
36    cwd.join(DEFAULT_CONTEXT_PATH)
37}

39fn ensure_header(path: &std::path::Path, agent_name: &str) -> std::io::Result<()> {
40    if path.exists() {
41        return Ok(());
42    }
43    if let Some(parent) = path.parent() {
44        std::fs::create_dir_all(parent)?;
45    }
46    let now = chrono::Utc::now().to_rfc3339();
47    let header = format!(
48        "# Agent context — {}\nUpdated: {}\n\n## Conversation\n\n",
49        agent_name, now
50    );
51    std::fs::write(path, header)
52}
53
54fn ensure_action_log_section(path: &std::path::Path) -> std::io::Result<()> {
55    let content = std::fs::read_to_string(path).unwrap_or_default();
56    if content.contains("## Action log") {
57        return Ok(());
58    }
59    let table_header = "\n## Action log\n\n| Time | Action | Detail | Result | Get Task (ms) | Do Task (ms) | Total (ms) |\n|------|--------|--------|--------|----------------|--------------|------------|\n";
60    let mut f = OpenOptions::new().append(true).open(path)?;
61    f.write_all(table_header.as_bytes())?;
62    Ok(())
63}

65/// Load the current context file content. Creates file with header if missing.
66pub fn load(agent_name: Option<&str>) -> Result<String, String> {
67    let path = get_context_path();
68    let name = agent_name.unwrap_or(DEFAULT_AGENT_NAME);
69    ensure_header(&path, name).map_err(|e| e.to_string())?;
70    std::fs::read_to_string(&path).map_err(|e| e.to_string())
71}

73/// Max length of user/agent content written to the conversation log (log injection / DoS).
74const MAX_CONVERSATION_LOG_LEN: usize = 32_768;
75
76/// Sanitize content for safe inclusion in the conversation log (prevents log injection).
77pub fn sanitize_for_conversation(s: &str) -> String {
78    let truncated = if s.len() > MAX_CONVERSATION_LOG_LEN {
79        &s[..MAX_CONVERSATION_LOG_LEN]
80    } else {
81        s
82    };
83    let with_newlines = truncated.replace("\n", "\n  ");
84    // Prevent injection of our own **User:** / **Agent:** structure
85    with_newlines
86        .replace("**User:**", "[User]:")
87        .replace("**Agent:**", "[Agent]:")
88}

90/// Append a conversation turn (user message + agent response) to the context file.
91pub fn append_conversation(
92    user_message: &str,
93    agent_response: &str,
94    agent_name: Option<&str>,
95) -> Result<(), String> {
96    let path = get_context_path();
97    let name = agent_name.unwrap_or(DEFAULT_AGENT_NAME);
98    ensure_header(&path, name).map_err(|e| e.to_string())?;
99
100    let now = chrono::Utc::now();
101    let ts = now.format("%Y-%m-%dT%H:%M");
102    let user_safe = sanitize_for_conversation(user_message);
103    let agent_safe = sanitize_for_conversation(agent_response);
104    let block = format!(
105        "\n### {}\n**User:** {}\n\n**Agent:** {}\n\n",
106        ts, user_safe, agent_safe
107    );
108
109    let mut f = OpenOptions::new()
110        .create(true)
111        .append(true)
112        .open(&path)
113        .map_err(|e| e.to_string())?;
114    f.write_all(block.as_bytes()).map_err(|e| e.to_string())?;
115    Ok(())
116}

118/// Append an action log entry (e.g. sh::run, command, result).
119pub fn append_log(action: &str, detail: &str, result: &str) -> Result<(), String> {
120    append_log_with_timing(action, detail, result, None, None)
121}

123/// Append an action log entry with task timing fields.
124///
125/// `get_task_ms` should represent the time spent choosing/acquiring a task.
126/// `do_task_ms` should represent the time spent executing the task.
127pub fn append_log_timed(
128    action: &str,
129    detail: &str,
130    result: &str,
131    get_task_ms: i64,
132    do_task_ms: i64,
133) -> Result<(), String> {
134    append_log_with_timing(action, detail, result, Some(get_task_ms), Some(do_task_ms))
135}
136
137fn append_log_with_timing(
138    action: &str,
139    detail: &str,
140    result: &str,
141    get_task_ms: Option<i64>,
142    do_task_ms: Option<i64>,
143) -> Result<(), String> {
144    let path = get_context_path();
145    ensure_header(&path, DEFAULT_AGENT_NAME).map_err(|e| e.to_string())?;
146    ensure_action_log_section(&path).map_err(|e| e.to_string())?;
147
148    let now = chrono::Utc::now().format("%H:%M:%S");
149    let get_ms = get_task_ms
150        .map(|v| v.to_string())
151        .unwrap_or_else(|| "-".to_string());
152    let do_ms = do_task_ms
153        .map(|v| v.to_string())
154        .unwrap_or_else(|| "-".to_string());
155    let total_ms = match (get_task_ms, do_task_ms) {
156        (Some(get), Some(exec)) => (get.saturating_add(exec)).to_string(),
157        _ => "-".to_string(),
158    };
159    let line = format!(
160        "| {} | {} | {} | {} | {} | {} | {} |\n",
161        now, action, detail, result, get_ms, do_ms, total_ms
162    );
163
164    let mut f = OpenOptions::new()
165        .create(true)
166        .append(true)
167        .open(&path)
168        .map_err(|e| e.to_string())?;
169    f.write_all(line.as_bytes()).map_err(|e| e.to_string())?;
170    Ok(())
171}

178/// Load recent context: last `max_lines` lines of the file, or full content if max_lines <= 0.
179/// Use for inclusion in agent context without loading the entire history.
180pub fn load_recent(agent_name: Option<&str>, max_lines: i64) -> Result<String, String> {
181    let full = load(agent_name)?;
182    if max_lines <= 0 {
183        return Ok(full);
184    }
185    let lines: Vec<&str> = full.lines().collect();
186    let n = max_lines as usize;
187    let start = if lines.len() <= n { 0 } else { lines.len() - n };
188    Ok(lines[start..].join("\n"))
189}

191/// Trim the context file to keep only the last `keep_tail_lines` lines of content (after the header).
192/// Preserves the "# Agent context" and "## Conversation" header, then keeps the last keep_tail_lines lines.
193pub fn trim_retention(keep_tail_lines: i64) -> Result<(), String> {
194    let path = get_context_path();
195    let content = std::fs::read_to_string(&path).map_err(|e| e.to_string())?;
196    if keep_tail_lines <= 0 {
197        return Ok(());
198    }
199    let lines: Vec<&str> = content.lines().collect();
200    let header_end = lines
201        .iter()
202        .position(|&l| l.trim() == "## Conversation")
203        .map(|i| i + 2)
204        .unwrap_or(0);
205    let body_lines = lines.len().saturating_sub(header_end);
206    let keep = keep_tail_lines as usize;
207    if body_lines <= keep {
208        return Ok(());
209    }
210    let drop = body_lines - keep;
211    let new_body_start = header_end + drop;
212    let new_content = [
213        lines[..header_end].join("\n"),
214        lines[new_body_start..].join("\n"),
215    ]
216    .join("\n");
217    std::fs::write(&path, new_content).map_err(|e| e.to_string())
218}

220/// Append a summary section (e.g. periodic or on-demand session summary).
221pub fn append_summary(summary_text: &str, title: Option<&str>) -> Result<(), String> {
222    let path = get_context_path();
223    ensure_header(&path, DEFAULT_AGENT_NAME).map_err(|e| e.to_string())?;
224
225    let now = chrono::Utc::now();
226    let ts = now.format("%Y-%m-%d %H:%M");
227    let heading = title
228        .map(|t| format!("## Summary — {}\n\n", t))
229        .unwrap_or_else(|| "## Summary\n\n".to_string());
230    let block = format!("{}\n{}\n\n{}\n\n", heading, ts, summary_text);
231
232    let mut f = OpenOptions::new()
233        .create(true)
234        .append(true)
235        .open(&path)
236        .map_err(|e| e.to_string())?;
237    f.write_all(block.as_bytes()).map_err(|e| e.to_string())?;
238    Ok(())
