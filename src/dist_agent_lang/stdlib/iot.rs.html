5#[cfg(feature = "http-interface")]
6fn value_to_serde_json(v: &Value) -> serde_json::Value {
7    match v {
8        Value::Int(n) => serde_json::json!(n),
9        Value::Float(f) => serde_json::json!(f),
10        Value::String(s) => serde_json::json!(s),
11        Value::Bool(b) => serde_json::json!(b),
12        Value::Null => serde_json::Value::Null,
13        Value::List(arr) => serde_json::Value::Array(arr.iter().map(value_to_serde_json).collect()),
14        Value::Map(m) => serde_json::Value::Object(
15            m.iter()
16                .map(|(k, v)| (k.clone(), value_to_serde_json(v)))
17                .collect(),
18        ),
19        Value::Struct(_, m) => serde_json::Value::Object(
20            m.iter()
21                .map(|(k, v)| (k.clone(), value_to_serde_json(v)))
22                .collect(),
23        ),
24        Value::Array(arr) => {
25            serde_json::Value::Array(arr.iter().map(value_to_serde_json).collect())
26        }
27        _ => serde_json::Value::String(v.to_string()),
28    }
29}

85/// Parse device type from config string (e.g. "robot_arm", "programmable_hardware", "drone").
86pub fn device_type_from_string(s: &str) -> Option<DeviceType> {
87    Some(match s.to_lowercase().replace('-', "_").as_str() {
88        "sensor_node" | "sensor" => DeviceType::SensorNode,
89        "actuator_node" | "actuator" => DeviceType::ActuatorNode,
90        "gateway" => DeviceType::Gateway,
91        "edge_computer" | "edge" => DeviceType::EdgeComputer,
92        "smart_device" | "smart" => DeviceType::SmartDevice,
93        "industrial_controller" | "industrial" | "plc" => DeviceType::IndustrialController,
94        "wearable" => DeviceType::Wearable,
95        "programmable_hardware" | "programmable" | "fpga" | "microcontroller" => {
96            DeviceType::ProgrammableHardware
97        }
98        "robot_arm" | "robotarm" | "manipulator" => DeviceType::RobotArm,
99        "mobile_robot" | "mobilerobot" | "agv" | "rover" => DeviceType::MobileRobot,
100        "drone" | "uav" => DeviceType::Drone,
101        "robot_controller" | "robotcontroller" | "motion_controller" => DeviceType::RobotController,
102        other if !other.is_empty() => DeviceType::Custom(other.to_string()),
103        _ => return None,
104    })
105}

1415#[cfg(feature = "http-interface")]
1416fn serde_json_to_value(j: &serde_json::Value) -> Value {
1417    match j {
1418        serde_json::Value::Null => Value::Null,
1419        serde_json::Value::Bool(b) => Value::Bool(*b),
1420        serde_json::Value::Number(n) => {
1421            if let Some(i) = n.as_i64() {
1422                Value::Int(i)
1423            } else if let Some(f) = n.as_f64() {
1424                Value::Float(f)
1425            } else {
1426                Value::String(n.to_string())
1427            }
1428        }
1429        serde_json::Value::String(s) => Value::String(s.clone()),
1430        serde_json::Value::Array(arr) => Value::List(arr.iter().map(serde_json_to_value).collect()),
1431        serde_json::Value::Object(obj) => {
1432            let m: HashMap<String, Value> = obj
1433                .iter()
1434                .map(|(k, v)| (k.clone(), serde_json_to_value(v)))
1435                .collect();
1436            Value::Map(m)
1437        }
1438    }

1509                "device_id".to_string(),
1510                Value::String(device_id.to_string()),
1511            );
1512            data.insert(
1513                "history_points".to_string(),
1514                Value::Int(sensor_history.len() as i64),
1515            );
1516            data.insert(
1517                "message".to_string(),
1518                Value::String("Predicting device failure probability".to_string()),
1519            );
1520            data
1521        },
1522        Some("iot"),
1523    );
1524
1525    #[cfg(feature = "http-interface")]
1526    if let Ok(url) = env::var("IOT_ML_API_URL") {
1527        let history: Vec<serde_json::Value> = sensor_history
1528            .iter()
1529            .map(|r| {
1530                serde_json::json!({
1531                    "timestamp": r.timestamp,
1532                    "value": value_to_serde_json(&r.value),
1533                })
1534            })
1535            .collect();
1536        let body = serde_json::json!({ "device_id": device_id, "history": history });
1537        let client = reqwest::blocking::Client::new();
1538        let mut req = client.post(url.trim()).json(&body);
1539        if let Ok(key) = env::var("IOT_ML_API_KEY") {
1540            req = req.header("Authorization", format!("Bearer {}", key));
1541        }
1542        if let Ok(resp) = req.send() {
1543            if let Ok(json) = resp.json::<serde_json::Value>() {
1544                if let Some(p) = json.get("probability").and_then(|v| v.as_f64()) {
1545                    return Ok(p.clamp(0.0, 1.0));
1546                }
1547            }
1548        }
1549    }
1550
1551    Ok(0.15)
1552}
