1//! Agent skills: built-in skill definitions and tools_description builder.
2//! See docs/AGENT_ASSISTANT_PLAN.md §3.
3//!
4//! Skills are named bundles (description, tools) that define what an agent can do.
5//! Default learning-path skills: development, creative, office, home.
6//! Hard skill: project_init (every agent can init a DAL project).
7
8/// Default skill set for the DAL assistant (smart agent assist).
9/// When an agent has no explicit skills (e.g. no mold or mold has no skills), use these.
10pub const DEFAULT_LEARNING_PATH_SKILLS: &[&str] = &["development", "creative", "office", "home"];
11
12/// Hard skill: always available to every agent. Used when building tools description.
13pub const PROJECT_INIT_SKILL: &str = "project_init";
14
15/// Short description for each built-in skill (for the prompt).
16fn skill_description(name: &str) -> &'static str {
17    match name {
18        "project_init" => "Initialize and set up a DAL project (dal.toml + entry file); run dal init or create layout.",
19        "development" => "Coding, scripts, DAL projects: read/edit files, dal check/run, debug, explain code.",
20        "creative" => "Writing, design, ideation: drafts, outlines, copy, storytelling, tone, brainstorm.",
21        "office" => "Tasks, scheduling, docs, meetings: summarize, action lists, email/note drafts, coordination.",
22        "home" => "Personal tasks, routines, health, family, hobbies: reminders, recipes, how-tos, lists.",
23        _ => "",
24    }
25}
26
27/// Base tools (all agents): reply, run, search, ask_user, dal_init.
28const BASE_TOOLS: &str = "reply, run (shell), search, ask_user, dal_init";
29/// Development skill tools: read_file, write_file, list_dir, dal_check, dal_run.
30const DEVELOPMENT_TOOLS: &str = "read_file, write_file, list_dir, dal_check, dal_run";
31
32/// Build the tools_description string for the agent prompt from the given skill names.
33/// Always includes project_init (hard skill). If skills is empty, uses DEFAULT_LEARNING_PATH_SKILLS.
34/// Development skill adds read_file, write_file, list_dir, dal_check, dal_run.
35pub fn tools_description_for_skills(skills: &[String]) -> String {
36    let mut effective: Vec<String> = skills.iter().map(String::clone).collect();
37    if effective.is_empty() {
38        effective = DEFAULT_LEARNING_PATH_SKILLS
39            .iter()
40            .map(|s| (*s).to_string())
41            .collect();
42    }
43    // Hard skill: project_init always in effect for tools description
44    if !effective.iter().any(|s| s == PROJECT_INIT_SKILL) {
45        effective.insert(0, PROJECT_INIT_SKILL.to_string());
46    }
47
48    let skill_descriptions: Vec<String> = effective
49        .iter()
50        .filter_map(|name| {
51            let d = skill_description(name);
52            if d.is_empty() {
53                None
54            } else {
55                Some(format!("- {}: {}", name, d))
56            }
57        })
58        .collect();
59
60    let skills_block = if skill_descriptions.is_empty() {
61        "You are a helpful assistant.".to_string()
62    } else {
63        format!(
64            "You are a DAL assistant serving your principal. Skills:\n{}",
65            skill_descriptions.join("\n")
66        )
67    };
68
69    let has_development = effective.iter().any(|s| s == "development");
70    let tools_list = if has_development {
71        format!("{}; {}", BASE_TOOLS, DEVELOPMENT_TOOLS)
72    } else {
73        BASE_TOOLS.to_string()
74    };
75
76    let json_tools = if has_development {
77        r#"Reply with JSON: {"action":"reply","text":"..."} or {"action":"run","cmd":"..."} or {"action":"search","query":"..."} or {"action":"ask_user","message":"..."} or {"action":"dal_init","template":"general"|"chain"|"iot"|"agent"} or {"action":"read_file","path":"..."} or {"action":"write_file","path":"...","contents":"..."} or {"action":"list_dir","path":"."} or {"action":"dal_check","path":"file.dal"} or {"action":"dal_run","path":"file.dal"}. Paths are relative to working directory."#
78    } else {
79        r#"Reply with JSON: {"action":"reply","text":"..."} or {"action":"run","cmd":"..."} or {"action":"search","query":"..."} or {"action":"ask_user","message":"..."} or {"action":"dal_init","template":"general"|"chain"|"iot"|"agent"} to initialize a DAL project (omit template for general)."#
80    };
81
82    format!(
83        "{} Available tools: {}. {} When the objective is satisfied, use reply with your final answer. Respect the constraints below.",
84        skills_block, tools_list, json_tools
85    )
86}

88#[cfg(test)]
89mod tests {
90    use super::*;
91
92    #[test]
93    fn default_skills_produce_non_empty_description() {
94        let desc = tools_description_for_skills(&[]);
95        assert!(desc.contains("assistant"));
96        assert!(desc.contains("development"));
97        assert!(desc.contains("creative"));
98        assert!(desc.contains("office"));
99        assert!(desc.contains("home"));
100        assert!(desc.contains("project_init"));
101        assert!(desc.contains("reply"));
102        assert!(desc.contains("ask_user"));
103    }
104
105    #[test]
106    fn explicit_skills_used() {
107        let desc = tools_description_for_skills(&["development".to_string()]);
108        assert!(desc.contains("development"));
109        assert!(desc.contains("project_init"));
110    }
