1//! Fleet: named set of agents, optionally created from a mold (N instances).
2//! Off-chain only. Storage: in-memory; when base path is provided, also persisted to `.dal/fleets.json`.
3//! See COMPREHENSIVE_AGENT_AND_MOLD_PLANS.md §5.
4
5use std::collections::HashMap;
6use std::fs;
7use std::path::{Path, PathBuf};
8use std::sync::{Mutex, OnceLock};
9
10/// A fleet is a named set of agent IDs. Optionally created from a mold (one mold + N instances).
11#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
12pub struct Fleet {
13    pub name: String,
14    /// If created from mold, the mold source (path or ipfs://cid).
15    pub mold_source: Option<String>,
16    /// Agent IDs in this fleet (order preserved).
17    pub member_ids: Vec<String>,
18}
19
20static FLEETS: OnceLock<Mutex<HashMap<String, Fleet>>> = OnceLock::new();
21
22fn fleets() -> std::sync::MutexGuard<'static, HashMap<String, Fleet>> {
23    FLEETS
24        .get_or_init(|| Mutex::new(HashMap::new()))
25        .lock()
26        .unwrap()
27}
28
29fn path_for_fleets(base: &Path) -> PathBuf {
30    base.join(".dal").join("fleets.json")
31}
32
33fn load_from_file(base: &Path) -> HashMap<String, Fleet> {
34    let p = path_for_fleets(base);
35    if let Ok(data) = fs::read_to_string(&p) {
36        if let Ok(map) = serde_json::from_str::<HashMap<String, Fleet>>(&data) {
37            return map;
38        }
39    }
40    HashMap::new()
41}
42
43fn save_to_file(base: &Path, map: &HashMap<String, Fleet>) -> Result<(), String> {
44    let p = path_for_fleets(base);
45    if let Some(parent) = p.parent() {
46        fs::create_dir_all(parent).map_err(|e| format!("create .dal: {}", e))?;
47    }
48    let data = serde_json::to_string_pretty(map).map_err(|e| format!("serialize fleets: {}", e))?;
49    fs::write(&p, data).map_err(|e| format!("write {}: {}", p.display(), e))?;
50    Ok(())
51}
52
53fn ensure_loaded(base: Option<&Path>) {
54    if let Some(b) = base {
55        let loaded = load_from_file(b);
56        let mut f = fleets();
57        *f = loaded;
58    }
59}
60
61fn save_if_base(base: Option<&Path>) -> Result<(), String> {
62    if let Some(b) = base {
63        let f = fleets();
64        save_to_file(b, &f)?;
65    }
66    Ok(())
67}

69/// Create an empty fleet with the given name. Fails if name already exists.
70/// If base is Some, fleet data is persisted to base/.dal/fleets.json.
71pub fn create(name: &str, base: Option<&Path>) -> Result<(), String> {
72    ensure_loaded(base);
73    let name = name.to_string();
74    if name.is_empty() {
75        return Err("fleet name cannot be empty".to_string());
76    }
77    let mut f = fleets();
78    if f.contains_key(&name) {
79        return Err(format!("fleet '{}' already exists", name));
80    }
81    f.insert(
82        name.clone(),
83        Fleet {
84            name,
85            mold_source: None,
86            member_ids: Vec::new(),
87        },
88    );
89    drop(f);
90    save_if_base(base)
91}

93/// Create a fleet by spawning N agents from a mold. Same principal/mold/fallback rules as single create.
94/// Each agent is named `<fleet_name>_0`, `<fleet_name>_1`, ...; params apply to each.
95/// Fleet is persisted to base/.dal/fleets.json.
96pub fn create_from_mold(
97    name: &str,
98    mold_source: &str,
99    count: u32,
100    base: &Path,
101    params: Option<&HashMap<String, String>>,
102) -> Result<(), String> {
103    if count == 0 {
104        return Err("fleet from mold requires count >= 1".to_string());
105    }
106    if count > 1000 {
107        return Err("fleet from mold count capped at 1000".to_string());
108    }
109    ensure_loaded(Some(base));
110    let name = name.to_string();
111    let f = fleets();
112    if f.contains_key(&name) {
113        return Err(format!("fleet '{}' already exists", name));
114    }
115    drop(f);
116
117    let mut member_ids = Vec::with_capacity(count as usize);
118    for i in 0..count {
119        let agent_name = format!("{}_{}", name, i);
120        let ctx = crate::mold::create_from_mold_source(
121            mold_source,
122            base,
123            Some(agent_name.as_str()),
124            params,
125        )
126        .map_err(|e| format!("spawn agent {} from mold: {}", i, e))?;
127        member_ids.push(ctx.agent_id);
128    }
129
130    let mut f = fleets();
131    f.insert(
132        name.clone(),
133        Fleet {
134            name,
135            mold_source: Some(mold_source.to_string()),
136            member_ids,
137        },
138    );
139    drop(f);
140    save_if_base(Some(base))
141}

143/// List fleet names. If base is Some, loads from base/.dal/fleets.json first.
144pub fn list(base: Option<&Path>) -> Vec<String> {
145    ensure_loaded(base);
146    let f = fleets();
147    let mut names: Vec<String> = f.keys().cloned().collect();
148    names.sort();
149    names
150}

152/// Get fleet by name, if it exists. If base is Some, loads from file first.
153pub fn show(name: &str, base: Option<&Path>) -> Option<Fleet> {
154    ensure_loaded(base);
155    let f = fleets();
156    f.get(name).cloned()
157}

159/// Remove a fleet (agents remain in runtime; only the grouping is removed). Returns true if removed.
160/// If base is Some, persists after remove.
161pub fn delete(name: &str, base: Option<&Path>) -> Result<bool, String> {
162    ensure_loaded(base);
163    let mut f = fleets();
164    let removed = f.remove(name).is_some();
165    drop(f);
166    if removed {
167        save_if_base(base)?;
168    }
169    Ok(removed)
170}

172/// Resize fleet to N members. If base is Some, persists after change.
173/// - Scale down (N < current): truncates member list to N (agent contexts are not removed).
174/// - Scale up (N > current): only allowed when fleet has mold_source; spawns (N - current) new agents from the same mold and appends to members.
175/// - Empty fleets (no mold_source) cannot be scaled up.
176pub fn scale(name: &str, n: u32, base: Option<&Path>) -> Result<(), String> {
177    if n > 1000 {
178        return Err("fleet scale count capped at 1000".to_string());
179    }
180    ensure_loaded(base);
181    let base_path = base.ok_or("scale requires a base path for persistence")?;
182
183    let mut f = fleets();
184    let fleet = f
185        .get(name)
186        .cloned()
187        .ok_or_else(|| format!("fleet '{}' not found", name))?;
188    let current = fleet.member_ids.len() as u32;
189
190    if n == current {
191        drop(f);
192        return Ok(());
193    }
194
195    if n < current {
196        let mut fleet = fleet;
197        fleet.member_ids.truncate(n as usize);
198        f.insert(name.to_string(), fleet);
199        drop(f);
200        return save_if_base(Some(base_path));
201    }
202
203    // n > current: scale up
204    let mold_source = fleet.mold_source.as_deref().ok_or_else(|| {
205        format!(
206            "fleet '{}' has no mold; cannot scale up (create from mold first)",
207            name
208        )
209    })?;
210
211    let mut member_ids = fleet.member_ids;
212    drop(f);
213
214    for i in current..n {
215        let agent_name = format!("{}_{}", name, i);
216        let ctx = crate::mold::create_from_mold_source(
217            mold_source,
218            base_path,
219            Some(agent_name.as_str()),
220            None,
221        )
222        .map_err(|e| format!("spawn agent {} from mold: {}", i, e))?;
223        member_ids.push(ctx.agent_id);
224    }
225
226    let mut f = fleets();
227    if let Some(fleet) = f.get_mut(name) {
228        fleet.member_ids = member_ids;
229    }
230    drop(f);
231    save_if_base(Some(base_path))
232}

238    #[test]
239    fn create_and_list_empty_fleet() {
240        let name = format!(
241            "test_fleet_{}",
242            std::time::SystemTime::now()
243                .duration_since(std::time::UNIX_EPOCH)
244                .unwrap()
245                .as_millis()
246        );
247        assert!(create(&name, None).is_ok());
248        let names = list(None);
249        assert!(names.contains(&name));
250        assert!(show(&name, None).is_some());
251        assert_eq!(show(&name, None).unwrap().member_ids.len(), 0);
252        let _ = delete(&name, None);
253    }
254
255    #[test]
