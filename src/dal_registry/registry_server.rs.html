1//! DAL package registry server. Production-ready: auth, file storage, index API.
2//!
3//! Endpoints:
4//! - GET /v1/packages/:name     -> index JSON (versions + url per version)
5//! - GET /v1/packages/:name/versions/:version/tarball -> tarball bytes
6//! - PUT /v1/packages/:name/versions/:version -> publish (body = tarball, Authorization: Bearer <token>)
7//!
8//! Env: REGISTRY_STORAGE_PATH, REGISTRY_TOKEN (auth), REGISTRY_PUBLIC_URL (for index urls), PORT.
9
10use axum::{
11    body::Body,
12    extract::{Path, State},
13    http::{Request, StatusCode},
14    response::{IntoResponse, Json, Response},
15    routing::{get, put},
16    Router,
17};
18use serde_json::json;
19use std::path::PathBuf;
20use std::sync::Arc;
21use tower_http::cors::CorsLayer;

23fn storage_path() -> PathBuf {
24    std::env::var("REGISTRY_STORAGE_PATH")
25        .map(PathBuf::from)
26        .unwrap_or_else(|_| PathBuf::from("./registry-storage"))
27}
28
29fn auth_token() -> Option<String> {
30    std::env::var("REGISTRY_TOKEN")
31        .or_else(|_| std::env::var("DAL_REGISTRY_TOKEN"))
32        .ok()
33}
34
35fn public_url() -> String {
36    std::env::var("REGISTRY_PUBLIC_URL").unwrap_or_else(|_| "http://localhost:8787".to_string())
37}

39#[derive(Clone)]
40struct AppState {
41    storage: PathBuf,
42    public_url: String,
43    auth_token: Option<String>,
44}

46fn sanitize_name(s: &str) -> bool {
47    !s.is_empty()
48        && !s.contains("..")
49        && s.chars().all(|c| {
50            c.is_alphanumeric() || c == '/' || c == '@' || c == '-' || c == '_' || c == '.'
51        })
52}
53
54fn sanitize_version(s: &str) -> bool {
55    !s.is_empty()
56        && !s.contains("..")
57        && s.chars()
58            .all(|c| c.is_alphanumeric() || c == '.' || c == '-' || c == '_')
59}

61/// Verify path is under storage root (prevents path traversal).
62fn path_under_storage(path: &std::path::Path, storage: &std::path::Path) -> bool {
63    let storage_canon = match storage.canonicalize() {
64        Ok(p) => p,
65        Err(_) => return false,
66    };
67    if let Ok(path_canon) = path.canonicalize() {
68        return path_canon.starts_with(&storage_canon);
69    }
70    // Path may not exist yet; walk up to first existing ancestor and verify it's under storage
71    let mut current = path;
72    while let Some(parent) = current.parent() {
73        if let Ok(parent_canon) = parent.canonicalize() {
74            return parent_canon.starts_with(&storage_canon);
75        }
76        current = parent;
77    }
78    false
79}

81/// Map package name to filesystem-safe dir name (e.g. @dal/testing -> _at_dal_slash_testing).
82fn name_to_storage_dir(name: &str) -> String {
83    name.replace('@', "_at_").replace('/', "_slash_")
84}

86async fn get_package_index(
87    State(state): State<Arc<AppState>>,
88    Path(name): Path<String>,
89) -> Response {
90    if !sanitize_name(&name) {
91        return (StatusCode::BAD_REQUEST, "invalid package name").into_response();
92    }
93    let dir = name_to_storage_dir(&name);
94    let index_path = state.storage.join("packages").join(dir).join("index.json");
95    if !path_under_storage(&index_path, &state.storage) {
96        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
97    }
98    match tokio::fs::read_to_string(&index_path).await {
99        Ok(s) => {
100            let v: serde_json::Value = match serde_json::from_str(&s) {
101                Ok(j) => j,
102                Err(_) => {
103                    return (StatusCode::INTERNAL_SERVER_ERROR, "invalid index.json")
104                        .into_response()
105                }
106            };
107            (StatusCode::OK, Json(v)).into_response()
108        }
109        Err(e) if e.kind() == std::io::ErrorKind::NotFound => (
110            StatusCode::NOT_FOUND,
111            format!("package not found: {}", name),
112        )
113            .into_response(),
114        Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "read error").into_response(),
115    }
116}

118async fn get_tarball(
119    State(state): State<Arc<AppState>>,
120    Path((name, version)): Path<(String, String)>,
121) -> Response {
122    if !sanitize_name(&name) || !sanitize_version(&version) {
123        return (StatusCode::BAD_REQUEST, "invalid name or version").into_response();
124    }
125    let dir = name_to_storage_dir(&name);
126    let tgz_path = state
127        .storage
128        .join("packages")
129        .join(dir)
130        .join("versions")
131        .join(format!("{}.tgz", version));
132    if !path_under_storage(&tgz_path, &state.storage) {
133        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
134    }
135    match tokio::fs::read(&tgz_path).await {
136        Ok(bytes) => (
137            StatusCode::OK,
138            [("Content-Type", "application/gzip")],
139            bytes,
140        )
141            .into_response(),
142        Err(e) if e.kind() == std::io::ErrorKind::NotFound => (
143            StatusCode::NOT_FOUND,
144            format!("version not found: {}@{}", name, version),
145        )
146            .into_response(),
147        Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "read error").into_response(),
148    }
149}

151async fn put_version(
152    State(state): State<Arc<AppState>>,
153    Path((name, version)): Path<(String, String)>,
154    request: Request<Body>,
155) -> Response {
156    if !sanitize_name(&name) || !sanitize_version(&version) {
157        return (StatusCode::BAD_REQUEST, "invalid name or version").into_response();
158    }
159    if let Some(ref token) = state.auth_token {
160        let auth = request
161            .headers()
162            .get("Authorization")
163            .and_then(|v| v.to_str().ok())
164            .unwrap_or("");
165        if !auth.starts_with("Bearer ")
166            || auth.strip_prefix("Bearer ").unwrap_or("") != token.as_str()
167        {
168            return (StatusCode::UNAUTHORIZED, "invalid or missing token").into_response();
169        }
170    }
171    const MAX_TARBALL: usize = 50 * 1024 * 1024; // 50 MiB
172    let body = match axum::body::to_bytes(request.into_body(), MAX_TARBALL).await {
173        Ok(b) => b,
174        Err(_) => return (StatusCode::BAD_REQUEST, "body error").into_response(),
175    };
176    let dir = name_to_storage_dir(&name);
177    let pkg_dir = state.storage.join("packages").join(dir);
178    let versions_dir = pkg_dir.join("versions");
179    if !path_under_storage(&versions_dir, &state.storage) {
180        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
181    }
182    if let Err(e) = tokio::fs::create_dir_all(&versions_dir).await {
183        return (
184            StatusCode::INTERNAL_SERVER_ERROR,
185            format!("create dir: {}", e),
186        )
187            .into_response();
188    }
189    let tgz_path = versions_dir.join(format!("{}.tgz", version));
190    if !path_under_storage(&tgz_path, &state.storage) {
191        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
192    }
193    if let Err(e) = tokio::fs::write(&tgz_path, body.as_ref()).await {
194        return (StatusCode::INTERNAL_SERVER_ERROR, format!("write: {}", e)).into_response();
195    }
196    let index_path = pkg_dir.join("index.json");
197    if !path_under_storage(&index_path, &state.storage) {
198        return (StatusCode::BAD_REQUEST, "invalid path").into_response();
199    }
200    let base = state.public_url.trim_end_matches('/');
201    let url = format!("{}/v1/packages/{}/versions/{}/tarball", base, name, version);
202    let mut index: serde_json::Map<String, serde_json::Value> =
203        match tokio::fs::read_to_string(&index_path).await {
204            Ok(s) => serde_json::from_str(&s).unwrap_or_else(|_| {
205                let mut m = serde_json::Map::new();
206                m.insert("name".to_string(), json!(&name));
207                m.insert("versions".to_string(), json!({}));
208                m
209            }),
210            Err(_) => {
211                let mut m = serde_json::Map::new();
212                m.insert("name".to_string(), json!(&name));
213                m.insert("versions".to_string(), json!({}));
214                m
215            }
216        };
217    let versions = index
218        .get_mut("versions")
219        .and_then(|v| v.as_object_mut())
220        .expect("versions object");
221    versions.insert(version.clone(), json!({ "url": url }));
222    let index_bytes = serde_json::to_vec_pretty(&index).expect("json");
223    if let Err(e) = tokio::fs::write(&index_path, index_bytes).await {
224        return (
225            StatusCode::INTERNAL_SERVER_ERROR,
226            format!("write index: {}", e),
227        )
228            .into_response();
229    }
230    (
231        StatusCode::CREATED,
232        Json(json!({ "ok": true, "name": name, "version": version })),
233    )
234        .into_response()
235}

237#[tokio::main]
238async fn main() {
239    let storage = storage_path();
240    let public_url = public_url();
241    let auth_token = auth_token();
242    if auth_token.is_none() {
243        eprintln!(
244            "⚠️  REGISTRY_TOKEN (or DAL_REGISTRY_TOKEN) not set; publish is unauthenticated."
245        );
246    }
247    if let Err(e) = std::fs::create_dir_all(&storage) {
248        eprintln!("Failed to create storage dir {:?}: {}", storage, e);
249        std::process::exit(1);
250    }
251    let state = Arc::new(AppState {
252        storage,
253        public_url,
254        auth_token,
255    });
256    let app = Router::new()
257        .route("/v1/packages/:name", get(get_package_index))
258        .route(
259            "/v1/packages/:name/versions/:version/tarball",
260            get(get_tarball),
261        )
262        .route("/v1/packages/:name/versions/:version", put(put_version))
263        .layer(CorsLayer::permissive())
264        .with_state(state);
265    let port: u16 = std::env::var("PORT")
266        .ok()
267        .and_then(|p| p.parse().ok())
268        .unwrap_or(8787);
269    let addr = std::net::SocketAddr::from(([0, 0, 0, 0], port));
270    println!(
271        "DAL registry listening on http://{} (storage: {:?})",
272        addr,
273        storage_path()
274    );
275    let listener = tokio::net::TcpListener::bind(addr).await.expect("bind");
276    axum::serve(listener, app).await.expect("serve");
277}
